041/**
042 * <p>
043 * Defines an email address associated with the person.
044 * </p>
045 * 
046 * <p>
047 * <b>Code sample</b>
048 * </p>
049 * 
050 * <pre class="brush:java">
051 * VCard vcard = new VCard();
052 * 
053 * Email email = new Email(&quot;johndoe@hotmail.com&quot;);
054 * email.addType(EmailType.HOME);
055 * vcard.addEmail(email);
056 * 
057 * email = new Email(&quot;jdoe@company.com&quot;);
058 * email.addType(EmailType.WORK);
059 * email.setPref(1); //the most preferred email
060 * vcard.addEmail(email);
061 * </pre>
062 * 
063 * <p>
064 * <b>Property name:</b> {@code EMAIL}
065 * </p>
066 * <p>
067 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
068 * </p>
069 * @author Michael Angstadt
070 */
071public class Email extends TextProperty implements HasAltId {
072        /**
073         * Creates an email property.
074         * @param email the email (e.g. "johndoe@example.com")
075         */
076        public Email(String email) {
077                super(email);
078        }
079
080        /**
081         * Gets all the TYPE parameters.
082         * @return the TYPE parameters or empty set if there are none
083         */
084        public Set<EmailType> getTypes() {
085                Set<String> values = parameters.getTypes();
086                Set<EmailType> types = new HashSet<EmailType>(values.size());
087                for (String value : values) {
088                        types.add(EmailType.get(value));
089                }
090                return types;
091        }
092
093        /**
094         * Adds a TYPE parameter.
095         * @param type the TYPE parameter to add
096         */
097        public void addType(EmailType type) {
098                parameters.addType(type.getValue());
099        }
100
101        /**
102         * Removes a TYPE parameter.
103         * @param type the TYPE parameter to remove
104         */
105        public void removeType(EmailType type) {
106                parameters.removeType(type.getValue());
107        }
108
109        @Override
110        public List<Integer[]> getPids() {
111                return super.getPids();
112        }
113
114        @Override
115        public void addPid(int localId, int clientPidMapRef) {
116                super.addPid(localId, clientPidMapRef);
117        }
118
119        @Override
120        public void removePids() {
121                super.removePids();
122        }
123
124        @Override
125        public Integer getPref() {
126                return super.getPref();
127        }
128
129        @Override
130        public void setPref(Integer pref) {
131                super.setPref(pref);
132        }
133
134        //@Override
135        public String getAltId() {
136                return parameters.getAltId();
137        }
138
139        //@Override
140        public void setAltId(String altId) {
141                parameters.setAltId(altId);
142        }
143
144        @Override
145        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
146                super._validate(warnings, version, vcard);
147
148                for (EmailType type : getTypes()) {
149                        if (type == EmailType.PREF) {
150                                //ignore because it is converted to a PREF parameter for 4.0 vCards
151                                continue;
152                        }
153                        if (!type.isSupported(version)) {
154                                warnings.add(new Warning(9, type.getValue()));
155                        }
156                }
157        }
158}
