041/**
042 * <p>
043 * Defines the members that make up the group. This property can only be used if
044 * the vCard's {@link Kind} property is set to "group".
045 * </p>
046 * 
047 * <p>
048 * <b>Code sample</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * VCard vcard = new VCard();
053 * 
054 * //kind property must be set to &quot;group&quot; in order to add members
055 * vcard.setKind(Kind.group());
056 * 
057 * //static factory methods
058 * Member member = Member.email(&quot;johndoe@hotmail.com&quot;);
059 * vcard.addMember(member);
060 * 
061 * //reference another vCard by putting its UID property here
062 * member = new Member(&quot;urn:uuid:03a0e51f-d1aa-4385-8a53-e29025acd8af&quot;);
063 * vcard.addMember(member);
064 * </pre>
065 * 
066 * <p>
067 * <b>Property name:</b> {@code MEMBER}
068 * </p>
069 * <p>
070 * <b>Supported versions:</b> {@code 4.0}
071 * </p>
072 * @author Michael Angstadt
073 */
074public class Member extends UriProperty implements HasAltId {
075        /**
076         * Creates a member property.
077         * @param uri the URI representing the member
078         */
079        public Member(String uri) {
080                super(uri);
081        }
082
083        /**
084         * Creates a member property whose value is an email address.
085         * @param email the email address
086         * @return the property
087         */
088        public static Member email(String email) {
089                return new Member("mailto:" + email);
090        }
091
092        /**
093         * Creates a member property whose value is an instant messenger handle.
094         * @param protocol the instant messenger protocol (e.g. "aim")
095         * @param handle the instant messenger handle (e.g. "johndoe")
096         * @return the property
097         */
098        public static Member im(String protocol, String handle) {
099                return new Member(protocol + ":" + handle);
100        }
101
102        /**
103         * Creates a member property whose value is a telephone number.
104         * @param telUri the telephone number
105         * @return the property
106         */
107        public static Member telephone(TelUri telUri) {
108                return new Member(telUri.toString());
109        }
110
111        @Override
112        public Set<VCardVersion> _supportedVersions() {
113                return EnumSet.of(VCardVersion.V4_0);
114        }
115
116        /**
117         * Gets the URI value.
118         * @return the URI value or null if no URI value is set
119         */
120        public String getUri() {
121                return getValue();
122        }
123
124        /**
125         * Sets the URI.
126         * @param uri the URI
127         */
128        public void setUri(String uri) {
129                setValue(uri);
130        }
131
132        @Override
133        public List<Integer[]> getPids() {
134                return super.getPids();
135        }
136
137        @Override
138        public void addPid(int localId, int clientPidMapRef) {
139                super.addPid(localId, clientPidMapRef);
140        }
141
142        @Override
143        public void removePids() {
144                super.removePids();
145        }
146
147        @Override
148        public Integer getPref() {
149                return super.getPref();
150        }
151
152        @Override
153        public void setPref(Integer pref) {
154                super.setPref(pref);
155        }
156
157        //@Override
158        public String getAltId() {
159                return parameters.getAltId();
160        }
161
162        //@Override
163        public void setAltId(String altId) {
164                parameters.setAltId(altId);
165        }
166
167        /**
168         * Gets the MEDIATYPE parameter.
169         * <p>
170         * <b>Supported versions:</b> {@code 4.0}
171         * </p>
172         * @return the media type or null if not set
173         */
174        public String getMediaType() {
175                return parameters.getMediaType();
176        }
177
178        /**
179         * Sets the MEDIATYPE parameter.
180         * <p>
181         * <b>Supported versions:</b> {@code 4.0}
182         * </p>
183         * @param mediaType the media type or null to remove
184         */
185        public void setMediaType(String mediaType) {
186                parameters.setMediaType(mediaType);
187        }
188
189        @Override
190        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
191                super._validate(warnings, version, vcard);
192
193                if (vcard.getKind() == null || !vcard.getKind().isGroup()) {
194                        warnings.add(new Warning(17));
195                }
196        }
197}
