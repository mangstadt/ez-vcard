046/**
047 * Parses the components out of each line in a plain-text vCard data stream.
048 * @author Michael Angstadt
049 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
050 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
051 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
052 */
053public class VCardRawReader implements Closeable {
054        private final Reader reader;
055        private final Buffer buffer = new Buffer();
056        private final Buffer unfoldedLine = new Buffer();
057
058        private boolean eos = false;
059        private boolean caretDecodingEnabled = true;
060        private VCardVersion version = VCardVersion.V2_1; //initialize to 2.1, since the VERSION property can exist anywhere in the file in this version
061        private int prevChar = -1;
062        private int propertyLineNum = 1;
063        private int lineNum = 1;
064
065        /**
066         * @param reader the reader to read from
067         */
068        public VCardRawReader(Reader reader) {
069                this.reader = reader;
070        }
071
072        /**
073         * Gets the line number of the line that was just read. If the line was
074         * folded, this will be the line number of the first line.
075         * @return the line number
076         */
077        public int getLineNumber() {
078                return propertyLineNum;
079        }
080
081        /**
082         * Gets the vCard version that the reader is currently parsing with.
083         * @return the vCard version
084         */
085        public VCardVersion getVersion() {
086                return version;
087        }

089        /**
090         * Reads the next line from the input stream. Folded lines are automatically
091         * unfolded.
092         * @return the parsed components of the line
093         * @throws VCardParseException if a line cannot be parsed
094         * @throws IOException if there's a problem reading from the stream
095         */
096        public VCardRawLine readLine() throws IOException {
097                if (eos) {
098                        return null;
099                }
100
101                propertyLineNum = lineNum;
102                buffer.clear();
103                unfoldedLine.clear();
104
105                /*
106                 * The property's group.
107                 */
108                String group = null;
109
110                /*
111                 * The property's name.
112                 */
113                String propertyName = null;
114
115                /*
116                 * The name of the parameter we're currently inside of.
117                 */
118                String curParamName = null;
119
120                /*
121                 * The property's parameters.
122                 */
123                VCardParameters parameters = new VCardParameters();
124
125                /*
126                 * The character that used to escape the current character (for
127                 * parameter values).
128                 */
129                char escapeChar = 0;
130
131                /*
132                 * Are we currently inside a parameter value that is surrounded with
133                 * double-quotes?
134                 */
135                boolean inQuotes = false;
136
137                /*
138                 * Are we currently inside the property value?
139                 */
140                boolean inValue = false;
141
142                /*
143                 * Does the line use quoted-printable encoding, and does it end all of
144                 * its folded lines with a "=" character?
145                 */
146                boolean quotedPrintableLine = false;
147
148                /*
149                 * The current character.
150                 */
151                char ch = 0;
152
153                /*
154                 * The previous character.
155                 */
156                char prevChar;
157
158                while (true) {
159                        prevChar = ch;
160
161                        int read = nextChar();
162                        if (read < 0) {
163                                eos = true;
164                                break;
165                        }
166
167                        ch = (char) read;
168
169                        if (prevChar == '\r' && ch == '\n') {
170                                /*
171                                 * The newline was already processed when the "\r" character was
172                                 * encountered, so ignore the accompanying "\n" character.
173                                 */
174                                continue;
175                        }
176
177                        if (isNewline(ch)) {
178                                quotedPrintableLine = (inValue && prevChar == '=' && isQuotedPrintable(parameters));
179                                if (quotedPrintableLine) {
180                                        /*
181                                         * Remove the "=" character that some vCards put at the end
182                                         * of quoted-printable lines that are followed by a folded
183                                         * line.
184                                         */
185                                        buffer.chop();
186                                        unfoldedLine.chop();
187                                }
188
189                                //keep track of the current line number
190                                lineNum++;
191
192                                continue;
193                        }
194
195                        if (isNewline(prevChar)) {
196                                if (isWhitespace(ch)) {
197                                        /*
198                                         * This line is a continuation of the previous line (the
199                                         * line is folded).
200                                         */
201                                        continue;
202                                }
203
204                                if (quotedPrintableLine) {
205                                        /*
206                                         * The property's parameters indicate that the property
207                                         * value is quoted-printable. And the previous line ended
208                                         * with an equals sign. This means that folding whitespace
209                                         * may not be prepended to folded lines like it should...
210                                         */
211                                } else {
212                                        /*
213                                         * We're reached the end of the property.
214                                         */
215                                        this.prevChar = ch;
216                                        break;
217                                }
218                        }
219
220                        unfoldedLine.append(ch);
221
222                        if (inValue) {
223                                buffer.append(ch);
224                                continue;
225                        }
226
227                        if (escapeChar != 0) {
228                                //this character was escaped
229                                if (escapeChar == '\\') {
230                                        if (ch == '\\') {
231                                                buffer.append(ch);
232                                        } else if (ch == 'n' || ch == 'N') {
233                                                //newlines appear as "\n" or "\N" (see RFC 2426 p.7)
234                                                buffer.append(NEWLINE);
235                                        } else if (ch == '"' && version != VCardVersion.V2_1) {
236                                                //double quotes don't need to be escaped in 2.1 parameter values because they have no special meaning
237                                                buffer.append(ch);
238                                        } else if (ch == ';' && version == VCardVersion.V2_1) {
239                                                //semi-colons can only be escaped in 2.1 parameter values (see section 2 of specs)
240                                                //if a 3.0/4.0 param value has semi-colons, the value should be surrounded in double quotes
241                                                buffer.append(ch);
242                                        } else {
243                                                //treat the escape character as a normal character because it's not a valid escape sequence
244                                                buffer.append(escapeChar).append(ch);
245                                        }
246                                } else if (escapeChar == '^') {
247                                        if (ch == '^') {
248                                                buffer.append(ch);
249                                        } else if (ch == 'n') {
250                                                buffer.append(NEWLINE);
251                                        } else if (ch == '\'') {
252                                                buffer.append('"');
253                                        } else {
254                                                //treat the escape character as a normal character because it's not a valid escape sequence
255                                                buffer.append(escapeChar).append(ch);
256                                        }
257                                }
258                                escapeChar = 0;
259                                continue;
260                        }
261
262                        if (ch == '\\' || (ch == '^' && version != VCardVersion.V2_1 && caretDecodingEnabled)) {
263                                //an escape character was read
264                                escapeChar = ch;
265                                continue;
266                        }
267
268                        if (ch == '.' && group == null && propertyName == null) {
269                                //set the group
270                                group = buffer.getAndClear();
271                                continue;
272                        }
273
274                        if ((ch == ';' || ch == ':') && !inQuotes) {
275                                if (propertyName == null) {
276                                        //property name
277                                        propertyName = buffer.getAndClear();
278                                } else {
279                                        //parameter value
280                                        String paramValue = buffer.getAndClear();
281                                        if (version == VCardVersion.V2_1) {
282                                                //2.1 allows whitespace to surround the "=", so remove it
283                                                paramValue = StringUtils.ltrim(paramValue);
284                                        }
285                                        parameters.put(curParamName, paramValue);
286                                        curParamName = null;
287                                }
288
289                                if (ch == ':') {
290                                        //the rest of the line is the property value
291                                        inValue = true;
292                                }
293                                continue;
294                        }
295
296                        if (ch == ',' && !inQuotes && version != VCardVersion.V2_1) {
297                                //multi-valued parameter
298                                parameters.put(curParamName, buffer.getAndClear());
299                                continue;
300                        }
301
302                        if (ch == '=' && curParamName == null) {
303                                //parameter name
304                                String paramName = buffer.getAndClear();
305                                if (version == VCardVersion.V2_1) {
306                                        //2.1 allows whitespace to surround the "=", so remove it
307                                        paramName = StringUtils.rtrim(paramName);
308                                }
309                                curParamName = paramName;
310                                continue;
311                        }
312
313                        if (ch == '"' && version != VCardVersion.V2_1) {
314                                //2.1 doesn't use the quoting mechanism
315                                inQuotes = !inQuotes;
316                                continue;
317                        }
318
319                        buffer.append(ch);
320                }
321
322                if (unfoldedLine.length() == 0) {
323                        //input stream was empty
324                        return null;
325                }
326
327                if (propertyName == null) {
328                        throw new VCardParseException(unfoldedLine.get(), propertyLineNum);
329                }
330
331                String value = buffer.getAndClear();
332
333                if ("VERSION".equalsIgnoreCase(propertyName)) {
334                        VCardVersion version = VCardVersion.valueOfByStr(value);
335                        if (version == null) {
336                                throw new InvalidVersionException(value, unfoldedLine.get(), propertyLineNum);
337                        }
338                        this.version = version;
339                }
340
341                return new VCardRawLine(group, propertyName, parameters, value);
342        }

344        /**
345         * <p>
346         * Gets whether the reader will decode parameter values that use circumflex
347         * accent encoding (enabled by default). This escaping mechanism allows
348         * newlines and double quotes to be included in parameter values.
349         * </p>
350         * 
351         * <table class="simpleTable">
352         * <tr>
353         * <th>Raw Character</th>
354         * <th>Encoded Character</th>
355         * </tr>
356         * <tr>
357         * <td>{@code "}</td>
358         * <td>{@code ^'}</td>
359         * </tr>
360         * <tr>
361         * <td><i>newline</i></td>
362         * <td>{@code ^n}</td>
363         * </tr>
364         * <tr>
365         * <td>{@code ^}</td>
366         * <td>{@code ^^}</td>
367         * </tr>
368         * </table>
369         * 
370         * <p>
371         * Example:
372         * </p>
373         * 
374         * <pre>
375         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
376         *  sburgh, PA 15212":40.446816;80.00566
377         * </pre>
378         * 
379         * @return true if circumflex accent decoding is enabled, false if not
380         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
381         */
382        public boolean isCaretDecodingEnabled() {
383                return caretDecodingEnabled;
384        }
385
386        /**
387         * <p>
388         * Sets whether the reader will decode parameter values that use circumflex
389         * accent encoding (enabled by default). This escaping mechanism allows
390         * newlines and double quotes to be included in parameter values.
391         * </p>
392         * 
393         * <table class="simpleTable">
394         * <tr>
395         * <th>Raw Character</th>
396         * <th>Encoded Character</th>
397         * </tr>
398         * <tr>
399         * <td>{@code "}</td>
400         * <td>{@code ^'}</td>
401         * </tr>
402         * <tr>
403         * <td><i>newline</i></td>
404         * <td>{@code ^n}</td>
405         * </tr>
406         * <tr>
407         * <td>{@code ^}</td>
408         * <td>{@code ^^}</td>
409         * </tr>
410         * </table>
411         * 
412         * <p>
413         * Example:
414         * </p>
415         * 
416         * <pre>
417         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
418         *  sburgh, PA 15212":geo:40.446816,-80.00566
419         * </pre>
420         * 
421         * @param enable true to use circumflex accent decoding, false not to
422         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
423         */
424        public void setCaretDecodingEnabled(boolean enable) {
425                caretDecodingEnabled = enable;
426        }

428        /**
429         * Gets the character encoding of the reader.
430         * @return the character encoding or null if none is defined
431         */
432        public Charset getEncoding() {
433                if (reader instanceof InputStreamReader) {
434                        InputStreamReader isr = (InputStreamReader) reader;
435                        String charsetStr = isr.getEncoding();
436                        return (charsetStr == null) ? null : Charset.forName(charsetStr);
437                }
438
439                return null;
440        }
441
442        private int nextChar() throws IOException {
443                if (prevChar >= 0) {
444                        /*
445                         * Use the character that was left over from the previous invocation
446                         * of "readLine()".
447                         */
448                        int ch = prevChar;
449                        prevChar = -1;
450                        return ch;
451                }
452
453                return reader.read();
454        }
455
456        private boolean isNewline(char ch) {
457                return ch == '\n' || ch == '\r';
458        }
459
460        private boolean isWhitespace(char ch) {
461                return ch == ' ' || ch == '\t';
462        }
463
464        /**
465         * Determines if the property value is quoted-printable. This must be
466         * checked in order to account for the fact that some vCards fold
467         * quoted-printed lines in a non-standard way.
468         * @param parameters the property's parameters
469         * @return true if the property is quoted-printable, false if not
470         */
471        private boolean isQuotedPrintable(VCardParameters parameters) {
472                if (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE) {
473                        return true;
474                }
475
476                List<String> namelessValues = parameters.get(null);
477                for (String value : namelessValues) {
478                        if ("QUOTED-PRINTABLE".equalsIgnoreCase(value)) {
479                                return true;
480                        }
481                }
482
483                return false;
484        }

486        /**
487         * Wraps a {@link StringBuilder} object, providing utility methods.
488         */
489        private static class Buffer {
490                private final StringBuilder sb = new StringBuilder();
491
492                /**
493                 * Clears the buffer.
494                 * @return this
495                 */
496                public Buffer clear() {
497                        sb.setLength(0);
498                        return this;
499                }
500
501                /**
502                 * Gets the buffer's contents.
503                 * @return the buffer's contents
504                 */
505                public String get() {
506                        return sb.toString();
507                }
508
509                /**
510                 * Gets the buffer's contents, then clears it.
511                 * @return the buffer's contents
512                 */
513                public String getAndClear() {
514                        String string = get();
515                        clear();
516                        return string;
517                }
518
519                /**
520                 * Appends a character to the buffer.
521                 * @param ch the character to append
522                 * @return this
523                 */
524                public Buffer append(char ch) {
525                        sb.append(ch);
526                        return this;
527                }
528
529                /**
530                 * Appends a character sequence to the buffer.
531                 * @param string the character sequence to append
532                 * @return this
533                 */
534                public Buffer append(CharSequence string) {
535                        sb.append(string);
536                        return this;
537                }
538
539                /**
540                 * Removes the last character from the buffer.
541                 * @return this
542                 */
543                public Buffer chop() {
544                        sb.setLength(sb.length() - 1);
545                        return this;
546                }
547
548                /**
549                 * Gets the length of the buffer.
550                 * @return the buffer's length
551                 */
552                public int length() {
553                        return sb.length();
554                }
555        }
556
557        /**
558         * Closes the underlying {@link Reader} object.
559         */
560        public void close() throws IOException {
561                reader.close();
562        }
