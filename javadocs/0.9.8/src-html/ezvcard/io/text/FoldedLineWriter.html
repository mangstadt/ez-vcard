040/**
041 * Automatically folds lines as they are written.
042 * @author Michael Angstadt
043 */
044public class FoldedLineWriter extends Writer {
045        private final Writer writer;
046        private int curLineLength = 0;
047        private Integer lineLength = 75;
048        private String indent = " ";
049        private String newline = "\r\n";
050
051        /**
052         * Creates a folded line writer.
053         * @param writer the writer object to wrap
054         */
055        public FoldedLineWriter(Writer writer) {
056                this.writer = writer;
057        }
058
059        /**
060         * Writes a string, followed by a newline.
061         * @param str the text to write
062         * @throws IOException if there's a problem writing to the output stream
063         */
064        public void writeln(String str) throws IOException {
065                write(str);
066                write(newline);
067        }
068
069        /**
070         * Writes a string.
071         * @param str the string to write
072         * @param quotedPrintable true to encode the string in quoted-printable
073         * encoding, false not to
074         * @param charset the character set to use when encoding into
075         * quoted-printable, or null to use the writer's character encoding (only
076         * applicable if "quotedPrintable" is set to true)
077         * @return this
078         * @throws IOException if there's a problem writing to the output stream
079         */
080        public FoldedLineWriter append(CharSequence str, boolean quotedPrintable, Charset charset) throws IOException {
081                write(str, quotedPrintable, charset);
082                return this;
083        }
084
085        /**
086         * Writes a string.
087         * @param str the string to write
088         * @param quotedPrintable true to encode the string in quoted-printable
089         * encoding, false not to
090         * @param charset the character set to use when encoding into
091         * quoted-printable, or null to use the writer's character encoding (only
092         * applicable if "quotedPrintable" is set to true)
093         * @throws IOException if there's a problem writing to the output stream
094         */
095        public void write(CharSequence str, boolean quotedPrintable, Charset charset) throws IOException {
096                write(str.toString().toCharArray(), 0, str.length(), quotedPrintable, charset);
097        }
098
099        @Override
100        public void write(char[] cbuf, int off, int len) throws IOException {
101                write(cbuf, off, len, false, null);
102        }
103
104        /**
105         * Writes a portion of an array of characters.
106         * @param cbuf the array of characters
107         * @param off the offset from which to start writing characters
108         * @param len the number of characters to write
109         * @param quotedPrintable true to encode the string in quoted-printable
110         * encoding, false not to
111         * @param charset the character set to use when encoding into
112         * quoted-printable, or null to use the writer's character encoding (only
113         * applicable if "quotedPrintable" is set to true)
114         * @throws IOException if there's a problem writing to the output stream
115         */
116        public void write(char[] cbuf, int off, int len, boolean quotedPrintable, Charset charset) throws IOException {
117                if (quotedPrintable) {
118                        if (charset == null) {
119                                charset = Charset.forName("UTF-8");
120                        }
121
122                        QuotedPrintableCodec codec = new QuotedPrintableCodec(charset.name());
123                        try {
124                                String str = new String(cbuf, off, len);
125                                String encoded = codec.encode(str);
126
127                                cbuf = encoded.toCharArray();
128                                off = 0;
129                                len = cbuf.length;
130                        } catch (EncoderException e) {
131                                /*
132                                 * Thrown if an unsupported charset is passed into the codec.
133                                 * This should never be thrown though, because we already know
134                                 * the charset is valid (a Charset object is passed into the
135                                 * method).
136                                 */
137                                throw new RuntimeException(e);
138                        }
139                }
140
141                if (lineLength == null) {
142                        /*
143                         * If line folding is disabled, then write directly to the Writer.
144                         */
145                        writer.write(cbuf, off, len);
146                        return;
147                }
148
149                int effectiveLineLength = lineLength;
150                if (quotedPrintable) {
151                        /*
152                         * Account for the "=" character that must be appended onto each
153                         * line.
154                         */
155                        effectiveLineLength -= 1;
156                }
157
158                int encodedCharPos = -1;
159                int start = off;
160                int end = off + len;
161                for (int i = start; i < end; i++) {
162                        char c = cbuf[i];
163
164                        /*
165                         * Keep track of the quoted-printable characters to prevent them
166                         * from being cut in two at a folding boundary.
167                         */
168                        if (encodedCharPos >= 0) {
169                                encodedCharPos++;
170                                if (encodedCharPos == 3) {
171                                        encodedCharPos = -1;
172                                }
173                        }
174
175                        if (c == '\n') {
176                                writer.write(cbuf, start, i - start + 1);
177                                curLineLength = 0;
178                                start = i + 1;
179                                continue;
180                        }
181
182                        if (c == '\r') {
183                                if (i == end - 1 || cbuf[i + 1] != '\n') {
184                                        writer.write(cbuf, start, i - start + 1);
185                                        curLineLength = 0;
186                                        start = i + 1;
187                                } else {
188                                        curLineLength++;
189                                }
190                                continue;
191                        }
192
193                        if (c == '=' && quotedPrintable) {
194                                encodedCharPos = 0;
195                        }
196
197                        if (curLineLength >= effectiveLineLength) {
198                                /*
199                                 * If the last characters on the line are whitespace, then
200                                 * exceed the max line length in order to include the whitespace
201                                 * on the same line. Otherwise, the whitespace will be lost
202                                 * because it will merge with the padding on the next, folded
203                                 * line.
204                                 */
205                                if (Character.isWhitespace(c)) {
206                                        while (Character.isWhitespace(c) && i < end - 1) {
207                                                i++;
208                                                c = cbuf[i];
209                                        }
210                                        if (i >= end - 1) {
211                                                /*
212                                                 * The rest of the char array is whitespace, so leave
213                                                 * the loop.
214                                                 */
215                                                break;
216                                        }
217                                }
218
219                                /*
220                                 * If we are in the middle of a quoted-printable encoded
221                                 * character, then exceed the max line length so the sequence
222                                 * doesn't get split up across multiple lines.
223                                 */
224                                if (encodedCharPos > 0) {
225                                        i += 3 - encodedCharPos;
226                                        if (i >= end - 1) {
227                                                /*
228                                                 * The rest of the char array was a quoted-printable
229                                                 * encoded char, so leave the loop.
230                                                 */
231                                                break;
232                                        }
233                                }
234
235                                writer.write(cbuf, start, i - start);
236                                if (quotedPrintable) {
237                                        writer.write('=');
238                                }
239                                writer.write(newline);
240                                writer.write(indent);
241                                curLineLength = indent.length() + 1;
242                                start = i;
243
244                                continue;
245                        }
246
247                        curLineLength++;
248                }
249
250                writer.write(cbuf, start, end - start);
251        }
252
253        /**
254         * Closes the writer.
255         */
256        @Override
257        public void close() throws IOException {
258                writer.close();
259        }
260
261        /**
262         * Flushes the writer.
263         */
264        @Override
265        public void flush() throws IOException {
266                writer.flush();
267        }
268
269        /**
270         * Gets the maximum length a line can be before it is folded (excluding the
271         * newline, defaults to 75).
272         * @return the line length or null if folding is disabled
273         */
274        public Integer getLineLength() {
275                return lineLength;
276        }
277
278        /**
279         * Sets the maximum length a line can be before it is folded (excluding the
280         * newline, defaults to 75).
281         * @param lineLength the line length or null to disable folding
282         * @throws IllegalArgumentException if the line length is less than or equal
283         * to zero
284         */
285        public void setLineLength(Integer lineLength) {
286                if (lineLength != null && lineLength <= 0) {
287                        throw new IllegalArgumentException("Line length must be greater than 0.");
288                }
289                this.lineLength = lineLength;
290        }
291
292        /**
293         * Gets the string that is prepended to each folded line (defaults to a
294         * single space character).
295         * @return the indent string
296         */
297        public String getIndent() {
298                return indent;
299        }
300
301        /**
302         * Sets the string that is prepended to each folded line (defaults to a
303         * single space character).
304         * @param indent the indent string
305         * @throws IllegalArgumentException if the length of the indent string is
306         * greater than the max line length
307         */
308        public void setIndent(String indent) {
309                if (lineLength != null && indent.length() >= lineLength) {
310                        throw new IllegalArgumentException("The length of the indent string must be less than the max line length.");
311                }
312                this.indent = indent;
313        }
314
315        /**
316         * Gets the newline sequence that is used to separate lines (defaults to
317         * CRLF).
318         * @return the newline sequence
319         */
320        public String getNewline() {
321                return newline;
322        }
323
324        /**
325         * Sets the newline sequence that is used to separate lines (defaults to
326         * CRLF).
327         * @param newline the newline sequence
328         */
329        public void setNewline(String newline) {
330                this.newline = newline;
331        }
332
333        /**
334         * Gets the wrapped {@link Writer} object.
335         * @return the wrapped writer
336         */
337        public Writer getWriter() {
338                return writer;
339        }
340
341        /**
342         * Gets the writer's character encoding.
343         * @return the writer's character encoding or null if undefined
344         */
345        public Charset getEncoding() {
346                if (!(writer instanceof OutputStreamWriter)) {
347                        return null;
348                }
349
350                OutputStreamWriter osw = (OutputStreamWriter) writer;
351                String charsetStr = osw.getEncoding();
352                return (charsetStr == null) ? null : Charset.forName(charsetStr);
353        }
354}
