081/**
082 * <p>
083 * Writes xCards (XML-encoded vCards) in a streaming fashion.
084 * </p>
085 * <p>
086 * <b>Example:</b>
087 * 
088 * <pre class="brush:java">
089 * VCard vcard1 = ...
090 * VCard vcard2 = ...
091 * File file = new File("vcards.xml");
092 * XCardWriter writer = null;
093 * try {
094 *   writer = new XCardWriter(file);
095 *   writer.write(vcard1);
096 *   writer.write(vcard2);
097 * } finally {
098 *   if (writer != null) writer.close();
099 * }
100 * </pre>
101 * 
102 * </p>
103 * @author Michael Angstadt
104 * @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
105 */
106public class XCardWriter extends StreamWriter {

265        private static Map<String, String> createOutputProperties(int indent, String xmlVersion) {
266                Map<String, String> properties = new HashMap<String, String>();
267                properties.put(OutputKeys.METHOD, "xml");
268
269                if (indent >= 0) {
270                        properties.put(OutputKeys.INDENT, "yes");
271                        properties.put("{http://xml.apache.org/xslt}indent-amount", indent + "");
272                }
273
274                if (xmlVersion != null) {
275                        properties.put(OutputKeys.VERSION, xmlVersion);
276                }
277
278                return properties;
279        }

281        private XCardWriter(Writer writer, Node parent, Map<String, String> outputProperties) {
282                this.writer = writer;
283
284                if (parent instanceof Document) {
285                        Node root = parent.getFirstChild();
286                        if (root != null) {
287                                parent = root;
288                        }
289                }
290                this.vcardsElementExists = isVCardsElement(parent);
291
292                try {
293                        SAXTransformerFactory factory = (SAXTransformerFactory) TransformerFactory.newInstance();
294                        handler = factory.newTransformerHandler();
295                } catch (TransformerConfigurationException e) {
296                        throw new RuntimeException(e);
297                }
298
299                Transformer transformer = handler.getTransformer();
300
301                /*
302                 * Using Transformer#setOutputProperties(Properties) doesn't work for
303                 * some reason for setting the number of indentation spaces.
304                 */
305                for (Map.Entry<String, String> entry : outputProperties.entrySet()) {
306                        String key = entry.getKey();
307                        String value = entry.getValue();
308                        transformer.setOutputProperty(key, value);
309                }
310
311                Result result = (writer == null) ? new DOMResult(parent) : new StreamResult(writer);
312                handler.setResult(result);
313        }
314
315        private boolean isVCardsElement(Node node) {
316                if (node == null) {
317                        return false;
318                }
319
320                if (!(node instanceof Element)) {
321                        return false;
322                }
323
324                return XmlUtils.hasQName(node, XCardQNames.VCARDS);
325        }

327        @Override
328        protected void _write(VCard vcard, List<VCardProperty> properties) throws IOException {
329                try {
330                        if (!started) {
331                                handler.startDocument();
332
333                                if (!vcardsElementExists) {
334                                        //don't output a <vcards> element if the parent is a <vcards> element
335                                        start(VCARDS);
336                                }
337
338                                started = true;
339                        }
340
341                        ListMultimap<String, VCardProperty> propertiesByGroup = new ListMultimap<String, VCardProperty>(); //group the types by group name (null = no group name)
342                        for (VCardProperty property : properties) {
343                                propertiesByGroup.put(property.getGroup(), property);
344                        }
345
346                        start(VCARD);
347
348                        for (Map.Entry<String, List<VCardProperty>> entry : propertiesByGroup) {
349                                String groupName = entry.getKey();
350                                if (groupName != null) {
351                                        AttributesImpl attr = new AttributesImpl();
352                                        attr.addAttribute(XCardQNames.NAMESPACE, "", "name", "", groupName);
353
354                                        start(GROUP, attr);
355                                }
356
357                                for (VCardProperty property : entry.getValue()) {
358                                        write(property, vcard);
359                                }
360
361                                if (groupName != null) {
362                                        end(GROUP);
363                                }
364                        }
365
366                        end(VCARD);
367                } catch (SAXException e) {
368                        throw new IOException(e);
369                }
370        }

377        /**
378         * Registers the data type of an experimental parameter. Experimental
379         * parameters use the "unknown" data type by default.
380         * @param parameterName the parameter name (e.g. "x-foo")
381         * @param dataType the data type or null to remove
382         */
383        public void registerParameterDataType(String parameterName, VCardDataType dataType) {
384                parameterName = parameterName.toLowerCase();
385                if (dataType == null) {
386                        parameterDataTypes.remove(parameterName);
387                } else {
388                        parameterDataTypes.put(parameterName, dataType);
389                }
390        }

392        /**
393         * Terminates the XML document and closes the output stream.
394         */
395        public void close() throws IOException {
396                try {
397                        if (!started) {
398                                handler.startDocument();
399
400                                if (!vcardsElementExists) {
401                                        //don't output a <vcards> element if the parent is a <vcards> element
402                                        start(VCARDS);
403                                }
404                        }
405
406                        if (!vcardsElementExists) {
407                                end(VCARDS);
408                        }
409                        handler.endDocument();
410                } catch (SAXException e) {
411                        throw new IOException(e);
412                }
413
414                if (writer != null) {
415                        writer.close();
416                }
417        }

419        @SuppressWarnings({ "rawtypes", "unchecked" })
420        private void write(VCardProperty property, VCard vcard) throws SAXException {
421                VCardPropertyScribe scribe = index.getPropertyScribe(property);
422                VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
423
424                //get the property element to write
425                Element propertyElement;
426                if (property instanceof Xml) {
427                        Xml xml = (Xml) property;
428                        Document value = xml.getValue();
429                        if (value == null) {
430                                return;
431                        }
432                        propertyElement = XmlUtils.getRootElement(value);
433                } else {
434                        QName qname = scribe.getQName();
435                        propertyElement = DOC.createElementNS(qname.getNamespaceURI(), qname.getLocalPart());
436                        try {
437                                scribe.writeXml(property, propertyElement);
438                        } catch (SkipMeException e) {
439                                return;
440                        } catch (EmbeddedVCardException e) {
441                                return;
442                        }
443                }
444
445                start(propertyElement);
446
447                write(parameters);
448                write(propertyElement);
449
450                end(propertyElement);
451        }
452
453        private void write(Element propertyElement) throws SAXException {
454                NodeList children = propertyElement.getChildNodes();
455                for (int i = 0; i < children.getLength(); i++) {
456                        Node child = children.item(i);
457
458                        if (child instanceof Element) {
459                                Element element = (Element) child;
460
461                                if (element.hasChildNodes()) {
462                                        start(element);
463                                        write(element);
464                                        end(element);
465                                } else {
466                                        childless(element);
467                                }
468
469                                continue;
470                        }
471
472                        if (child instanceof Text) {
473                                Text text = (Text) child;
474                                text(text.getTextContent());
475                                continue;
476                        }
477                }
478        }
479
480        private void write(VCardParameters parameters) throws SAXException {
481                if (parameters.isEmpty()) {
482                        return;
483                }
484
485                start(PARAMETERS);
486
487                for (Map.Entry<String, List<String>> parameter : parameters) {
488                        String parameterName = parameter.getKey().toLowerCase();
489                        start(parameterName);
490
491                        for (String parameterValue : parameter.getValue()) {
492                                VCardDataType dataType = parameterDataTypes.get(parameterName);
493                                String dataTypeElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
494
495                                start(dataTypeElementName);
496                                text(parameterValue);
497                                end(dataTypeElementName);
498                        }
499
500                        end(parameterName);
501                }
502
503                end(PARAMETERS);
504        }

506        /**
507         * Makes an childless element appear as {@code<foo />} instead of
508         * {@code<foo></foo>}
509         * @param element the element
510         * @throws SAXException
511         */
512        private void childless(Element element) throws SAXException {
513                Attributes attributes = getElementAttributes(element);
514                handler.startElement(element.getNamespaceURI(), "", element.getLocalName(), attributes);
515                handler.endElement(element.getNamespaceURI(), "", element.getLocalName());
516        }
517
518        private void start(Element element) throws SAXException {
519                Attributes attributes = getElementAttributes(element);
520                start(element.getNamespaceURI(), element.getLocalName(), attributes);
521        }
522
523        private void start(String element) throws SAXException {
524                start(element, new AttributesImpl());
525        }
526
527        private void start(QName qname) throws SAXException {
528                start(qname, new AttributesImpl());
529        }

531        private void start(QName qname, Attributes attributes) throws SAXException {
532                start(qname.getNamespaceURI(), qname.getLocalPart(), attributes);

534
535        private void start(String element, Attributes attributes) throws SAXException {
536                start(targetVersion.getXmlNamespace(), element, attributes);
537        }
538
539        private void start(String namespace, String element, Attributes attributes) throws SAXException {
540                handler.startElement(namespace, "", element, attributes);
541        }
542
543        private void end(Element element) throws SAXException {
544                end(element.getNamespaceURI(), element.getLocalName());
545        }
546
547        private void end(String element) throws SAXException {
548                end(targetVersion.getXmlNamespace(), element);
549        }
550
551        private void end(QName qname) throws SAXException {
552                end(qname.getNamespaceURI(), qname.getLocalPart());
553        }
554
555        private void end(String namespace, String element) throws SAXException {
556                handler.endElement(namespace, "", element);
557        }
558
559        private void text(String text) throws SAXException {
560                handler.characters(text.toCharArray(), 0, text.length());
561        }
562
563        private Attributes getElementAttributes(Element element) {
564                AttributesImpl attributes = new AttributesImpl();
565                NamedNodeMap attributeNodes = element.getAttributes();
566                for (int i = 0; i < attributeNodes.getLength(); i++) {
567                        Node node = attributeNodes.item(i);
568
569                        String localName = node.getLocalName();
570                        if ("xmlns".equals(localName)) {
571                                continue;
572                        }
573
574                        attributes.addAttribute(node.getNamespaceURI(), "", localName, "", node.getNodeValue());
575                }
576                return attributes;
577        }
