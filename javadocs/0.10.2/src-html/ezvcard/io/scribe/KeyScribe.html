031/**
032 * Marshals {@link Key} properties.
033 * @author Michael Angstadt
034 */
035public class KeyScribe extends BinaryPropertyScribe<Key, KeyType> {
036        public KeyScribe() {
037                super(Key.class, "KEY");
038        }
039
040        @Override
041        protected KeyType _mediaTypeFromTypeParameter(String type) {
042                return KeyType.get(type, null, null);
043        }
044
045        @Override
046        protected KeyType _mediaTypeFromMediaTypeParameter(String mediaType) {
047                return KeyType.get(null, mediaType, null);
048        }
049
050        @Override
051        protected KeyType _mediaTypeFromFileExtension(String extension) {
052                return KeyType.find(null, null, extension);
053        }
054
055        @Override
056        protected Key _newInstance(String uri, KeyType contentType) {
057                return new Key(uri, contentType);
058        }
059
060        @Override
061        protected Key _newInstance(byte[] data, KeyType contentType) {
062                return new Key(data, contentType);
063        }
064}
