038    /**
039     * Marshals {@link PlaceProperty} properties.
040     * @param <T> the property class
041     * @author Michael Angstadt
042     */
043    public abstract class PlacePropertyScribe<T extends PlaceProperty> extends VCardPropertyScribe<T> {
044            public PlacePropertyScribe(Class<T> clazz, String propertyName) {
045                    super(clazz, propertyName);
046            }
047    
048            @Override
049            protected VCardDataType _defaultDataType(VCardVersion version) {
050                    return VCardDataType.TEXT;
051            }
052    
053            @Override
054            protected VCardDataType _dataType(T property, VCardVersion version) {
055                    if (property.getText() != null) {
056                            return VCardDataType.TEXT;
057                    }
058    
059                    if (property.getUri() != null || property.getGeoUri() != null) {
060                            return VCardDataType.URI;
061                    }
062    
063                    return _defaultDataType(version);
064            }
065    
066            @Override
067            protected String _writeText(T property, VCardVersion version) {
068                    String text = property.getText();
069                    if (text != null) {
070                            return escape(text);
071                    }
072    
073                    String uri = property.getUri();
074                    if (uri != null) {
075                            return uri;
076                    }
077    
078                    GeoUri geoUri = property.getGeoUri();
079                    if (geoUri != null) {
080                            return geoUri.toString();
081                    }
082    
083                    return "";
084            }
085    
086            @Override
087            protected T _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
088                    T property = newInstance();
089                    value = unescape(value);
090    
091                    if (dataType == VCardDataType.TEXT) {
092                            property.setText(value);
093                            return property;
094                    }
095    
096                    if (dataType == VCardDataType.URI) {
097                            try {
098                                    property.setGeoUri(GeoUri.parse(value));
099                            } catch (IllegalArgumentException e) {
100                                    property.setUri(value);
101                            }
102                            return property;
103                    }
104    
105                    property.setText(value);
106                    return property;
107            }
108    
109            @Override
110            protected void _writeXml(T property, XCardElement parent) {
111                    String text = property.getText();
112                    if (text != null) {
113                            parent.append(VCardDataType.TEXT, text);
114                            return;
115                    }
116    
117                    String uri = property.getUri();
118                    if (uri != null) {
119                            parent.append(VCardDataType.URI, uri);
120                            return;
121                    }
122    
123                    GeoUri geoUri = property.getGeoUri();
124                    if (geoUri != null) {
125                            parent.append(VCardDataType.URI, geoUri.toString());
126                            return;
127                    }
128    
129                    parent.append(VCardDataType.TEXT, "");
130            }
131    
132            @Override
133            protected T _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
134                    T property = newInstance();
135    
136                    String text = element.first(VCardDataType.TEXT);
137                    if (text != null) {
138                            property.setText(text);
139                            return property;
140                    }
141    
142                    String uri = element.first(VCardDataType.URI);
143                    if (uri != null) {
144                            try {
145                                    property.setGeoUri(GeoUri.parse(uri));
146                            } catch (IllegalArgumentException e) {
147                                    property.setUri(uri);
148                            }
149                            return property;
150                    }
151    
152                    throw missingXmlElements(VCardDataType.TEXT, VCardDataType.URI);
153            }
154    
155            @Override
156            protected JCardValue _writeJson(T property) {
157                    String text = property.getText();
158                    if (text != null) {
159                            return JCardValue.single(text);
160                    }
161    
162                    String uri = property.getUri();
163                    if (uri != null) {
164                            return JCardValue.single(uri);
165                    }
166    
167                    GeoUri geoUri = property.getGeoUri();
168                    if (geoUri != null) {
169                            return JCardValue.single(geoUri.toString());
170                    }
171    
172                    return JCardValue.single("");
173            }
174    
175            @Override
176            protected T _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
177                    T property = newInstance();
178                    String valueStr = value.asSingle();
179    
180                    if (dataType == VCardDataType.TEXT) {
181                            property.setText(valueStr);
182                            return property;
183                    }
184    
185                    if (dataType == VCardDataType.URI) {
186                            try {
187                                    property.setGeoUri(GeoUri.parse(valueStr));
188                            } catch (IllegalArgumentException e) {
189                                    property.setUri(valueStr);
190                            }
191                            return property;
192                    }
193    
194                    property.setText(valueStr);
195                    return property;
196            }
197    
198            protected abstract T newInstance();
199    }
