044    /**
045     * Writes data to an vCard JSON data stream (jCard).
046     * @author Michael Angstadt
047     * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
048     */
049    public class JCardRawWriter implements Closeable, Flushable {
050            private final Writer writer;
051            private final boolean wrapInArray;
052            private JsonGenerator generator;
053            private boolean indent = false;
054            private boolean open = false;
055    
056            /**
057             * Creates a new raw writer.
058             * @param writer the writer to the data stream
059             * @param wrapInArray true to wrap everything in an array, false not to
060             * (useful when writing more than one vCard)
061             */
062            public JCardRawWriter(Writer writer, boolean wrapInArray) {
063                    this.writer = writer;
064                    this.wrapInArray = wrapInArray;
065            }
066    
067            /**
068             * Gets whether or not the JSON will be pretty-printed.
069             * @return true if it will be pretty-printed, false if not (defaults to
070             * false)
071             */
072            public boolean isIndent() {
073                    return indent;
074            }
075    
076            /**
077             * Sets whether or not to pretty-print the JSON.
078             * @param indent true to pretty-print it, false not to (defaults to false)
079             */
080            public void setIndent(boolean indent) {
081                    this.indent = indent;
082            }
083    
084            /**
085             * Writes the beginning of a new "vcard" component.
086             * @throws IOException if there's an I/O problem
087             */
088            public void writeStartVCard() throws IOException {
089                    if (generator == null) {
090                            init();
091                    }
092    
093                    if (open) {
094                            writeEndVCard();
095                    }
096    
097                    generator.writeStartArray();
098                    indent(0);
099                    generator.writeString("vcard");
100                    generator.writeStartArray(); //start properties array
101    
102                    open = true;
103            }
104    
105            /**
106             * Closes the "vcard" component array.
107             * @throws IllegalStateException if the component was never opened (
108             * {@link #writeStartVCard} must be called first)
109             * @throws IOException if there's an I/O problem
110             */
111            public void writeEndVCard() throws IOException {
112                    if (!open) {
113                            throw new IllegalStateException("Call \"writeStartVCard\" first.");
114                    }
115    
116                    generator.writeEndArray(); //end the properties array
117                    generator.writeEndArray(); //end the "vcard" component array
118    
119                    open = false;
120            }
121    
122            /**
123             * Writes a property to the current component.
124             * @param propertyName the property name (e.g. "version")
125             * @param dataType the data type or null for "unknown"
126             * @param value the property value
127             * @throws IllegalStateException if the "vcard" component was never opened
128             * or was just closed ({@link #writeStartVCard} must be called first)
129             * @throws IOException if there's an I/O problem
130             */
131            public void writeProperty(String propertyName, VCardDataType dataType, JCardValue value) throws IOException {
132                    writeProperty(null, propertyName, new VCardParameters(), dataType, value);
133            }
134    
135            /**
136             * Writes a property to the current vCard.
137             * @param group the group or null if there is no group
138             * @param propertyName the property name (e.g. "version")
139             * @param parameters the parameters
140             * @param dataType the data type or null for "unknown"
141             * @param value the property value
142             * @throws IllegalStateException if the "vcard" component was never opened
143             * or was just closed ({@link #writeStartVCard} must be called first)
144             * @throws IOException if there's an I/O problem
145             */
146            public void writeProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) throws IOException {
147                    if (!open) {
148                            throw new IllegalStateException("Call \"writeStartVCard\" first.");
149                    }
150    
151                    generator.writeStartArray();
152                    indent(2);
153    
154                    //write the property name
155                    generator.writeString(propertyName);
156    
157                    //write parameters
158                    generator.writeStartObject();
159                    for (Map.Entry<String, List<String>> entry : parameters) {
160                            String name = entry.getKey().toLowerCase();
161                            List<String> values = entry.getValue();
162                            if (values.isEmpty()) {
163                                    continue;
164                            }
165    
166                            if (values.size() == 1) {
167                                    generator.writeStringField(name, values.get(0));
168                            } else {
169                                    generator.writeArrayFieldStart(name);
170                                    for (String paramValue : values) {
171                                            generator.writeString(paramValue);
172                                    }
173                                    generator.writeEndArray();
174                            }
175                    }
176    
177                    //write group
178                    if (group != null) {
179                            generator.writeStringField("group", group);
180                    }
181    
182                    generator.writeEndObject(); //end parameters object
183    
184                    //write data type
185                    generator.writeString((dataType == null) ? "unknown" : dataType.getName().toLowerCase());
186    
187                    //write value
188                    if (value.getValues().isEmpty()) {
189                            generator.writeString("");
190                    } else {
191                            for (JsonValue jsonValue : value.getValues()) {
192                                    writeValue(jsonValue);
193                            }
194                    }
195    
196                    generator.writeEndArray();
197            }
198    
199            private void writeValue(JsonValue jsonValue) throws IOException {
200                    if (jsonValue.isNull()) {
201                            generator.writeNull();
202                            return;
203                    }
204    
205                    Object val = jsonValue.getValue();
206                    if (val != null) {
207                            if (val instanceof Byte) {
208                                    generator.writeNumber((Byte) val);
209                            } else if (val instanceof Short) {
210                                    generator.writeNumber((Short) val);
211                            } else if (val instanceof Integer) {
212                                    generator.writeNumber((Integer) val);
213                            } else if (val instanceof Long) {
214                                    generator.writeNumber((Long) val);
215                            } else if (val instanceof Float) {
216                                    generator.writeNumber((Float) val);
217                            } else if (val instanceof Double) {
218                                    generator.writeNumber((Double) val);
219                            } else if (val instanceof Boolean) {
220                                    generator.writeBoolean((Boolean) val);
221                            } else {
222                                    generator.writeString(val.toString());
223                            }
224                            return;
225                    }
226    
227                    List<JsonValue> array = jsonValue.getArray();
228                    if (array != null) {
229                            generator.writeStartArray();
230                            for (JsonValue element : array) {
231                                    writeValue(element);
232                            }
233                            generator.writeEndArray();
234                            return;
235                    }
236    
237                    Map<String, JsonValue> object = jsonValue.getObject();
238                    if (object != null) {
239                            generator.writeStartObject();
240                            for (Map.Entry<String, JsonValue> entry : object.entrySet()) {
241                                    generator.writeFieldName(entry.getKey());
242                                    writeValue(entry.getValue());
243                            }
244                            generator.writeEndObject();
245                            return;
246                    }
247            }
248    
249            /**
250             * Checks to see if pretty-printing is enabled, and adds indentation
251             * whitespace if it is.
252             * @param spaces the number of spaces to indent with
253             * @throws IOException
254             */
255            private void indent(int spaces) throws IOException {
256                    if (!indent) {
257                            return;
258                    }
259    
260                    generator.writeRaw(NEWLINE);
261                    for (int i = 0; i < spaces; i++) {
262                            generator.writeRaw(' ');
263                    }
264            }
265    
266            /**
267             * Flushes the JSON stream.
268             */
269            public void flush() throws IOException {
270                    if (generator == null) {
271                            return;
272                    }
273    
274                    generator.flush();
275            }
276    
277            /**
278             * Finishes writing the JSON document so that it is syntactically correct.
279             * No more data can be written once this method is called.
280             * @throws IOException if there's a problem closing the stream
281             */
282            public void closeJsonStream() throws IOException {
283                    if (generator == null) {
284                            return;
285                    }
286    
287                    while (open) {
288                            writeEndVCard();
289                    }
290    
291                    if (wrapInArray) {
292                            indent(0);
293                            generator.writeEndArray();
294                    }
295    
296                    generator.close();
297            }
298    
299            /**
300             * Finishes writing the JSON document and closes the underlying
301             * {@link Writer}.
302             * @throws IOException if there's a problem closing the stream
303             */
304            public void close() throws IOException {
305                    if (generator == null) {
306                            return;
307                    }
308    
309                    closeJsonStream();
310                    writer.close();
311            }
312    
313            private void init() throws IOException {
314                    JsonFactory factory = new JsonFactory();
315                    factory.configure(Feature.AUTO_CLOSE_TARGET, false);
316                    generator = factory.createJsonGenerator(writer);
317    
318                    if (wrapInArray) {
319                            generator.writeStartArray();
320                            indent(0);
321                    }
322            }
323    }
