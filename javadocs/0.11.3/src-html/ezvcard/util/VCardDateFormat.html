044/**
045 * Defines all of the date formats that are used in vCards, and also
046 * parses/formats vCard dates. These date formats are defined in the ISO8601
047 * specification.
048 * @author Michael Angstadt
049 */
050public enum VCardDateFormat {
051        //@formatter:off
052        /**
053         * Example: 20120701
054         */
055        DATE_BASIC(
056        "yyyyMMdd"),
057        
058        /**
059         * Example: 2012-07-01
060         */
061        DATE_EXTENDED(
062        "yyyy-MM-dd"),
063        
064        /**
065         * Example: 20120701T142110-0500
066         */
067        DATE_TIME_BASIC(
068        "yyyyMMdd'T'HHmmssZ"),
069        
070        /**
071         * Example: 2012-07-01T14:21:10-05:00
072         */
073        DATE_TIME_EXTENDED(
074        "yyyy-MM-dd'T'HH:mm:ssZ"){
075                @SuppressWarnings("serial")
076                @Override
077                public DateFormat getDateFormat(TimeZone timezone) {
078                        DateFormat df = new SimpleDateFormat(formatStr, Locale.ROOT){
079                                @Override
080                                public StringBuffer format(Date date, StringBuffer toAppendTo, FieldPosition fieldPosition){
081                                        StringBuffer sb = super.format(date, toAppendTo, fieldPosition);
082                                        
083                                        //add a colon between the hour and minute offsets
084                                        sb.insert(sb.length()-2, ':');
085                                        
086                                        return sb;
087                                }
088                        };
089                        
090                        if (timezone != null){
091                                df.setTimeZone(timezone);
092                        }
093                        
094                        return df;
095                }
096        },
097        
098        /**
099         * Example: 20120701T192110Z
100         */
101        UTC_DATE_TIME_BASIC(
102        "yyyyMMdd'T'HHmmss'Z'"){
103                @Override
104                public DateFormat getDateFormat(TimeZone timezone) {
105                        //always use the UTC timezone
106                        TimeZone utc = TimeZone.getTimeZone("UTC");
107                        return super.getDateFormat(utc);
108                }
109        },
110        
111        /**
112         * Example: 2012-07-01T19:21:10Z
113         */
114        UTC_DATE_TIME_EXTENDED(
115        "yyyy-MM-dd'T'HH:mm:ss'Z'"){
116                @Override
117                public DateFormat getDateFormat(TimeZone timezone) {
118                        //always use the UTC timezone
119                        TimeZone utc = TimeZone.getTimeZone("UTC");
120                        return super.getDateFormat(utc);
121                }
122        },
123        
124        /**
125         * Example: 2012-07-01T14:21:10-0500
126         */
127        HCARD_DATE_TIME(
128        "yyyy-MM-dd'T'HH:mm:ssZ")
129        
130        ;
131        //@formatter:on
132
133        /**
134         * The {@link SimpleDateFormat} format string used for parsing dates.
135         */
136        protected final String formatStr;
137
138        /**
139         * @param formatStr the {@link SimpleDateFormat} format string used for
140         * formatting dates.
141         */
142        VCardDateFormat(String formatStr) {
143                this.formatStr = formatStr;
144        }
145
146        /**
147         * Builds a {@link DateFormat} object for parsing and formating dates in
148         * this format.
149         * @return the {@link DateFormat} object
150         */
151        public DateFormat getDateFormat() {
152                return getDateFormat(null);
153        }
154
155        /**
156         * Builds a {@link DateFormat} object for parsing and formating dates in
157         * this format.
158         * @param timezone the timezone the date is in or null for the default
159         * timezone
160         * @return the {@link DateFormat} object
161         */
162        public DateFormat getDateFormat(TimeZone timezone) {
163                DateFormat df = new SimpleDateFormat(formatStr, Locale.ROOT);
164                if (timezone != null) {
165                        df.setTimeZone(timezone);
166                }
167                return df;
168        }
169
170        /**
171         * Formats a date in this vCard date format.
172         * @param date the date to format
173         * @return the date string
174         */
175        public String format(Date date) {
176                return format(date, null);
177        }
178
179        /**
180         * Formats a date in this vCard date format.
181         * @param date the date to format
182         * @param timezone the timezone to format the date in or null for the
183         * default timezone
184         * @return the date string
185         */
186        public String format(Date date, TimeZone timezone) {
187                DateFormat df = getDateFormat(timezone);
188                return df.format(date);
189        }
190
191        /**
192         * Parses a date string.
193         * @param dateStr the date string to parse (e.g. "20130609T181023Z")
194         * @return the parsed date
195         * @throws IllegalArgumentException if the date string isn't in one of the
196         * accepted ISO8601 formats
197         */
198        public static Date parse(String dateStr) {
199                return parseAsCalendar(dateStr).getTime();
200        }
201
202        /**
203         * <p>
204         * Parses a date string as a {@link Calendar} object. This allows the caller
205         * to retrieve the individual components of the original date string, which
206         * get lost with {@link Calendar#getTime}. This method was added at the
207         * request of a user who needed the UTC offset from the original timestamp
208         * string.
209         * </p>
210         * <p>
211         * Use {@link Calendar#isSet(int)} to determine if a field was included in the
212         * original timestamp string. Calls to this method should be made before
213         * calling {@link Calendar#get} because calling latter method can cause
214         * unset fields to become populated (as mentioned in the
215         * {@link Calendar#isSet(int) isSet} Javadocs).
216         * </p>
217         * <p>
218         * The calendar's timezone will be set to "GMT" if the "Z" suffix was used
219         * in the timestamp string. If a numeric offset was used, the timezone will
220         * look like "GMT-05:00". If no offset was specified, the timezone will be
221         * set to the local system's default timezone.
222         * </p>
223         * @param dateStr the date string to parse (e.g. "20130609T181023Z")
224         * @return the parsed date
225         * @throws IllegalArgumentException if the date string isn't in one of the
226         * accepted ISO8601 formats
227         */
228        public static Calendar parseAsCalendar(String dateStr) {
229                TimestampPattern p = new TimestampPattern(dateStr);
230                if (!p.matches()) {
231                        throw Messages.INSTANCE.getIllegalArgumentException(41, dateStr);
232                }
233
234                Calendar c = Calendar.getInstance(p.timezone());
235                c.clear();
236
237                c.set(Calendar.YEAR, p.year());
238                c.set(Calendar.MONTH, p.month() - 1);
239                c.set(Calendar.DATE, p.date());
240
241                if (p.hasTime()) {
242                        c.set(Calendar.HOUR_OF_DAY, p.hour());
243                        c.set(Calendar.MINUTE, p.minute());
244                        c.set(Calendar.SECOND, p.second());
245                        c.set(Calendar.MILLISECOND, p.millisecond());
246                }
247
248                return c;
249        }
250
251        /**
252         * Wrapper for a complex regular expression that parses multiple date
253         * formats.
254         */
255        private static class TimestampPattern {
256                //@formatter:off
257                private static final Pattern regex = Pattern.compile(
258                        "^(\\d{4})(" +
259                                "-?(\\d{2})-?(\\d{2})|" +
260                                "-(\\d{1,2})-(\\d{1,2})" + //allow single digit month and/or date as long as there are dashes
261                        ")" + 
262                        "(" +
263                                "T(\\d{2}):?(\\d{2}):?(\\d{2})(\\.\\d+)?" +
264                                "(" +
265                                        "Z|([-+])((\\d{2})|((\\d{2}):?(\\d{2})))" +
266                                ")?" +
267                        ")?$"
268                );
269                //@formatter:on
270
271                private final Matcher m;
272                private final boolean matches;
273
274                public TimestampPattern(String str) {
275                        m = regex.matcher(str);
276                        matches = m.find();
277                }
278
279                public boolean matches() {
280                        return matches;
281                }
282
283                public int year() {
284                        return parseInt(1);
285                }
286
287                public int month() {
288                        return parseInt(3, 5);
289                }
290
291                public int date() {
292                        return parseInt(4, 6);
293                }
294
295                public boolean hasTime() {
296                        return m.group(8) != null;
297                }
298
299                public int hour() {
300                        return parseInt(8);
301                }
302
303                public int minute() {
304                        return parseInt(9);
305                }
306
307                public int second() {
308                        return parseInt(10);
309                }
310
311                public int millisecond() {
312                        String s = m.group(11);
313                        if (s == null) {
314                                return 0;
315                        }
316
317                        double ms = Double.parseDouble(s) * 1000;
318                        return (int) Math.round(ms);
319                }
320
321                public TimeZone timezone() {
322                        String offsetStr = m.group(12);
323                        if (offsetStr == null) {
324                                return TimeZone.getDefault();
325                        }
326
327                        /*
328                         * Use the naked "GMT" timezone when "Z" is specified. This allows
329                         * the user to differentiate from when an offset of "00:00" is
330                         * explicitly specified (they refer to the same timezone, though).
331                         */
332                        if (offsetStr.equals("Z")) {
333                                return TimeZone.getTimeZone("GMT");
334                        }
335
336                        /*
337                         * Java is lenient regarding the format of the offset string. For
338                         * example, all of the following resolve to the same "GMT+03:00"
339                         * timezone: "GMT+3", "GMT+03", "GMT+3:00", "GMT+300"
340                         */
341                        return TimeZone.getTimeZone("GMT" + offsetStr);
342                }
343
344                private int parseInt(int... group) {
345                        for (int g : group) {
346                                String s = m.group(g);
347                                if (s != null) {
348                                        return Integer.parseInt(s);
349                                }
350                        }
351                        throw new NullPointerException();
352                }
353        }
354
355        /**
356         * Determines whether a date string has a time component.
357         * @param dateStr the date string (e.g. "20130601T120000")
358         * @return true if it has a time component, false if not
359         */
360        public static boolean dateHasTime(String dateStr) {
361                return dateStr.contains("T");
362        }
363
364        /**
365         * Determines whether a date string is in UTC time or has a timezone offset.
366         * @param dateStr the date string (e.g. "20130601T120000Z",
367         * "20130601T120000-0400")
368         * @return true if it has a timezone, false if not
369         */
370        public static boolean dateHasTimezone(String dateStr) {
371                return dateStr.endsWith("Z") || dateStr.matches(".*?[-+]\\d\\d:?\\d\\d");
372        }
373
374        /**
375         * Gets the {@link TimeZone} object that corresponds to the given ID.
376         * @param timezoneId the timezone ID (e.g. "America/New_York")
377         * @return the timezone object or null if not found
378         */
379        public static TimeZone parseTimeZoneId(String timezoneId) {
380                TimeZone timezone = TimeZone.getTimeZone(timezoneId);
381                return "GMT".equals(timezone.getID()) ? null : timezone;
382        }
383
384        /**
385         * Converts a {@link Date} object to a {@link Calendar} object.
386         * @param date the date (can be null)
387         * @return the calendar or null if the date is null
388         */
389        public static Calendar toCalendar(Date date) {
390                if (date == null) {
391                        return null;
392                }
393
394                Calendar c = Calendar.getInstance();
395                c.setTime(date);
396                return c;
397        }
