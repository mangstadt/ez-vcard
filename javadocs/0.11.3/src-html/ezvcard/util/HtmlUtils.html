037/**
038 * Generic HTML utility methods.
039 * @author Michael Angstadt
040 */
041public class HtmlUtils {
042        /**
043         * Determines whether the given element is a child of one of the given
044         * parent elements.
045         * @param child the child element
046         * @param possibleParents the possible parents
047         * @return true if it is a child, false if not
048         */
049        public static boolean isChildOf(Element child, Elements possibleParents) {
050                for (Element parent : child.parents()) {
051                        if (possibleParents.contains(parent)) {
052                                return true;
053                        }
054                }
055                return false;
056        }
057
058        /**
059         * Converts an HTML string to an HTML element.
060         * @param html the HTML
061         * @return the HTML element
062         */
063        public static Element toElement(String html) {
064                return toElement(html, null);
065        }
066
067        /**
068         * Converts an HTML string to an HTML element.
069         * @param html the HTML
070         * @param baseUrl the base URL
071         * @return the HTML element
072         */
073        public static Element toElement(String html, String baseUrl) {
074                Document d = (baseUrl == null) ? Jsoup.parse(html) : Jsoup.parse(html, baseUrl);
075                return d.getElementsByTag("body").first().children().first();
076        }
077
078        private HtmlUtils() {
079                //hide
080        }
081}
