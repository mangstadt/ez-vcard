032/**
033 * Singleton for accessing the i18n resource bundle.
034 * @author Michael Angstadt
035 */
036public enum Messages {
037        INSTANCE;
038
039        private final ResourceBundle messages;
040
041        Messages() {
042                messages = ResourceBundle.getBundle("ezvcard/messages");
043        }
044
045        /**
046         * Gets a validation warning message.
047         * @param code the message code
048         * @param args the message arguments
049         * @return the message or null if not found
050         */
051        public String getValidationWarning(int code, Object... args) {
052                return getMessage("validate." + code, args);
053        }
054
055        /**
056         * Gets a parser warning message.
057         * @param code the message code
058         * @param args the message arguments
059         * @return the message or null if not found
060         */
061        public String getParseMessage(int code, Object... args) {
062                return getMessage("parse." + code, args);
063        }
064
065        /**
066         * Gets an exception message.
067         * @param code the message code
068         * @param args the message arguments
069         * @return the message or null if not found
070         */
071        public String getExceptionMessage(int code, Object... args) {
072                String message = getMessage("exception." + code, args);
073                return (message == null) ? null : getMessage("exception.0", code, message);
074        }
075
076        /**
077         * Builds an {@link IllegalArgumentException} from an exception message.
078         * @param code the message code
079         * @param args the message arguments
080         * @return the exception or null if the message was not found
081         */
082        public IllegalArgumentException getIllegalArgumentException(int code, Object... args) {
083                String message = getExceptionMessage(code, args);
084                return (message == null) ? null : new IllegalArgumentException(message);
085        }
086
087        /**
088         * Gets a message.
089         * @param key the message key
090         * @param args the message arguments
091         * @return the message or null if not found
092         */
093        public String getMessage(String key, Object... args) {
094                try {
095                        String message = messages.getString(key);
096                        return MessageFormat.format(message, args);
097                } catch (MissingResourceException e) {
098                        return null;
099                }
100        }
101}
