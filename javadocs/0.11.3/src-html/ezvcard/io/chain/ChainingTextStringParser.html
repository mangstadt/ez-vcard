034/**
035 * Chainer class for parsing traditional, plain-text vCards from strings.
036 * @see Ezvcard#parse(String)
037 * @author Michael Angstadt
038 */
039public class ChainingTextStringParser extends ChainingTextParser<ChainingTextStringParser> {
040        public ChainingTextStringParser(String string) {
041                super(string);
042        }
043
044        @Override
045        public VCard first() {
046                try {
047                        return super.first();
048                } catch (IOException e) {
049                        //should never be thrown because we're reading from a string
050                        throw new RuntimeException(e);
051                }
052        }
053
054        @Override
055        public List<VCard> all() {
056                try {
057                        return super.all();
058                } catch (IOException e) {
059                        //should never be thrown because we're reading from a string
060                        throw new RuntimeException(e);
061                }
062        }
063}
