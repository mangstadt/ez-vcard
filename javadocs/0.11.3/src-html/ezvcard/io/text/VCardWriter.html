065/**
066 * <p>
067 * Writes {@link VCard} objects to a plain-text vCard data stream.
068 * </p>
069 * <p>
070 * <b>Example:</b>
071 * </p>
072 * 
073 * <pre class="brush:java">
074 * VCard vcard1 = ...
075 * VCard vcard2 = ...
076 * File file = new File("vcard.vcf");
077 * VCardWriter writer = null;
078 * try {
079 *   writer = new VCardWriter(file, VCardVersion.V3_0);
080 *   writer.write(vcard1);
081 *   writer.write(vcard2);
082 * } finally {
083 *   if (writer != null) writer.close();
084 * }
085 * </pre>
086 * 
087 * <p>
088 * <b>Changing the line folding settings:</b>
089 * </p>
090 * 
091 * <pre class="brush:java">
092 * VCardWriter writer = new VCardWriter(...);
093 * 
094 * //disable line folding
095 * writer.getVObjectWriter().getFoldedLineWriter().setLineLength(null);
096 * 
097 * //change line length
098 * writer.getVObjectWriter().getFoldedLineWriter().setLineLength(50);
099 * 
100 * //change folded line indent string
101 * writer.getVObjectWriter().getFoldedLineWriter().setIndent("\t");
102 * 
103 * </pre>
104 * @author Michael Angstadt
105 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
106 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
107 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
108 */
109public class VCardWriter extends StreamWriter implements Flushable {
110        private final VObjectWriter writer;
111        private final List<Boolean> prodIdStack = new ArrayList<Boolean>();
112        private VCardVersion targetVersion;
113        private TargetApplication targetApplication;
114        private Boolean includeTrailingSemicolons;

116        /**
117         * @param out the output stream to write to
118         * @param targetVersion the version that the vCards should conform to (if
119         * set to "4.0", vCards will be written in UTF-8 encoding)
120         */
121        public VCardWriter(OutputStream out, VCardVersion targetVersion) {
122                this((targetVersion == VCardVersion.V4_0) ? new Utf8Writer(out) : new OutputStreamWriter(out), targetVersion);
123        }

294        @Override
295        @SuppressWarnings({ "rawtypes", "unchecked" })
296        protected void _write(VCard vcard, List<VCardProperty> propertiesToAdd) throws IOException {
297                VCardVersion targetVersion = getTargetVersion();
298                TargetApplication targetApplication = getTargetApplication();
299
300                Boolean includeTrailingSemicolons = this.includeTrailingSemicolons;
301                if (includeTrailingSemicolons == null) {
302                        includeTrailingSemicolons = (targetVersion == VCardVersion.V4_0);
303                }
304
305                WriteContext context = new WriteContext(targetVersion, targetApplication, includeTrailingSemicolons);
306
307                writer.writeBeginComponent("VCARD");
308                writer.writeVersion(targetVersion.getVersion());
309
310                for (VCardProperty property : propertiesToAdd) {
311                        VCardPropertyScribe scribe = index.getPropertyScribe(property);
312
313                        String value = null;
314                        VCard nestedVCard = null;
315                        try {
316                                value = scribe.writeText(property, context);
317                        } catch (SkipMeException e) {
318                                continue;
319                        } catch (EmbeddedVCardException e) {
320                                nestedVCard = e.getVCard();
321                        }
322
323                        VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
324
325                        if (nestedVCard != null) {
326                                writeNestedVCard(nestedVCard, property, scribe, parameters, value);
327                                continue;
328                        }
329
330                        handleValueParameter(property, scribe, parameters);
331                        handleLabelParameter(property, parameters);
332                        handleQuotedPrintableEncodingParameter(property, parameters);
333
334                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), new VObjectParameters(parameters.getMap()), value);
335
336                        fixBinaryPropertyForOutlook(property);
337                }
338
339                writer.writeEndComponent("VCARD");
340        }
341
342        @SuppressWarnings("rawtypes")
343        private void writeNestedVCard(VCard nestedVCard, VCardProperty property, VCardPropertyScribe scribe, VCardParameters parameters, String value) throws IOException {
344                if (targetVersion == VCardVersion.V2_1) {
345                        //write a nested vCard (2.1 style)
346                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), new VObjectParameters(parameters.getMap()), value);
347                        prodIdStack.add(addProdId);
348                        addProdId = false;
349                        write(nestedVCard);
350                        addProdId = prodIdStack.remove(prodIdStack.size() - 1);
351                } else {
352                        //write an embedded vCard (3.0 style)
353                        StringWriter sw = new StringWriter();
354                        VCardWriter agentWriter = new VCardWriter(sw, targetVersion);
355                        agentWriter.getVObjectWriter().getFoldedLineWriter().setLineLength(null);
356                        agentWriter.setAddProdId(false);
357                        agentWriter.setCaretEncodingEnabled(isCaretEncodingEnabled());
358                        agentWriter.setIncludeTrailingSemicolons(this.includeTrailingSemicolons);
359                        agentWriter.setScribeIndex(index);
360                        agentWriter.setTargetApplication(targetApplication);
361                        agentWriter.setVersionStrict(versionStrict);
362                        try {
363                                agentWriter.write(nestedVCard);
364                        } catch (IOException e) {
365                                //should never be thrown because we're writing to a string
366                        } finally {
367                                IOUtils.closeQuietly(agentWriter);
368                        }
369
370                        String vcardStr = sw.toString();
371                        vcardStr = VObjectPropertyValues.escape(vcardStr);
372                        writer.writeProperty(property.getGroup(), scribe.getPropertyName(), new VObjectParameters(parameters.getMap()), vcardStr);
373                }
374        }

376        /**
377         * <p>
378         * Sets the property's VALUE parameter. This method only adds a VALUE
379         * parameter if all the following conditions are met:
380         * </p>
381         * <ol>
382         * <li>The data type is NOT "unknown"</li>
383         * <li>The data type is different from the property's default data type</li>
384         * <li>The data type does not fall under the "date/time special case" (see
385         * {@link #isDateTimeValueParameterSpecialCase})</li>
386         * </ol>
387         * @param property the property
388         * @param scribe the property scribe
389         * @param parameters the property parameters
390         */
391        @SuppressWarnings({ "rawtypes", "unchecked" })
392        private void handleValueParameter(VCardProperty property, VCardPropertyScribe scribe, VCardParameters parameters) {
393                VCardDataType dataType = scribe.dataType(property, targetVersion);
394                if (dataType == null) {
395                        return;
396                }
397
398                VCardDataType defaultDataType = scribe.defaultDataType(targetVersion);
399                if (dataType == defaultDataType) {
400                        return;
401                }
402
403                if (isDateTimeValueParameterSpecialCase(defaultDataType, dataType)) {
404                        return;
405                }
406
407                parameters.setValue(dataType);
408        }
409
410        /**
411         * <p>
412         * Escapes newline sequences in the LABEL parameter of {@link Address}
413         * properties. Newlines cannot normally be escaped in parameter values.
414         * </p>
415         * <p>
416         * Only version 4.0 allows this (and only version 4.0 defines a LABEL
417         * parameter), but this method does this for all versions for compatibility.
418         * </p>
419         * @param property the property
420         * @param parameters the property parameters
421         */
422        private void handleLabelParameter(VCardProperty property, VCardParameters parameters) {
423                if (!(property instanceof Address)) {
424                        return;
425                }
426
427                String label = parameters.getLabel();
428                if (label == null) {
429                        return;
430                }
431
432                label = escapeNewlines(label);
433                parameters.setLabel(label);
434        }

436        /**
437         * Disables quoted-printable encoding on the given property if the target
438         * vCard version does not support this encoding scheme.
439         * @param property the property
440         * @param parameters the property parameters
441         */
442        private void handleQuotedPrintableEncodingParameter(VCardProperty property, VCardParameters parameters) {
443                if (targetVersion == VCardVersion.V2_1) {
444                        return;
445                }
446
447                if (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE) {
448                        parameters.setEncoding(null);
449                        parameters.setCharset(null);
450                }
451        }

453        /**
454         * @see TargetApplication#OUTLOOK
455         */
456        private void fixBinaryPropertyForOutlook(VCardProperty property) throws IOException {
457                if (targetApplication != TargetApplication.OUTLOOK) {
458                        return;
459                }
460
461                if (getTargetVersion() == VCardVersion.V4_0) {
462                        //only do this for 2.1 and 3.0 vCards
463                        return;
464                }
465
466                if (!(property instanceof BinaryProperty)) {
467                        //property does not have binary data
468                        return;
469                }
470
471                BinaryProperty<?> binaryProperty = (BinaryProperty<?>) property;
472                if (binaryProperty.getData() == null) {
473                        //property value is not base64-encoded
474                        return;
475                }
476
477                writer.getFoldedLineWriter().writeln();
478        }

480        /**
481         * Determines if the given default data type is "date-and-or-time" and the
482         * given data type is time-based. Properties that meet this criteria should
483         * NOT be given a VALUE parameter.
484         * @param defaultDataType the property's default data type
485         * @param dataType the current property instance's data type
486         * @return true if the default data type is "date-and-or-time" and the data
487         * type is time-based, false otherwise
488         */
489        private boolean isDateTimeValueParameterSpecialCase(VCardDataType defaultDataType, VCardDataType dataType) {
490                //@formatter:off
491                return
492                defaultDataType == VCardDataType.DATE_AND_OR_TIME &&
493                (
494                        dataType == VCardDataType.DATE ||
495                        dataType == VCardDataType.DATE_TIME ||
496                        dataType == VCardDataType.TIME
497                );
498                //@formatter:on
499        }
