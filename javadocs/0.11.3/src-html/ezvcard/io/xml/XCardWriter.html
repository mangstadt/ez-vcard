077/**
078 * <p>
079 * Writes xCards (XML-encoded vCards) in a streaming fashion.
080 * </p>
081 * <p>
082 * <b>Example:</b>
083 * </p>
084 * 
085 * <pre class="brush:java">
086 * VCard vcard1 = ...
087 * VCard vcard2 = ...
088 * File file = new File("vcards.xml");
089 * XCardWriter writer = null;
090 * try {
091 *   writer = new XCardWriter(file);
092 *   writer.write(vcard1);
093 *   writer.write(vcard2);
094 * } finally {
095 *   if (writer != null) writer.close();
096 * }
097 * </pre>
098 * @author Michael Angstadt
099 * @see <a href="http://tools.ietf.org/html/rfc6351">RFC 6351</a>
100 */

239        private XCardWriter(Writer writer, Node parent, Map<String, String> outputProperties) {
240                this.writer = writer;
241
242                if (parent instanceof Document) {
243                        Node root = ((Document) parent).getDocumentElement();
244                        if (root != null) {
245                                parent = root;
246                        }
247                }
248                this.vcardsElementExists = isVCardsElement(parent);
249
250                try {
251                        SAXTransformerFactory factory = (SAXTransformerFactory) TransformerFactory.newInstance();
252                        handler = factory.newTransformerHandler();
253                } catch (TransformerConfigurationException e) {
254                        throw new RuntimeException(e);
255                }
256
257                Transformer transformer = handler.getTransformer();
258
259                /*
260                 * Using Transformer#setOutputProperties(Properties) doesn't work for
261                 * some reason for setting the number of indentation spaces.
262                 */
263                for (Map.Entry<String, String> entry : outputProperties.entrySet()) {
264                        String key = entry.getKey();
265                        String value = entry.getValue();
266                        transformer.setOutputProperty(key, value);
267                }
268
269                Result result = (writer == null) ? new DOMResult(parent) : new StreamResult(writer);
270                handler.setResult(result);
271        }
272
273        private boolean isVCardsElement(Node node) {
274                if (node == null) {
275                        return false;
276                }
277
278                if (!(node instanceof Element)) {
279                        return false;
280                }
281
282                return XmlUtils.hasQName(node, VCARDS);
283        }
284
285        @Override
286        protected void _write(VCard vcard, List<VCardProperty> properties) throws IOException {
287                try {
288                        if (!started) {
289                                handler.startDocument();
290
291                                if (!vcardsElementExists) {
292                                        //don't output a <vcards> element if the parent is a <vcards> element
293                                        start(VCARDS);
294                                }
295
296                                started = true;
297                        }
298
299                        ListMultimap<String, VCardProperty> propertiesByGroup = new ListMultimap<String, VCardProperty>(); //group the types by group name (null = no group name)
300                        for (VCardProperty property : properties) {
301                                propertiesByGroup.put(property.getGroup(), property);
302                        }
303
304                        start(VCARD);
305
306                        for (Map.Entry<String, List<VCardProperty>> entry : propertiesByGroup) {
307                                String groupName = entry.getKey();
308                                if (groupName != null) {
309                                        AttributesImpl attr = new AttributesImpl();
310                                        attr.addAttribute(XCardQNames.NAMESPACE, "", "name", "", groupName);
311
312                                        start(GROUP, attr);
313                                }
314
315                                for (VCardProperty property : entry.getValue()) {
316                                        write(property, vcard);
317                                }
318
319                                if (groupName != null) {
320                                        end(GROUP);
321                                }
322                        }
323
324                        end(VCARD);
325                } catch (SAXException e) {
326                        throw new IOException(e);
327                }
328        }
329
330        /**
331         * Terminates the XML document and closes the output stream.
332         */
333        public void close() throws IOException {
334                try {
335                        if (!started) {
336                                handler.startDocument();
337
338                                if (!vcardsElementExists) {
339                                        //don't output a <vcards> element if the parent is a <vcards> element
340                                        start(VCARDS);
341                                }
342                        }
343
344                        if (!vcardsElementExists) {
345                                end(VCARDS);
346                        }
347                        handler.endDocument();
348                } catch (SAXException e) {
349                        throw new IOException(e);
350                }
351
352                if (writer != null) {
353                        writer.close();
354                }
355        }

357        @SuppressWarnings({ "rawtypes", "unchecked" })
358        private void write(VCardProperty property, VCard vcard) throws SAXException {
359                VCardPropertyScribe scribe = index.getPropertyScribe(property);
360                VCardParameters parameters = scribe.prepareParameters(property, targetVersion, vcard);
361
362                removeUnsupportedParameters(parameters);
363
364                //get the property element to write
365                Element propertyElement;
366                if (property instanceof Xml) {
367                        Xml xml = (Xml) property;
368                        Document value = xml.getValue();
369                        if (value == null) {
370                                return;
371                        }
372                        propertyElement = value.getDocumentElement();
373                } else {
374                        QName qname = scribe.getQName();
375                        propertyElement = DOC.createElementNS(qname.getNamespaceURI(), qname.getLocalPart());
376                        try {
377                                scribe.writeXml(property, propertyElement);
378                        } catch (SkipMeException e) {
379                                return;
380                        } catch (EmbeddedVCardException e) {
381                                return;
382                        }
383                }
384
385                start(propertyElement);
386
387                write(parameters);
388                write(propertyElement);
389
390                end(propertyElement);
391        }
392
393        private void write(Element propertyElement) throws SAXException {
394                NodeList children = propertyElement.getChildNodes();
395                for (int i = 0; i < children.getLength(); i++) {
396                        Node child = children.item(i);
397
398                        if (child instanceof Element) {
399                                Element element = (Element) child;
400
401                                if (element.hasChildNodes()) {
402                                        start(element);
403                                        write(element);
404                                        end(element);
405                                } else {
406                                        childless(element);
407                                }
408
409                                continue;
410                        }
411
412                        if (child instanceof Text) {
413                                Text text = (Text) child;
414                                text(text.getTextContent());
415                                continue;
416                        }
417                }
418        }
419
420        private void write(VCardParameters parameters) throws SAXException {
421                if (parameters.isEmpty()) {
422                        return;
423                }
424
425                start(PARAMETERS);
426
427                for (Map.Entry<String, List<String>> parameter : parameters) {
428                        String parameterName = parameter.getKey().toLowerCase();
429                        start(parameterName);
430
431                        for (String parameterValue : parameter.getValue()) {
432                                VCardDataType dataType = parameterDataTypes.get(parameterName);
433                                String dataTypeElementName = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
434
435                                start(dataTypeElementName);
436                                text(parameterValue);
437                                end(dataTypeElementName);
438                        }
439
440                        end(parameterName);
441                }
442
443                end(PARAMETERS);
444        }

446        /**
447         * Makes an childless element appear as {@code <foo />} instead of
448         * {@code <foo></foo>}
449         * @param element the element
450         * @throws SAXException if there is a problem creating the element
451         */
452        private void childless(Element element) throws SAXException {
453                Attributes attributes = getElementAttributes(element);
454                handler.startElement(element.getNamespaceURI(), "", element.getLocalName(), attributes);
455                handler.endElement(element.getNamespaceURI(), "", element.getLocalName());
456        }
457
458        private void start(Element element) throws SAXException {
459                Attributes attributes = getElementAttributes(element);
460                start(element.getNamespaceURI(), element.getLocalName(), attributes);
461        }
462
463        private void start(String element) throws SAXException {
464                start(element, new AttributesImpl());
465        }
466
467        private void start(QName qname) throws SAXException {
468                start(qname, new AttributesImpl());
469        }
470
471        private void start(QName qname, Attributes attributes) throws SAXException {
472                start(qname.getNamespaceURI(), qname.getLocalPart(), attributes);
473        }
474
475        private void start(String element, Attributes attributes) throws SAXException {
476                start(targetVersion.getXmlNamespace(), element, attributes);
477        }
478
479        private void start(String namespace, String element, Attributes attributes) throws SAXException {
480                handler.startElement(namespace, "", element, attributes);
481        }
482
483        private void end(Element element) throws SAXException {
484                end(element.getNamespaceURI(), element.getLocalName());
485        }
486
487        private void end(String element) throws SAXException {
488                end(targetVersion.getXmlNamespace(), element);
489        }
490
491        private void end(QName qname) throws SAXException {
492                end(qname.getNamespaceURI(), qname.getLocalPart());
493        }
494
495        private void end(String namespace, String element) throws SAXException {
496                handler.endElement(namespace, "", element);
497        }
498
499        private void text(String text) throws SAXException {
500                handler.characters(text.toCharArray(), 0, text.length());
501        }
502
503        private Attributes getElementAttributes(Element element) {
504                AttributesImpl attributes = new AttributesImpl();
505                NamedNodeMap attributeNodes = element.getAttributes();
506                for (int i = 0; i < attributeNodes.getLength(); i++) {
507                        Node node = attributeNodes.item(i);
508
509                        String localName = node.getLocalName();
510                        if ("xmlns".equals(localName)) {
511                                continue;
512                        }
513
514                        attributes.addAttribute(node.getNamespaceURI(), "", localName, "", node.getNodeValue());
515                }
516                return attributes;
517        }
