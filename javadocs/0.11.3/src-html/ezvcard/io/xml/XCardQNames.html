001package ezvcard.io.xml;
002
003import javax.xml.namespace.QName;
004
005import ezvcard.VCardVersion;
006
007/**
008 * Contains the XML element names of the standard xCard elements (except the
009 * property element names).
010 * @author Michael Angstadt
011 */
012public interface XCardQNames {
013        String NAMESPACE = VCardVersion.V4_0.getXmlNamespace();
014        QName VCARDS = new QName(NAMESPACE, "vcards");
015        QName VCARD = new QName(NAMESPACE, "vcard");
016        QName GROUP = new QName(NAMESPACE, "group");
017        QName PARAMETERS = new QName(NAMESPACE, "parameters");
018}
