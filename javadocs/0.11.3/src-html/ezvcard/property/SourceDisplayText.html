035/**
036 * <p>
037 * Defines a textual representation of the {@link Source} property.
038 * </p>
039 * 
040 * <p>
041 * <b>Code sample</b>
042 * </p>
043 * 
044 * <pre class="brush:java">
045 * VCard vcard = new VCard();
046 * 
047 * SourceDisplayText name = new SourceDisplayText("Download the latest version of my vCard");
048 * vcard.setSourceDisplayText(name);
049 * </pre>
050 * 
051 * <p>
052 * <b>Property name:</b> {@code NAME}
053 * </p>
054 * <p>
055 * <b>Supported versions:</b> {@code 3.0}
056 * </p>
057 * @author Michael Angstadt
058 * @see <a href="http://tools.ietf.org/html/rfc2426#page-5">RFC 2426 p.5</a>
059 */
060@SupportedVersions(VCardVersion.V3_0)
061public class SourceDisplayText extends TextProperty {
062        /**
063         * Creates a source display text property.
064         * @param displayText the display text
065         */
066        public SourceDisplayText(String displayText) {
067                super(displayText);
068        }
069
070        /**
071         * Copy constructor.
072         * @param original the property to make a copy of
073         */
074        public SourceDisplayText(SourceDisplayText original) {
075                super(original);
076        }
077
078        @Override
079        public SourceDisplayText copy() {
080                return new SourceDisplayText(this);
081        }
082}
