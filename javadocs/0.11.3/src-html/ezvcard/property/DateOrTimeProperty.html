046/**
047 * Represents a property with a date and/or time value.
048 * @author Michael Angstadt
049 */
050public class DateOrTimeProperty extends VCardProperty implements HasAltId {
051        private String text;
052        private Calendar date;
053        private PartialDate partialDate;
054        private boolean dateHasTime;
055
056        /**
057         * Creates a date-and-or-time property.
058         * @param date the date value
059         */
060        public DateOrTimeProperty(Date date) {
061                this(VCardDateFormat.toCalendar(date));
062        }
063
064        /**
065         * Creates a date-and-or-time property.
066         * @param date the date value
067         * @param hasTime true to include the date's time component, false if it's
068         * strictly a date
069         */
070        public DateOrTimeProperty(Date date, boolean hasTime) {
071                setDate(date, hasTime);
072        }
073
074        /**
075         * Creates a date-and-or-time property.
076         * @param date the date value
077         */
078        public DateOrTimeProperty(Calendar date) {
079                this(date, false);
080        }
081
082        /**
083         * Creates a date-and-or-time property.
084         * @param date the date value
085         * @param hasTime true to include the date's time component, false if it's
086         * strictly a date
087         */
088        public DateOrTimeProperty(Calendar date, boolean hasTime) {
089                setDate(date, hasTime);
090        }
091
092        /**
093         * Creates a date-and-or-time property.
094         * @param partialDate the partial date value (vCard 4.0 only)
095         */
096        public DateOrTimeProperty(PartialDate partialDate) {
097                setPartialDate(partialDate);
098        }
099
100        /**
101         * Creates a date-and-or-time property.
102         * @param text the text value (vCard 4.0 only)
103         */
104        public DateOrTimeProperty(String text) {
105                setText(text);
106        }
107
108        /**
109         * Copy constructor.
110         * @param original the property to make a copy of
111         */
112        public DateOrTimeProperty(DateOrTimeProperty original) {
113                super(original);
114                text = original.text;
115                date = (original.date == null) ? null : (Calendar) original.date.clone();
116                partialDate = original.partialDate;
117                dateHasTime = original.dateHasTime;
118        }
119
120        /**
121         * Gets the date value.
122         * @return the date value or null if not set
123         */
124        public Date getDate() {
125                return (date == null) ? null : date.getTime();
126        }
127
128        /**
129         * Sets the value of this property to a complete date.
130         * @param date the date
131         * @param hasTime true to include the date's time component, false if it's
132         * strictly a date
133         */
134        public void setDate(Date date, boolean hasTime) {
135                setDate(VCardDateFormat.toCalendar(date), hasTime);
136        }
137
138        /**
139         * Sets the value of this property to a complete date.
140         * @param date the date
141         * @param hasTime true to include the date's time component, false if it's
142         * strictly a date
143         */
144        public void setDate(Calendar date, boolean hasTime) {
145                this.date = date;
146                this.dateHasTime = (date != null) && hasTime;
147                text = null;
148                partialDate = null;
149        }
150
151        /**
152         * <p>
153         * Gets the date value as a {@link Calendar} object. This is useful for
154         * retrieving the components of the original timestamp string from the
155         * source vCard data.
156         * </p>
157         * <p>
158         * Use {@link Calendar#isSet} to determine if a field was included in the
159         * original timestamp string. Calls to this method should be made before
160         * calling {@link Calendar#get} because calling latter method can cause
161         * unset fields to become populated (as mentioned in the
162         * {@link Calendar#isSet isSet} Javadocs).
163         * </p>
164         * <p>
165         * The calendar's timezone will be set to "GMT" if the "Z" suffix was used
166         * in the timestamp string. If a numeric offset was used, the timezone will
167         * look like "GMT-05:00". If no offset was specified, the timezone will be
168         * set to the local system's default timezone.
169         * </p>
170         * @return the date value or null if not set
171         */
172        public Calendar getCalendar() {
173                return (date == null) ? null : (Calendar) date.clone();
174        }
175
176        /**
177         * Gets the reduced accuracy or truncated date. This is only supported by
178         * vCard 4.0.
179         * @return the reduced accuracy or truncated date or null if not set
180         * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 p.12-14</a>
181         */
182        public PartialDate getPartialDate() {
183                return partialDate;
184        }
185
186        /**
187         * <p>
188         * Sets the value of this property to a reduced accuracy or truncated date.
189         * This is only supported by vCard 4.0.
190         * </p>
191         * 
192         * <pre class="brush:java">
193         * Birthday bday = new Birthday();
194         * bday.setPartialDate(PartialDate.date(null, 4, 20)); //April 20
195         * </pre>
196         * @param partialDate the reduced accuracy or truncated date
197         * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 p.12-14</a>
198         */
199        public void setPartialDate(PartialDate partialDate) {
200                this.partialDate = partialDate;
201                dateHasTime = (partialDate != null) && partialDate.hasTimeComponent();
202                text = null;
203                date = null;
204        }
205
206        /**
207         * Gets the text value of this type. This is only supported by vCard 4.0.
208         * @return the text value or null if not set
209         */
210        public String getText() {
211                return text;
212        }
213
214        /**
215         * Sets the value of this property to a text string. This is only supported
216         * by vCard 4.0.
217         * @param text the text value
218         */
219        public void setText(String text) {
220                this.text = text;
221                date = null;
222                partialDate = null;
223                dateHasTime = false;
224        }
225
226        /**
227         * Determines whether the "date" or "partialDate" fields have a time
228         * component.
229         * @return true if the date has a time component, false if it's strictly a
230         * date, and false if a text value is defined
231         */
232        public boolean hasTime() {
233                return dateHasTime;
234        }
235
236        /**
237         * <p>
238         * Gets the type of calendar that is used for a date or date-time property
239         * value.
240         * </p>
241         * <p>
242         * <b>Supported versions:</b> {@code 4.0}
243         * </p>
244         * @return the type of calendar or null if not found
245         * @see VCardParameters#getCalscale
246         */
247        public Calscale getCalscale() {
248                return parameters.getCalscale();
249        }
250
251        /**
252         * <p>
253         * Sets the type of calendar that is used for a date or date-time property
254         * value.
255         * </p>
256         * <p>
257         * <b>Supported versions:</b> {@code 4.0}
258         * </p>
259         * @param calscale the type of calendar or null to remove
260         * @see VCardParameters#setCalscale
261         */
262        public void setCalscale(Calscale calscale) {
263                parameters.setCalscale(calscale);
264        }
265
266        @Override
267        public String getLanguage() {
268                return super.getLanguage();
269        }
270
271        @Override
272        public void setLanguage(String language) {
273                super.setLanguage(language);
274        }
275
276        //@Override
277        public String getAltId() {
278                return parameters.getAltId();
279        }
280
281        //@Override
282        public void setAltId(String altId) {
283                parameters.setAltId(altId);
284        }
285
286        @Override
287        protected void _validate(List<ValidationWarning> warnings, VCardVersion version, VCard vcard) {
288                if (date == null && partialDate == null && text == null) {
289                        warnings.add(new ValidationWarning(8));
290                }
291
292                if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
293                        if (text != null) {
294                                warnings.add(new ValidationWarning(11));
295                        }
296                        if (partialDate != null) {
297                                warnings.add(new ValidationWarning(12));
298                        }
299                }
300        }
301
302        @Override
303        protected Map<String, Object> toStringValues() {
304                Map<String, Object> values = new LinkedHashMap<String, Object>();
305                values.put("text", text);
306                values.put("date", getDate());
307                values.put("dateHasTime", dateHasTime);
308                values.put("partialDate", partialDate);
309                return values;
310        }
311
312        @Override
313        public int hashCode() {
314                final int prime = 31;
315                int result = super.hashCode();
316
317                /*
318                 * Call getDate() method to compare the actual timestamp of the calendar
319                 * object.
320                 */
321                result = prime * result + ((getDate() == null) ? 0 : getDate().hashCode());
322
323                result = prime * result + (dateHasTime ? 1231 : 1237);
324                result = prime * result + ((partialDate == null) ? 0 : partialDate.hashCode());
325                result = prime * result + ((text == null) ? 0 : text.hashCode());
326                return result;
327        }
328
329        @Override
330        public boolean equals(Object obj) {
331                if (this == obj) return true;
332                if (!super.equals(obj)) return false;
333                DateOrTimeProperty other = (DateOrTimeProperty) obj;
334
335                /*
336                 * Call getDate() method to compare the actual timestamp of the calendar
337                 * object.
338                 */
339                if (getDate() == null) {
340                        if (other.getDate() != null) return false;
341                } else if (!getDate().equals(other.getDate())) return false;
342
343                if (dateHasTime != other.dateHasTime) return false;
344                if (partialDate == null) {
345                        if (other.partialDate != null) return false;
346                } else if (!partialDate.equals(other.partialDate)) return false;
347                if (text == null) {
348                        if (other.text != null) return false;
349                } else if (!text.equals(other.text)) return false;
350                return true;
351        }
