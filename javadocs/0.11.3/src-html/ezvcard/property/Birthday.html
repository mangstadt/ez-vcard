037/**
038 * <p>
039 * Defines the person's birthday.
040 * </p>
041 * 
042 * <p>
043 * <b>Code sample (creating)</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * VCard vcard = new VCard();
048 * 
049 * //date
050 * Calendar c = Calendar.getInstance();
051 * c.clear();
052 * c.set(Calendar.YEAR, 1912);
053 * c.set(Calendar.MONTH, Calendar.JUNE);
054 * c.set(Calendar.DAY_OF_MONTH, 23);
055 * Birthday bday = new Birthday(c.getTime());
056 * vcard.setBirthday(bday);
057 * 
058 * //partial date (e.g. just the month and date, vCard 4.0 only)
059 * bday = new Birthday(PartialDate.date(null, 6, 23)); //June 23
060 * vcard.setBirthday(bday);
061 * 
062 * //plain text value (vCard 4.0 only)
063 * bday = new Birthday("Don't even go there, dude...");
064 * vcard.setBirthday(bday);
065 * </pre>
066 * 
067 * <p>
068 * <b>Code sample (retrieving)</b>
069 * </p>
070 * 
071 * <pre class="brush:java">
072 * VCard vcard = ...
073 * Birthday bday = vcard.getBirthday();
074 * 
075 * Date date = bday.getDate();
076 * if (date != null) {
077 *   //property value is a date
078 * }
079 * 
080 * PartialDate partialDate = bday.getPartialDate();
081 * if (partialDate != null) {
082 *   //property value is a partial date
083 *   int year = partialDate.getYear();
084 *   int month = partialDate.getMonth();
085 * }
086 * 
087 * String text = bday.getText();
088 * if (text != null) {
089 *   //property value is plain text
090 * }
091 * </pre>
092 * 
093 * <p>
094 * <b>Property name:</b> {@code BDAY}
095 * </p>
096 * <p>
097 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
098 * </p>
099 * @author Michael Angstadt
100 * @see <a href="http://tools.ietf.org/html/rfc6350#page-30">RFC 6350 p.30</a>
101 * @see <a href="http://tools.ietf.org/html/rfc2426#page-11">RFC 2426 p.11</a>
102 * @see <a href="http://www.imc.org/pdi/vcard-21.doc">vCard 2.1 p.11</a>
103 */
104public class Birthday extends DateOrTimeProperty {
105        /**
106         * Creates a birthday property.
107         * @param date the birthday
108         */
109        public Birthday(Date date) {
110                super(date);
111        }
112
113        /**
114         * Creates a birthday property.
115         * @param date the birthday
116         * @param hasTime true to include the date's time component, false if it's
117         * strictly a date
118         */
119        public Birthday(Date date, boolean hasTime) {
120                super(date, hasTime);
121        }
122        
123        /**
124         * Creates a birthday property.
125         * @param date the birthday
126         */
127        public Birthday(Calendar date) {
128                super(date);
129        }
130
131        /**
132         * Creates a birthday property.
133         * @param date the birthday
134         * @param hasTime true to include the date's time component, false if it's
135         * strictly a date
136         */
137        public Birthday(Calendar date, boolean hasTime) {
138                super(date, hasTime);
139        }
140
141        /**
142         * Creates a birthday property.
143         * @param partialDate the birthday (vCard 4.0 only)
144         */
145        public Birthday(PartialDate partialDate) {
146                super(partialDate);
147        }
148
149        /**
150         * Creates a birthday property.
151         * @param text the text value (vCard 4.0 only)
152         */
153        public Birthday(String text) {
154                super(text);
155        }
156
157        /**
158         * Copy constructor.
159         * @param original the property to make a copy of
160         */
161        public Birthday(Birthday original) {
162                super(original);
163        }
164
165        @Override
166        public Birthday copy() {
167                return new Birthday(this);
168        }
169}
