036/**
037 * <p>
038 * Defines the function that the person plays within his or her organization.
039 * </p>
040 * 
041 * <p>
042 * <b>Code sample</b>
043 * </p>
044 * 
045 * <pre class="brush:java">
046 * VCard vcard = new VCard();
047 * 
048 * Role role = new Role("Project Lead");
049 * vcard.addRole(role);
050 * </pre>
051 * 
052 * <p>
053 * <b>Property name:</b> {@code ROLE}
054 * </p>
055 * <p>
056 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
057 * </p>
058 * @author Michael Angstadt
059 * @see <a href="http://tools.ietf.org/html/rfc6350#page-39">RFC 6350 p.39</a>
060 * @see <a href="http://tools.ietf.org/html/rfc2426#page-18">RFC 2426 p.18</a>
061 * @see <a href="http://www.imc.org/pdi/vcard-21.doc">vCard 2.1 p.17</a>
062 */
063public class Role extends TextProperty implements HasAltId {
064        /**
065         * Creates a role property.
066         * @param role the role (e.g. "Software Developer")
067         */
068        public Role(String role) {
069                super(role);
070        }
071
072        /**
073         * Copy constructor.
074         * @param original the property to make a copy of
075         */
076        public Role(Role original) {
077                super(original);
078        }
079
080        @Override
081        public String getLanguage() {
082                return super.getLanguage();
083        }
084
085        @Override
086        public void setLanguage(String language) {
087                super.setLanguage(language);
088        }
089
090        /**
091         * Gets the TYPE parameter.
092         * <p>
093         * <b>Supported versions:</b> {@code 4.0}
094         * </p>
095         * @return the TYPE value (typically, this will be either "work" or "home")
096         * or null if it doesn't exist
097         */
098        public String getType() {
099                return parameters.getType();
100        }
101
102        /**
103         * Sets the TYPE parameter.
104         * <p>
105         * <b>Supported versions:</b> {@code 4.0}
106         * </p>
107         * @param type the TYPE value (this should be either "work" or "home") or
108         * null to remove
109         */
110        public void setType(String type) {
111                parameters.setType(type);
112        }
113
114        @Override
115        public List<Pid> getPids() {
116                return super.getPids();
117        }
118
119        @Override
120        public Integer getPref() {
121                return super.getPref();
122        }
123
124        @Override
125        public void setPref(Integer pref) {
126                super.setPref(pref);
127        }
128
129        //@Override
130        public String getAltId() {
131                return parameters.getAltId();
132        }
133
134        //@Override
135        public void setAltId(String altId) {
136                parameters.setAltId(altId);
137        }
138
139        @Override
140        public Role copy() {
141                return new Role(this);
142        }
143}
