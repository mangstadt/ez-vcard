032/**
033 * Represents a property whose value is a URI.
034 * @author Michael Angstadt
035 */
036public class UriProperty extends TextProperty {
037        /**
038         * Creates a property whose value is a URI.
039         * @param uri the property value
040         */
041        public UriProperty(String uri) {
042                super(uri);
043        }
044
045        /**
046         * Copy constructor.
047         * @param original the property to make a copy of
048         */
049        public UriProperty(UriProperty original) {
050                super(original);
051        }
052}
