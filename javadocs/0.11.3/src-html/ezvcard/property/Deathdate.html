039/**
040 * <p>
041 * Defines the person's time of death.
042 * </p>
043 * 
044 * <p>
045 * <b>Code sample (creating)</b>
046 * </p>
047 * 
048 * <pre class="brush:java">
049 * VCard vcard = new VCard();
050 * 
051 * //date
052 * Calendar c = Calendar.getInstance();
053 * c.clear();
054 * c.set(Calendar.YEAR, 1954);
055 * c.set(Calendar.MONTH, Calendar.JUNE);
056 * c.set(Calendar.DAY_OF_MONTH, 7);
057 * Deathdate deathdate = new Deathdate(c.getTime());
058 * vcard.setDeathdate(deathdate);
059 * 
060 * //partial date (e.g. just the month and date)
061 * deathdate = new Deathdate(PartialDate.date(null, 6, 7)); //June 7
062 * vcard.setDeathdate(deathdate);
063 * 
064 * //plain text value
065 * deathdate = new Deathdate("In the 1950s");
066 * vcard.setDeathdate(deathdate);
067 * </pre>
068 * 
069 * <p>
070 * <b>Code sample (retrieving)</b>
071 * </p>
072 * 
073 * <pre class="brush:java">
074 * VCard vcard = ...
075 * Deathdate deathdate = vcard.getDeathdate();
076 * 
077 * Date date = deathdate.getDate();
078 * if (date != null) {
079 *   //property value is a date
080 * }
081 * 
082 * PartialDate partialDate = deathdate.getPartialDate();
083 * if (partialDate != null) {
084 *   //property value is a partial date
085 *   int year = partialDate.getYear();
086 *   int month = partialDate.getMonth();
087 * }
088 * 
089 * String text = deathdate.getText();
090 * if (text != null) {
091 *   //property value is plain text
092 * }
093 * </pre>
094 * 
095 * <p>
096 * <b>Property name:</b> {@code DEATHDATE}
097 * </p>
098 * <p>
099 * <b>Supported versions:</b> {@code 4.0}
100 * </p>
101 * @author Michael Angstadt
102 * @see <a href="http://tools.ietf.org/html/rfc6474#page-4">RFC 6474 p.4</a>
103 */
104@SupportedVersions(VCardVersion.V4_0)
105public class Deathdate extends DateOrTimeProperty {
106        /**
107         * Creates a deathdate property.
108         * @param date the deathdate
109         */
110        public Deathdate(Date date) {
111                super(date);
112        }
113
114        /**
115         * Creates a deathdate property.
116         * @param date the deathdate
117         * @param hasTime true to include the date's time component, false if it's
118         * strictly a date
119         */
120        public Deathdate(Date date, boolean hasTime) {
121                super(date, hasTime);
122        }
123        
124        /**
125         * Creates a deathdate property.
126         * @param date the deathdate
127         */
128        public Deathdate(Calendar date) {
129                super(date);
130        }
131
132        /**
133         * Creates a deathdate property.
134         * @param date the deathdate
135         * @param hasTime true to include the date's time component, false if it's
136         * strictly a date
137         */
138        public Deathdate(Calendar date, boolean hasTime) {
139                super(date, hasTime);
140        }
141
142        /**
143         * Creates a deathdate property.
144         * @param partialDate the deathdate (vCard 4.0 only)
145         */
146        public Deathdate(PartialDate partialDate) {
147                super(partialDate);
148        }
149
150        /**
151         * Creates a deathdate property.
152         * @param text the text value (vCard 4.0 only)
153         */
154        public Deathdate(String text) {
155                super(text);
156        }
157
158        /**
159         * Copy constructor.
160         * @param original the property to make a copy of
161         */
162        public Deathdate(Deathdate original) {
163                super(original);
164        }
165
166        @Override
167        public Deathdate copy() {
168                return new Deathdate(this);
169        }
170}
