043    /**
044     * Represents a property whose value contains binary data (for example,
045     * {@link Photo}).
046     * @author Michael Angstadt
047     * 
048     * @param <T> the class used for representing the content type of the resource
049     */
050    public abstract class BinaryProperty<T extends MediaTypeParameter> extends VCardProperty implements HasAltId {
051            /**
052             * The decoded data.
053             */
054            protected byte[] data;
055    
056            /**
057             * The URL to the resource.
058             */
059            protected String url;
060    
061            /**
062             * The content type of the resource (for example, a JPEG image).
063             */
064            protected T contentType;
065    
066            public BinaryProperty() {
067                    //empty
068            }
069    
070            /**
071             * Creates a binary property.
072             * @param url the URL to the resource
073             * @param type the content type
074             */
075            public BinaryProperty(String url, T type) {
076                    setUrl(url, type);
077            }
078    
079            /**
080             * Creates a binary property.
081             * @param data the binary data
082             * @param type the content type
083             */
084            public BinaryProperty(byte[] data, T type) {
085                    setData(data, type);
086            }
087    
088            /**
089             * Creates a binary property.
090             * @param in an input stream to the binary data (will be closed)
091             * @param type the content type
092             * @throws IOException if there is a problem reading from the input stream
093             */
094            public BinaryProperty(InputStream in, T type) throws IOException {
095                    this(IOUtils.toByteArray(in, true), type);
096            }
097    
098            /**
099             * Creates a binary property.
100             * @param file the file containing the binary data
101             * @param type the content type
102             * @throws IOException if there is a problem reading from the file
103             */
104            public BinaryProperty(File file, T type) throws IOException {
105                    this(new FileInputStream(file), type);
106            }
107    
108            /**
109             * Gets the binary data of the resource.
110             * @return the binary data or null if there is none
111             */
112            public byte[] getData() {
113                    return data;
114            }
115    
116            /**
117             * Sets the binary data of the resource.
118             * @param data the binary data
119             * @param type the content type (e.g. "JPEG image")
120             */
121            public void setData(byte[] data, T type) {
122                    this.url = null;
123                    this.data = data;
124                    setContentType(type);
125            }
126    
127            /**
128             * Gets the URL to the resource
129             * @return the URL or null if there is none
130             */
131            public String getUrl() {
132                    return url;
133            }
134    
135            /**
136             * Sets the URL to the resource.
137             * @param url the URL
138             * @param type the content type (e.g. "JPEG image")
139             */
140            public void setUrl(String url, T type) {
141                    this.url = url;
142                    this.data = null;
143                    setContentType(type);
144            }
145    
146            /**
147             * Gets the content type of the resource.
148             * @return the content type (e.g. "JPEG image")
149             */
150            public T getContentType() {
151                    return contentType;
152            }
153    
154            /**
155             * Sets the content type of the resource.
156             * @param contentType the content type (e.g. "JPEG image")
157             */
158            public void setContentType(T contentType) {
159                    this.contentType = contentType;
160            }
161    
162            /**
163             * Gets the vCard 4.0 TYPE parameter. This should NOT be used to get the
164             * TYPE parameter for 2.1/3.0 vCards. Use {@link #getContentType} instead.
165             * <p>
166             * <b>Supported versions:</b> {@code 4.0}
167             * </p>
168             * @return the TYPE value (typically, this will be either "work" or "home")
169             * or null if it doesn't exist
170             */
171            public String getType() {
172                    return parameters.getType();
173            }
174    
175            /**
176             * Sets the vCard 4.0 TYPE parameter. This should NOT be used to set the
177             * TYPE parameter for 2.1/3.0 vCards. Use {@link #setContentType} instead.
178             * <p>
179             * <b>Supported versions:</b> {@code 4.0}
180             * </p>
181             * @param type the TYPE value (should be either "work" or "home") or null to
182             * remove
183             */
184            public void setType(String type) {
185                    parameters.setType(type);
186            }
187    
188            @Override
189            public List<Integer[]> getPids() {
190                    return super.getPids();
191            }
192    
193            @Override
194            public void addPid(int localId, int clientPidMapRef) {
195                    super.addPid(localId, clientPidMapRef);
196            }
197    
198            @Override
199            public void removePids() {
200                    super.removePids();
201            }
202    
203            @Override
204            public Integer getPref() {
205                    return super.getPref();
206            }
207    
208            @Override
209            public void setPref(Integer pref) {
210                    super.setPref(pref);
211            }
212    
213            //@Override
214            public String getAltId() {
215                    return parameters.getAltId();
216            }
217    
218            //@Override
219            public void setAltId(String altId) {
220                    parameters.setAltId(altId);
221            }
222    
223            @Override
224            protected void _validate(List<String> warnings, VCardVersion version, VCard vcard) {
225                    if (url == null && data == null) {
226                            warnings.add("Property has neither a URL nor binary data attached to it.");
227                    }
228            }
229    }
