037/**
038 * Records vCard parse warnings.
039 * @author Michael Angstadt
040 */
041public class ParseWarnings {
042        private final List<String> warnings = new ArrayList<String>();
043
044        /**
045         * Adds a parse warning.
046         * @param line the line number or null if unknown
047         * @param propertyName the property name or null if N/A
048         * @param code the message code from the resource bundle
049         * @param args the message arguments
050         */
051        public void add(Integer line, String propertyName, int code, Object... args) {
052                String message = Messages.INSTANCE.getParseMessage(code, args);
053                add(line, propertyName, message);
054        }
055
056        /**
057         * Adds a parse warning.
058         * @param line the line number or null if unknown
059         * @param propertyName the property name or null if N/A
060         * @param message the warning message
061         */
062        public void add(Integer line, String propertyName, String message) {
063                if (line == null && propertyName == null) {
064                        warnings.add(message);
065                        return;
066                }
067
068                int code;
069                if (line == null && propertyName != null) {
070                        code = 35;
071                } else if (line != null && propertyName == null) {
072                        code = 37;
073                } else {
074                        code = 36;
075                }
076
077                String warning = Messages.INSTANCE.getParseMessage(code, line, propertyName, message);
078                warnings.add(warning);
079        }
080
081        /**
082         * Creates a copy of this warnings list.
083         * @return the copy
084         */
085        public List<String> copy() {
086                return new ArrayList<String>(warnings);
087        }
088
089        /**
090         * Clears the warnings list.
091         */
092        public void clear() {
093                warnings.clear();
094        }
095}
