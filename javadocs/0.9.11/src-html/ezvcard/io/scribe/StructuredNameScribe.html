039/**
040 * Marshals {@link StructuredName} properties.
041 * @author Michael Angstadt
042 */
043public class StructuredNameScribe extends VCardPropertyScribe<StructuredName> {
044        public StructuredNameScribe() {
045                super(StructuredName.class, "N");
046        }
047
048        @Override
049        protected VCardDataType _defaultDataType(VCardVersion version) {
050                return VCardDataType.TEXT;
051        }
052
053        @Override
054        protected String _writeText(StructuredName property, WriteContext context) {
055                //@formatter:off
056                return structured(context.isIncludeTrailingSemicolons(), new Object[] {
057                        property.getFamily(),
058                        property.getGiven(),
059                        property.getAdditionalNames(),
060                        property.getPrefixes(),
061                        property.getSuffixes()
062                });
063                //@formatter:on
064        }
065
066        @Override
067        protected StructuredName _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
068                StructuredName property = new StructuredName();
069                StructuredIterator it = structured(value);
070
071                property.setFamily(it.nextString());
072                property.setGiven(it.nextString());
073                property.getAdditionalNames().addAll(it.nextComponent());
074                property.getPrefixes().addAll(it.nextComponent());
075                property.getSuffixes().addAll(it.nextComponent());
076
077                return property;
078        }
079
080        @Override
081        protected void _writeXml(StructuredName property, XCardElement parent) {
082                parent.append("surname", property.getFamily()); //the XML element still needs to be printed if value == null
083                parent.append("given", property.getGiven());
084                parent.append("additional", property.getAdditionalNames());
085                parent.append("prefix", property.getPrefixes());
086                parent.append("suffix", property.getSuffixes());
087        }
088
089        @Override
090        protected StructuredName _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
091                StructuredName property = new StructuredName();
092
093                property.setFamily(s(element.first("surname")));
094                property.setGiven(s(element.first("given")));
095                property.getAdditionalNames().addAll(element.all("additional"));
096                property.getPrefixes().addAll(element.all("prefix"));
097                property.getSuffixes().addAll(element.all("suffix"));
098
099                return property;
100        }
101
102        private String s(String value) {
103                return (value == null || value.length() == 0) ? null : value;
104        }
105
106        @Override
107        protected StructuredName _parseHtml(HCardElement element, List<String> warnings) {
108                StructuredName property = new StructuredName();
109
110                property.setFamily(s(element.firstValue("family-name")));
111                property.setGiven(s(element.firstValue("given-name")));
112                property.getAdditionalNames().addAll(element.allValues("additional-name"));
113                property.getPrefixes().addAll(element.allValues("honorific-prefix"));
114                property.getSuffixes().addAll(element.allValues("honorific-suffix"));
115
116                return property;
117        }
118
119        @Override
120        protected JCardValue _writeJson(StructuredName property) {
121                return JCardValue.structured(property.getFamily(), property.getGiven(), property.getAdditionalNames(), property.getPrefixes(), property.getSuffixes());
122        }
123
124        @Override
125        protected StructuredName _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
126                StructuredName property = new StructuredName();
127                StructuredIterator it = structured(value);
128
129                property.setFamily(it.nextString());
130                property.setGiven(it.nextString());
131                property.getAdditionalNames().addAll(it.nextComponent());
132                property.getPrefixes().addAll(it.nextComponent());
133                property.getSuffixes().addAll(it.nextComponent());
134
135                return property;
136        }
137}
