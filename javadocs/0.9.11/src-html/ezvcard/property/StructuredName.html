037/**
038 * <p>
039 * Defines the individual components of the person's name.
040 * </p>
041 * 
042 * <p>
043 * <b>Code sample</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * VCard vcard = new VCard();
048 * 
049 * StructuredName n = new StructuredName();
050 * n.setFamily(&quot;House&quot;);
051 * n.setGiven(&quot;Gregory&quot;);
052 * n.getPrefixes().add(&quot;Dr&quot;);
053 * n.getSuffixes().add(&quot;MD&quot;);
054 * vcard.setStructuredName(n);
055 * </pre>
056 * 
057 * <p>
058 * <b>Property name:</b> {@code N}
059 * </p>
060 * <p>
061 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
062 * </p>
063 * @author Michael Angstadt
064 * @see <a href="http://tools.ietf.org/html/rfc6350#page-29">RFC 6350 p.29</a>
065 * @see <a href="http://tools.ietf.org/html/rfc2426#page-9">RFC 2426 p.9</a>
066 * @see <a href="http://www.imc.org/pdi/vcard-21.doc">vCard 2.1 p.9</a>
067 */
068public class StructuredName extends VCardProperty implements HasAltId {
069        private String family;
070        private String given;
071        private final List<String> additional;
072        private final List<String> prefixes;
073        private final List<String> suffixes;
074
075        public StructuredName() {
076                additional = new ArrayList<String>();
077                prefixes = new ArrayList<String>();
078                suffixes = new ArrayList<String>();
079        }
080
081        /**
082         * Copy constructor.
083         * @param original the property to make a copy of
084         */
085        public StructuredName(StructuredName original) {
086                super(original);
087                family = original.family;
088                given = original.given;
089                additional = new ArrayList<String>(original.additional);
090                prefixes = new ArrayList<String>(original.prefixes);
091                suffixes = new ArrayList<String>(original.suffixes);
092        }
093
094        /**
095         * Gets the family name (aka "surname" or "last name").
096         * @return the family name or null if not set
097         */
098        public String getFamily() {
099                return family;
100        }
101
102        /**
103         * Sets the family name (aka "surname" or "last name").
104         * @param family the family name or null to remove
105         */
106        public void setFamily(String family) {
107                this.family = family;
108        }
109
110        /**
111         * Gets the given name (aka "first name").
112         * @return the given name or null if not set
113         */
114        public String getGiven() {
115                return given;
116        }
117
118        /**
119         * Sets the given name (aka "first name").
120         * @param given the given name or null to remove
121         */
122        public void setGiven(String given) {
123                this.given = given;
124        }
125
126        /**
127         * Gets the list that stores additional names the person goes by (for
128         * example, a middle name).
129         * @return the additional names (this list is mutable)
130         */
131        public List<String> getAdditionalNames() {
132                return additional;
133        }
134
135        /**
136         * Gets the list that stores the person's honorific prefixes.
137         * @return the prefixes (e.g. "Dr.", "Mr.") (this list is mutable)
138         */
139        public List<String> getPrefixes() {
140                return prefixes;
141        }
142
143        /**
144         * Gets the list that stores the person's honorary suffixes.
145         * @return the suffixes (e.g. "M.D.", "Jr.") (this list is mutable)
146         */
147        public List<String> getSuffixes() {
148                return suffixes;
149        }
150
151        /**
152         * <p>
153         * Gets the list that holds string(s) which define how to sort the vCard.
154         * </p>
155         * <p>
156         * 2.1 and 3.0 vCards should use the {@link SortString} property instead.
157         * </p>
158         * <p>
159         * <b>Supported versions:</b> {@code 4.0}
160         * </p>
161         * @return the sort string(s) (this list is mutable). For example, if the
162         * family name is "d'Aboville" and the given name is "Christine", the sort
163         * strings might be ["Aboville", "Christine"].
164         */
165        public List<String> getSortAs() {
166                return parameters.getSortAs();
167        }
168
169        /**
170         * <p>
171         * Defines a sortable version of the person's name.
172         * </p>
173         * <p>
174         * 2.1 and 3.0 vCards should use the {@link SortString} property instead.
175         * </p>
176         * <p>
177         * <b>Supported versions:</b> {@code 4.0}
178         * </p>
179         * @param family the sortable version of the family name (for example,
180         * "Adboville" if the family name is "d'Aboville") or null to remove
181         */
182        public void setSortAs(String family) {
183                parameters.setSortAs(family);
184        }
185
186        /**
187         * <p>
188         * Defines a sortable version of the person's name.
189         * </p>
190         * <p>
191         * 2.1 and 3.0 vCards should use the {@link SortString} property instead.
192         * </p>
193         * <p>
194         * <b>Supported versions:</b> {@code 4.0}
195         * </p>
196         * @param family the sortable version of the family name (for example,
197         * "Adboville" if the family name is "d'Aboville")
198         * @param given the sortable version of the given name
199         */
200        public void setSortAs(String family, String given) {
201                parameters.setSortAs(family, given);
202        }
203
204        @Override
205        public String getLanguage() {
206                return super.getLanguage();
207        }
208
209        @Override
210        public void setLanguage(String language) {
211                super.setLanguage(language);
212        }
213
214        //@Override
215        public String getAltId() {
216                return parameters.getAltId();
217        }
218
219        //@Override
220        public void setAltId(String altId) {
221                parameters.setAltId(altId);
222        }
223
224        @Override
225        protected Map<String, Object> toStringValues() {
226                Map<String, Object> values = new LinkedHashMap<String, Object>();
227                values.put("family", family);
228                values.put("given", given);
229                values.put("additional", additional);
230                values.put("prefixes", prefixes);
231                values.put("suffixes", suffixes);
232                return values;
233        }
234
235        @Override
236        public StructuredName copy() {
237                return new StructuredName(this);
238        }
239
240        @Override
241        public int hashCode() {
242                final int prime = 31;
243                int result = super.hashCode();
244                result = prime * result + ((additional == null) ? 0 : additional.hashCode());
245                result = prime * result + ((family == null) ? 0 : family.hashCode());
246                result = prime * result + ((given == null) ? 0 : given.hashCode());
247                result = prime * result + ((prefixes == null) ? 0 : prefixes.hashCode());
248                result = prime * result + ((suffixes == null) ? 0 : suffixes.hashCode());
249                return result;
250        }
251
252        @Override
253        public boolean equals(Object obj) {
254                if (this == obj) return true;
255                if (!super.equals(obj)) return false;
256                StructuredName other = (StructuredName) obj;
257                if (additional == null) {
258                        if (other.additional != null) return false;
259                } else if (!additional.equals(other.additional)) return false;
260                if (family == null) {
261                        if (other.family != null) return false;
262                } else if (!family.equals(other.family)) return false;
263                if (given == null) {
264                        if (other.given != null) return false;
265                } else if (!given.equals(other.given)) return false;
266                if (prefixes == null) {
267                        if (other.prefixes != null) return false;
268                } else if (!prefixes.equals(other.prefixes)) return false;
269                if (suffixes == null) {
270                        if (other.suffixes != null) return false;
271                } else if (!suffixes.equals(other.suffixes)) return false;
272                return true;
273        }
274}
