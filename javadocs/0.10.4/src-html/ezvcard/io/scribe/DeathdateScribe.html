033/**
034 * Marshals {@link Deathdate} properties.
035 * @author Michael Angstadt
036 */
037public class DeathdateScribe extends DateOrTimePropertyScribe<Deathdate> {
038        public DeathdateScribe() {
039                super(Deathdate.class, "DEATHDATE");
040        }
041
042        @Override
043        protected Deathdate newInstance(String text) {
044                return new Deathdate(text);
045        }
046
047        @Override
048        protected Deathdate newInstance(Date date, boolean hasTime) {
049                return new Deathdate(date, hasTime);
050        }
051
052        @Override
053        protected Deathdate newInstance(PartialDate partialDate) {
054                return new Deathdate(partialDate);
055        }
056}
