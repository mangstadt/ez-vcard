044/**
045 * A mailing address.
046 * 
047 * <p>
048 * <b>Adding an address</b>
049 * </p>
050 * 
051 * <pre>
052 * VCard vcard = new VCard();
053 * AddressType adr = new AddressType();
054 * adr.setStreetAddress(&quot;123 Main St.&quot;);
055 * adr.setLocality(&quot;Austin&quot;);
056 * adr.setRegion(&quot;TX&quot;);
057 * adr.setPostalCode(&quot;12345&quot;);
058 * adr.setCountry(&quot;USA&quot;);
059 * adr.addType(AddressTypeParameter.WORK);
060 * adr.addType(AddressTypeParameter.DOM);
061 * 
062 * //optionally, provide the exact text to print out on the mailing label
063 * adr.setLabel(&quot;123 Main St.\nAustin, Tx 12345\nUSA&quot;);
064 * 
065 * vcard.addAddress(adr);
066 * </pre>
067 * 
068 * <p>
069 * <b>Getting the addresses</b>
070 * </p>
071 * 
072 * <pre>
073 * VCard vcard = ...
074 * for (AddressType adr : vcard.getAddresses()){
075 *   ...
076 * }
077 * </pre>
078 * 
079 * <p>
080 * vCard property name: ADR
081 * </p>
082 * <p>
083 * vCard versions: 2.1, 3.0, 4.0
084 * </p>
085 * @author Michael Angstadt
086 */

373         * Gets the timezone that's associated with this address.
374         * <p>
375         * vCard versions: 4.0
376         * </p>
377         * @return the timezone (e.g. "America/New_York") or null if it doesn't
378         * exist
379         */
380        public String getTimezone() {
381                String value = subTypes.getFirst("TZ");
382                if (value.matches("(?i)tz:.*")) {
383                        //remove the "tz:"
384                        value = (value.length() > 3) ? value.substring(3) : "";
385                }
386                return value;
387        }

400        @Override
401        protected void doMarshalSubTypes(VCardSubTypes copy, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode, VCard vcard) {
402                //replace "TYPE=pref" with "PREF=1"
403                if (version == VCardVersion.V4_0) {
404                        if (getTypes().contains(AddressTypeParameter.PREF)) {
405                                copy.removeType(AddressTypeParameter.PREF.getValue());
406                                copy.setPref(1);
407                        }
408                } else {
409                        copy.setPref(null);
410
411                        //find the ADR with the lowest PREF value in the vCard
412                        AddressType mostPreferred = null;
413                        for (AddressType adr : vcard.getAddresses()) {
414                                Integer pref = adr.getPref();
415                                if (pref != null) {
416                                        if (mostPreferred == null || pref < mostPreferred.getPref()) {
417                                                mostPreferred = adr;
418                                        }
419                                }
420                        }
421                        if (this == mostPreferred) {
422                                copy.addType(AddressTypeParameter.PREF.getValue());
423                        }
424                }
425
426                //remove the LABEL parameter
427                if (version != VCardVersion.V4_0) {
428                        copy.removeAll("LABEL");
429                }
430        }

432        @Override
433        protected void doMarshalValue(StringBuilder sb, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
434                if (poBox != null) {
435                        sb.append(VCardStringUtils.escape(poBox));
436                }
437                sb.append(';');
438
439                if (extendedAddress != null) {
440                        sb.append(VCardStringUtils.escape(extendedAddress));
441                }
442                sb.append(';');
443
444                if (streetAddress != null) {
445                        sb.append(VCardStringUtils.escape(streetAddress));
446                }
447                sb.append(';');
448
449                if (locality != null) {
450                        sb.append(VCardStringUtils.escape(locality));
451                }
452                sb.append(';');
453
454                if (region != null) {
455                        sb.append(VCardStringUtils.escape(region));
456                }
457                sb.append(';');
458
459                if (postalCode != null) {
460                        sb.append(VCardStringUtils.escape(postalCode));
461                }
462                sb.append(';');
463
464                if (country != null) {
465                        sb.append(VCardStringUtils.escape(country));
466                }
467        }

469        @Override
470        protected void doUnmarshalValue(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
471                String split[] = VCardStringUtils.splitBy(value, ';', false, true);
472
473                int i = 0;
474
475                poBox = (split.length > i && split[i].length() > 0) ? split[i] : null;
476                i++;
477
478                extendedAddress = (split.length > i && split[i].length() > 0) ? split[i] : null;
479                i++;
480
481                streetAddress = (split.length > i && split[i].length() > 0) ? split[i] : null;
482                i++;
483
484                locality = (split.length > i && split[i].length() > 0) ? split[i] : null;
485                i++;
486
487                region = (split.length > i && split[i].length() > 0) ? split[i] : null;
488                i++;
489
490                postalCode = (split.length > i && split[i].length() > 0) ? split[i] : null;
491                i++;
492
493                country = (split.length > i && split[i].length() > 0) ? split[i] : null;
494        }

496        @Override
497        protected void doMarshalValue(XCardElement parent, List<String> warnings, CompatibilityMode compatibilityMode) {
498                if (poBox != null) {
499                        parent.append("pobox", poBox);
500                }
501                if (extendedAddress != null) {
502                        parent.append("ext", extendedAddress);
503                }
504                if (streetAddress != null) {
505                        parent.append("street", streetAddress);
506                }
507                if (locality != null) {
508                        parent.append("locality", locality);
509                }
510                if (region != null) {
511                        parent.append("region", region);
512                }
513                if (postalCode != null) {
514                        parent.append("code", postalCode);
515                }
516                if (country != null) {
517                        parent.append("country", country);
518                }
519        }

532        @Override
533        protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
534                poBox = element.firstValue("post-office-box");
535                extendedAddress = element.firstValue("extended-address");
536                streetAddress = element.firstValue("street-address");
537                locality = element.firstValue("locality");
538                region = element.firstValue("region");
539                postalCode = element.firstValue("postal-code");
540                country = element.firstValue("country-name");
541                List<String> types = element.types();
542                for (String type : types) {
543                        subTypes.addType(type);
544                }
545        }
