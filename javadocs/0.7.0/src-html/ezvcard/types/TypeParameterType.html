034/**
035 * Represents a type that has a TYPE parameter.
036 * @author Michael Angstadt
037 */
038public abstract class TypeParameterType<T extends TypeParameter> extends VCardType {
039        public TypeParameterType(String name) {
040                super(name);
041        }
042        
043        /**
044         * Builds an object that wraps the value of the TYPE parameter.
045         * @param type the value of the TYPE parameter
046         * @return the object wrapper
047         */
048        protected abstract T buildTypeObj(String type);
049}
