035/**
036 * <p>
037 * Defines the location of the person's death.
038 * </p>
039 * 
040 * <p>
041 * <b>Code sample (creating)</b>
042 * </p>
043 * 
044 * <pre class="brush:java">
045 * VCard vcard = new VCard();
046 * 
047 * //text
048 * Deathplace deathplace = new Deathplace("Wilmslow, Cheshire, England");
049 * vcard.setDeathplace(deathplace);
050 * 
051 * //geo coordinates
052 * deathplace = new Deathplace(53.325, -2.239);
053 * vcard.setDeathplace(deathplace);
054 * 
055 * //URI
056 * deathplace = new Deathplace();
057 * deathplace.setUri("http://en.wikipedia.org/wiki/Wilmslow");
058 * vcard.setDeathplace(deathplace);
059 * </pre>
060 * 
061 * <p>
062 * <b>Code sample (retrieving)</b>
063 * </p>
064 * 
065 * <pre class="brush:java">
066 * VCard vcard = ...
067 * Deathplace deathplace = vcard.getDeathplace();
068 * 
069 * String text = deathplace.getText();
070 * if (text != null) {
071 *   //property value is plain text
072 * }
073 * 
074 * Double latitude = deathplace.getLatitude();
075 * Double longitude = deathplace.getLongitude();
076 * if (latitude != null) {
077 *   //property value is a set of geo coordinates
078 * }
079 * 
080 * String uri = deathplace.getUri();
081 * if (uri != null) {
082 *   //property value is a URI
083 * }
084 * </pre>
085 * 
086 * <p>
087 * <b>Property name:</b> {@code DEATHPLACE}
088 * </p>
089 * <p>
090 * <b>Supported versions:</b> {@code 4.0}
091 * </p>
092 * @author Michael Angstadt
093 * @see <a href="http://tools.ietf.org/html/rfc6474#page-3">RFC 6474 p.3</a>
094 */
095@SupportedVersions(VCardVersion.V4_0)
096public class Deathplace extends PlaceProperty {
097        /**
098         * Creates a new deathplace property.
099         */
100        public Deathplace() {
101                super();
102        }
103
104        /**
105         * Creates a new deathplace property.
106         * @param latitude the latitude coordinate of the place
107         * @param longitude the longitude coordinate of the place
108         */
109        public Deathplace(double latitude, double longitude) {
110                super(latitude, longitude);
111        }
112
113        /**
114         * Creates a new deathplace property.
115         * @param text a text value representing the place
116         */
117        public Deathplace(String text) {
118                super(text);
119        }
120
121        /**
122         * Copy constructor.
123         * @param original the property to make a copy of
124         */
125        public Deathplace(Deathplace original) {
126                super(original);
127        }
128
129        @Override
130        public Deathplace copy() {
131                return new Deathplace(this);
132        }
133}
