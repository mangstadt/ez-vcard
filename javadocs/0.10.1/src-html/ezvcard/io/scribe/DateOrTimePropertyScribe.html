045/**
046 * Marshals properties with date-time values.
047 * @author Michael Angstadt
048 * @param <T> the property class
049 */
050public abstract class DateOrTimePropertyScribe<T extends DateOrTimeProperty> extends VCardPropertyScribe<T> {
051        public DateOrTimePropertyScribe(Class<T> clazz, String propertyName) {
052                super(clazz, propertyName);
053        }
054
055        @Override
056        protected VCardDataType _defaultDataType(VCardVersion version) {
057                switch (version) {
058                case V2_1:
059                case V3_0:
060                        return null;
061                case V4_0:
062                        return VCardDataType.DATE_AND_OR_TIME;
063                }
064                return null;
065        }
066
067        @Override
068        protected VCardDataType _dataType(T property, VCardVersion version) {
069                switch (version) {
070                case V2_1:
071                case V3_0:
072                        return null;
073                case V4_0:
074                        if (property.getText() != null) {
075                                return VCardDataType.TEXT;
076                        }
077                        if (property.getDate() != null || property.getPartialDate() != null) {
078                                return property.hasTime() ? VCardDataType.DATE_TIME : VCardDataType.DATE;
079                        }
080                        return VCardDataType.DATE_AND_OR_TIME;
081                }
082                return null;
083        }
084
085        @Override
086        protected String _writeText(T property, WriteContext context) {
087                VCardVersion version = context.getVersion();
088                Date date = property.getDate();
089                if (date != null) {
090                        boolean extended = (version == VCardVersion.V3_0);
091                        return date(date).time(property.hasTime()).extended(extended).utc(false).write();
092                }
093
094                if (version == VCardVersion.V4_0) {
095                        String text = property.getText();
096                        if (text != null) {
097                                return VObjectPropertyValues.escape(text);
098                        }
099
100                        PartialDate partialDate = property.getPartialDate();
101                        if (partialDate != null) {
102                                return partialDate.toISO8601(false);
103                        }
104                }
105
106                return "";
107        }
108
109        @Override
110        protected T _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
111                value = VObjectPropertyValues.unescape(value);
112                if (version == VCardVersion.V4_0 && dataType == VCardDataType.TEXT) {
113                        return newInstance(value);
114                }
115
116                return parse(value, version, warnings);
117        }
118
119        @Override
120        protected void _writeXml(T property, XCardElement parent) {
121                Date date = property.getDate();
122                if (date != null) {
123                        boolean hasTime = property.hasTime();
124                        String value = date(date).time(hasTime).extended(false).utc(false).write();
125
126                        VCardDataType dataType = hasTime ? VCardDataType.DATE_TIME : VCardDataType.DATE;
127
128                        parent.append(dataType, value);
129                        return;
130                }
131
132                PartialDate partialDate = property.getPartialDate();
133                if (partialDate != null) {
134                        VCardDataType dataType;
135                        if (partialDate.hasTimeComponent() && partialDate.hasDateComponent()) {
136                                dataType = VCardDataType.DATE_TIME;
137                        } else if (partialDate.hasTimeComponent()) {
138                                dataType = VCardDataType.TIME;
139                        } else if (partialDate.hasDateComponent()) {
140                                dataType = VCardDataType.DATE;
141                        } else {
142                                dataType = VCardDataType.DATE_AND_OR_TIME;
143                        }
144
145                        parent.append(dataType, partialDate.toISO8601(false));
146                        return;
147                }
148
149                String text = property.getText();
150                if (text != null) {
151                        parent.append(VCardDataType.TEXT, text);
152                        return;
153                }
154
155                parent.append(VCardDataType.DATE_AND_OR_TIME, "");
156        }
157
158        @Override
159        protected T _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
160                String value = element.first(VCardDataType.DATE, VCardDataType.DATE_TIME, VCardDataType.DATE_AND_OR_TIME);
161                if (value != null) {
162                        return parse(value, element.version(), warnings);
163                }
164
165                value = element.first(VCardDataType.TEXT);
166                if (value != null) {
167                        return newInstance(value);
168                }
169
170                throw missingXmlElements(VCardDataType.DATE, VCardDataType.DATE_TIME, VCardDataType.DATE_AND_OR_TIME, VCardDataType.TEXT);
171        }
172
173        @Override
174        protected T _parseHtml(HCardElement element, List<String> warnings) {
175                String value = null;
176                if ("time".equals(element.tagName())) {
177                        String datetime = element.attr("datetime");
178                        if (datetime.length() > 0) {
179                                value = datetime;
180                        }
181                }
182                if (value == null) {
183                        value = element.value();
184                }
185                return parse(value, VCardVersion.V3_0, warnings);
186        }
187
188        @Override
189        protected JCardValue _writeJson(T property) {
190                Date date = property.getDate();
191                if (date != null) {
192                        boolean hasTime = property.hasTime();
193                        String value = date(date).time(hasTime).extended(true).utc(false).write();
194                        return JCardValue.single(value);
195                }
196
197                PartialDate partialDate = property.getPartialDate();
198                if (partialDate != null) {
199                        String value = partialDate.toISO8601(true);
200                        return JCardValue.single(value);
201                }
202
203                String text = property.getText();
204                if (text != null) {
205                        return JCardValue.single(text);
206                }
207
208                return JCardValue.single("");
209        }
210
211        @Override
212        protected T _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
213                String valueStr = value.asSingle();
214                if (dataType == VCardDataType.TEXT) {
215                        return newInstance(valueStr);
216                }
217
218                return parse(valueStr, VCardVersion.V4_0, warnings);
219        }
220
221        private T parse(String value, VCardVersion version, List<String> warnings) {
222                try {
223                        boolean hasTime = value.contains("T");
224                        return newInstance(date(value), hasTime);
225                } catch (IllegalArgumentException e) {
226                        if (version == VCardVersion.V2_1 || version == VCardVersion.V3_0) {
227                                throw new CannotParseException(5);
228                        }
229
230                        try {
231                                return newInstance(PartialDate.parse(value));
232                        } catch (IllegalArgumentException e2) {
233                                warnings.add(Messages.INSTANCE.getParseMessage(6));
234                                return newInstance(value);
235                        }
236                }
237        }
238
239        protected abstract T newInstance(String text);
240
241        protected abstract T newInstance(Date date, boolean hasTime);
242
243        protected abstract T newInstance(PartialDate partialDate);
244}
