040/**
041 * Marshals {@link Related} properties.
042 * @author Michael Angstadt
043 */
044public class RelatedScribe extends VCardPropertyScribe<Related> {
045        public RelatedScribe() {
046                super(Related.class, "RELATED");
047        }
048
049        @Override
050        protected VCardDataType _defaultDataType(VCardVersion version) {
051                return VCardDataType.URI;
052        }
053
054        @Override
055        protected VCardDataType _dataType(Related property, VCardVersion version) {
056                if (property.getUri() != null) {
057                        return VCardDataType.URI;
058                }
059                if (property.getText() != null) {
060                        return VCardDataType.TEXT;
061                }
062                return VCardDataType.URI;
063        }
064
065        @Override
066        protected String _writeText(Related property, WriteContext context) {
067                String uri = property.getUri();
068                if (uri != null) {
069                        return uri;
070                }
071
072                String text = property.getText();
073                if (text != null) {
074                        return VObjectPropertyValues.escape(text);
075                }
076
077                return "";
078        }
079
080        @Override
081        protected Related _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
082                value = VObjectPropertyValues.unescape(value);
083
084                Related property = new Related();
085                if (dataType == VCardDataType.TEXT) {
086                        property.setText(value);
087                } else {
088                        property.setUri(value);
089                }
090                return property;
091        }
092
093        @Override
094        protected void _writeXml(Related property, XCardElement parent) {
095                String uri = property.getUri();
096                if (uri != null) {
097                        parent.append(VCardDataType.URI, uri);
098                        return;
099                }
100
101                String text = property.getText();
102                if (text != null) {
103                        parent.append(VCardDataType.TEXT, text);
104                        return;
105                }
106
107                parent.append(VCardDataType.URI, "");
108        }
109
110        @Override
111        protected Related _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
112                String uri = element.first(VCardDataType.URI);
113                if (uri != null) {
114                        Related property = new Related();
115                        property.setUri(uri);
116                        return property;
117                }
118
119                String text = element.first(VCardDataType.TEXT);
120                if (text != null) {
121                        Related property = new Related();
122                        property.setText(text);
123                        return property;
124                }
125
126                throw missingXmlElements(VCardDataType.URI, VCardDataType.TEXT);
127        }
128
129        @Override
130        protected JCardValue _writeJson(Related property) {
131                String uri = property.getUri();
132                if (uri != null) {
133                        return JCardValue.single(uri);
134                }
135
136                String text = property.getText();
137                if (text != null) {
138                        return JCardValue.single(text);
139                }
140
141                return JCardValue.single("");
142        }
143
144        @Override
145        protected Related _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
146                String valueStr = value.asSingle();
147
148                Related property = new Related();
149                if (dataType == VCardDataType.TEXT) {
150                        property.setText(valueStr);
151                } else {
152                        property.setUri(valueStr);
153                }
154                return property;
155        }
156}
