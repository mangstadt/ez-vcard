033/**
034 * Manages the list of pre-defined values for a parameter that consists of a
035 * single String value.
036 * @author Michael Angstadt
037 * @param <T> the parameter class
038 */
039public class VCardParameterCaseClasses<T extends VCardParameter> extends CaseClasses<T, String> {
040        public VCardParameterCaseClasses(Class<T> clazz) {
041                super(clazz);
042        }
043
044        @Override
045        protected T create(String value) {
046                //reflection: return new ClassName(value);
047                try {
048                        //try (String) constructor
049                        Constructor<T> constructor = clazz.getDeclaredConstructor(String.class);
050                        constructor.setAccessible(true);
051                        return constructor.newInstance(value);
052                } catch (Exception e) {
053                        try {
054                                //try (String, VCardVersion...) constructor
055                                Constructor<T> constructor = clazz.getDeclaredConstructor(String.class, VCardVersion[].class);
056                                constructor.setAccessible(true);
057                                return constructor.newInstance(value, new VCardVersion[] {});
058                        } catch (Exception e2) {
059                                throw new RuntimeException(e2);
060                        }
061                }
062        }
063
064        @Override
065        protected boolean matches(T object, String value) {
066                return object.getValue().equalsIgnoreCase(value);
067        }
068}
