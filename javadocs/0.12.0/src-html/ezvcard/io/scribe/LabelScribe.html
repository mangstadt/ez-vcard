035/**
036 * Marshals {@link Label} properties.
037 * @author Michael Angstadt
038 */
039public class LabelScribe extends StringPropertyScribe<Label> {
040        public LabelScribe() {
041                super(Label.class, "LABEL");
042        }
043
044        @Override
045        protected Label _parseValue(String value) {
046                return new Label(value);
047        }
048
049        @Override
050        protected Label _parseHtml(HCardElement element, ParseContext context) {
051                Label property = new Label(element.value());
052
053                List<String> types = element.types();
054                property.getParameters().putAll(VCardParameters.TYPE, types);
055
056                return property;
057        }
058}
