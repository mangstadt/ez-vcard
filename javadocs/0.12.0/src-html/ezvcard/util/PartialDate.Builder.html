043/**
044 * <p>
045 * Represents a date in which some of the components are missing. This is used
046 * to represent reduced accuracy and truncated dates, as defined in ISO8601.
047 * </p>
048 * <p>
049 * A <b>reduced accuracy date</b> is a date where the "lesser" components are
050 * missing. For example, "12:30" is reduced accuracy because the "seconds"
051 * component is missing.
052 * </p>
053 * 
054 * <pre class="brush:java">
055 * PartialDate date = PartialDate.builder().hour(12).minute(30).build();
056 * </pre>
057 * 
058 * <p>
059 * A <b>truncated date</b> is a date where the "greater" components are missing.
060 * For example, "April 20" is truncated because the "year" component is missing.
061 * </p>
062 * 
063 * <pre class="brush:java">
064 * PartialDate date = PartialDate.builder().month(4).date(20).build();
065 * </pre>
066 * @author Michael Angstadt
067 */
068public final class PartialDate {
069        private static final int YEAR = 0;
070        private static final int MONTH = 1;
071        private static final int DATE = 2;
072        private static final int HOUR = 3;
073        private static final int MINUTE = 4;
074        private static final int SECOND = 5;
075        private static final int OFFSET_HOUR = 6;
076        private static final int OFFSET_MINUTE = 7;
077
078        //@formatter:off
079        private static final Format dateFormats[] = new Format[] {
080                new Format("(\\d{4})", YEAR),
081                new Format("(\\d{4})-(\\d{2})", YEAR, MONTH),
082                new Format("(\\d{4})-?(\\d{2})-?(\\d{2})", YEAR, MONTH, DATE),
083                new Format("--(\\d{2})-?(\\d{2})", MONTH, DATE),
084                new Format("--(\\d{2})", MONTH),
085                new Format("---(\\d{2})", DATE)
086        };
087        //@formatter:on
088
089        private static final String offsetRegex = "(([-+]\\d{1,2}):?(\\d{2})?)?";
090
091        //@formatter:off
092        private static final Format timeFormats[] = new Format[] {
093                new Format("(\\d{2})" + offsetRegex, HOUR, null, OFFSET_HOUR, OFFSET_MINUTE),
094                new Format("(\\d{2}):?(\\d{2})" + offsetRegex, HOUR, MINUTE, null, OFFSET_HOUR, OFFSET_MINUTE),
095                new Format("(\\d{2}):?(\\d{2}):?(\\d{2})" + offsetRegex, HOUR, MINUTE, SECOND, null, OFFSET_HOUR, OFFSET_MINUTE),
096                new Format("-(\\d{2}):?(\\d{2})" + offsetRegex, MINUTE, SECOND, null, OFFSET_HOUR, OFFSET_MINUTE),
097                new Format("-(\\d{2})" + offsetRegex, MINUTE, null, OFFSET_HOUR, OFFSET_MINUTE),
098                new Format("--(\\d{2})" + offsetRegex, SECOND, null, OFFSET_HOUR, OFFSET_MINUTE)
099        };
100        //@formatter:on

128        /**
129         * Parses a partial date from a string.
130         * @param string the string (e.g. "--0420T15")
131         * @return the parsed date
132         * @throws IllegalArgumentException if there's a problem parsing the date
133         * string
134         */
135        public static PartialDate parse(String string) {
136                int t = string.indexOf('T');
137                String beforeT, afterT;
138                if (t < 0) {
139                        beforeT = string;
140                        afterT = null;
141                } else {
142                        beforeT = string.substring(0, t);
143                        afterT = (t < string.length() - 1) ? string.substring(t + 1) : null;
144                }
145
146                Builder builder = new Builder();
147                boolean success;
148                if (afterT == null) {
149                        //date or time
150                        success = parseDate(beforeT, builder) || parseTime(beforeT, builder);
151                } else if (beforeT.isEmpty()) {
152                        //time
153                        success = parseTime(afterT, builder);
154                } else {
155                        //date and time
156                        success = parseDate(beforeT, builder) && parseTime(afterT, builder);
157                }
158
159                if (!success) {
160                        throw Messages.INSTANCE.getIllegalArgumentException(36, string);
161                }
162                return builder.build();
163        }
164
165        private static boolean parseDate(String value, Builder builder) {
166                return parseFormats(value, builder, dateFormats);
167        }
168
169        private static boolean parseTime(String value, Builder builder) {
170                return parseFormats(value, builder, timeFormats);
171        }
172
173        private static boolean parseFormats(String value, Builder builder, Format formats[]) {
174                for (Format regex : formats) {
175                        if (regex.parse(builder, value)) {
176                                return true;
177                        }
178                }
179                return false;
180        }

278        /**
279         * Gets the UTC offset.
280         * @return the UTC offset or null if not set
281         */
282        public ZoneOffset getUtcOffset() {
283                Integer hour = components[OFFSET_HOUR];
284                if (hour == null) {
285                        return null;
286                }
287
288                Integer minute = components[OFFSET_MINUTE];
289                if (minute == null) {
290                        return ZoneOffset.ofHours(hour);
291                }
292
293                return ZoneOffset.ofHoursMinutes(hour, minute);
294        }

312        /**
313         * Converts this partial date to its ISO 8601 representation.
314         * @param extended true to use extended format, false to use basic
315         * @return the ISO 8601 representation (e.g. "--0416")
316         * @throws IllegalStateException if an ISO 8601 representation of the date
317         * cannot be created because the date's components are invalid. This will
318         * not happen if the partial date is constructed using the
319         * {@link #builder()} method
320         */
321        public String toISO8601(boolean extended) {
322                /*
323                 * Micro-optimization: The max length a timestamp string can possibly be
324                 * is 25. Example: 2012-07-01T14:21:10-05:00
325                 */
326                final int maxPossibleLength = 25;
327
328                StringBuilder sb = new StringBuilder(maxPossibleLength);
329                NumberFormat nf = new DecimalFormat("00", DecimalFormatSymbols.getInstance(Locale.ROOT));
330
331                String yearStr = hasYear() ? getYear().toString() : null;
332                String monthStr = hasMonth() ? nf.format(getMonth()) : null;
333                String dateStr = hasDate() ? nf.format(getDate()) : null;
334
335                String dash = extended ? "-" : "";
336                if (hasYear() && !hasMonth() && !hasDate()) {
337                        sb.append(yearStr);
338                } else if (!hasYear() && hasMonth() && !hasDate()) {
339                        sb.append("--").append(monthStr);
340                } else if (!hasYear() && !hasMonth() && hasDate()) {
341                        sb.append("---").append(dateStr);
342                } else if (hasYear() && hasMonth() && !hasDate()) {
343                        sb.append(yearStr).append("-").append(monthStr);
344                } else if (!hasYear() && hasMonth() && hasDate()) {
345                        sb.append("--").append(monthStr).append(dash).append(dateStr);
346                } else if (hasYear() && !hasMonth() && hasDate()) {
347                        //this is checked for in the builder and should never be thrown here
348                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(38));
349                } else if (hasYear() && hasMonth() && hasDate()) {
350                        sb.append(yearStr).append(dash).append(monthStr).append(dash).append(dateStr);
351                }
352
353                if (hasTimeComponent()) {
354                        sb.append('T');
355
356                        String hourStr = hasHour() ? nf.format(getHour()) : null;
357                        String minuteStr = hasMinute() ? nf.format(getMinute()) : null;
358                        String secondStr = hasSecond() ? nf.format(getSecond()) : null;
359
360                        dash = extended ? ":" : "";
361                        if (hasHour() && !hasMinute() && !hasSecond()) {
362                                sb.append(hourStr);
363                        } else if (!hasHour() && hasMinute() && !hasSecond()) {
364                                sb.append("-").append(minuteStr);
365                        } else if (!hasHour() && !hasMinute() && hasSecond()) {
366                                sb.append("--").append(secondStr);
367                        } else if (hasHour() && hasMinute() && !hasSecond()) {
368                                sb.append(hourStr).append(dash).append(minuteStr);
369                        } else if (!hasHour() && hasMinute() && hasSecond()) {
370                                sb.append("-").append(minuteStr).append(dash).append(secondStr);
371                        } else if (hasHour() && !hasMinute() && hasSecond()) {
372                                //this is checked for in the builder and should never be thrown here
373                                throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(39));
374                        } else if (hasHour() && hasMinute() && hasSecond()) {
375                                sb.append(hourStr).append(dash).append(minuteStr).append(dash).append(secondStr);
376                        }
377
378                        Integer offsetHour = components[OFFSET_HOUR];
379                        if (offsetHour != null) {
380                                Integer offsetMinute = components[OFFSET_MINUTE];
381                                String offsetHourStr = nf.format(Math.abs(offsetHour));
382                                String offsetMinuteStr = nf.format(Math.abs(offsetMinute));
383
384                                sb.append((offsetHour < 0 || offsetMinute < 0) ? '-' : '+');
385                                sb.append(offsetHourStr).append(dash).append(offsetMinuteStr);
386                        }
387                }
388
389                return sb.toString();
390        }

392        @Override
393        public int hashCode() {
394                final int prime = 31;
395                int result = 1;
396                result = prime * result + Arrays.hashCode(components);
397                return result;
398        }

400        @Override
401        public boolean equals(Object obj) {
402                if (this == obj) return true;
403                if (obj == null) return false;
404                if (getClass() != obj.getClass()) return false;
405                PartialDate other = (PartialDate) obj;
406                if (!Arrays.equals(components, other.components)) return false;
407                return true;
408        }

415        /**
416         * Represents a string format that a partial date can be in.
417         */
418        private static class Format {
419                private Pattern regex;
420                private Integer[] componentIndexes;
421
422                /**
423                 * @param regex the regular expression that describes the format
424                 * @param componentIndexes the indexes of the
425                 * {@link PartialDate#components} array to assign the value of each
426                 * regex group to, or null to ignore the group
427                 */
428                public Format(String regex, Integer... componentIndexes) {
429                        this.regex = Pattern.compile('^' + regex + '$');
430                        this.componentIndexes = componentIndexes;
431                }
432
433                /**
434                 * Tries to parse a given string.
435                 * @param builder the object to assign the parsed data to
436                 * @param value the string to parse
437                 * @return true if the string was successfully parsed, false if not
438                 */
439                public boolean parse(Builder builder, String value) {
440                        Matcher m = regex.matcher(value);
441                        if (!m.find()) {
442                                return false;
443                        }
444
445                        /*
446                         * All this complicated handling of the offset is due to:
447                         * 
448                         * The hour portion can be zero, and zero can't be negative (e.g.
449                         * "-00:30").
450                         * 
451                         * There is no positive or negative sign next to the minute portion
452                         * in the string.
453                         */
454                        boolean offsetPositive = false;
455                        Integer offsetHour = null, offsetMinute = null;
456
457                        for (int i = 0; i < componentIndexes.length; i++) {
458                                Integer index = componentIndexes[i];
459                                if (index == null) {
460                                        continue;
461                                }
462
463                                int group = i + 1;
464                                String groupStr = m.group(group);
465                                if (groupStr != null) {
466                                        boolean startsWithPlus = groupStr.startsWith("+");
467                                        if (startsWithPlus) {
468                                                groupStr = groupStr.substring(1);
469                                        }
470
471                                        int component = Integer.parseInt(groupStr);
472                                        if (index == OFFSET_HOUR) {
473                                                offsetHour = component;
474                                                offsetPositive = startsWithPlus;
475                                                continue;
476                                        }
477                                        if (index == OFFSET_MINUTE) {
478                                                offsetMinute = component;
479                                                continue;
480                                        }
481                                        builder.components[index] = component;
482                                }
483                        }
484
485                        if (offsetHour != null) {
486                                if (offsetMinute == null) {
487                                        offsetMinute = 0;
488                                }
489                                if (!offsetPositive) {
490                                        offsetMinute *= -1;
491                                }
492                                builder.components[OFFSET_HOUR] = offsetHour;
493                                builder.components[OFFSET_MINUTE] = offsetMinute;
494                        }
495
496                        return true;
497                }
498        }

500        /**
501         * Constructs instances of the {@link PartialDate} class.
502         * @author Michael Angstadt
503         */
504        public static class Builder {
505                private final Integer[] components;
506
507                public Builder() {
508                        components = new Integer[8];
509                }
510
511                /**
512                 * @param original the partial date to copy
513                 */
514                public Builder(PartialDate original) {
515                        components = original.components.clone();
516                }
517
518                /**
519                 * Sets the year component.
520                 * @param year the year
521                 * @return this
522                 */
523                public Builder year(Integer year) {
524                        components[YEAR] = year;
525                        return this;
526                }
527
528                /**
529                 * Sets the month component.
530                 * @param month the month (1-12)
531                 * @return this
532                 * @throws IllegalArgumentException if the month is not between 1 and 12
533                 * inclusive
534                 */
535                public Builder month(Integer month) {
536                        if (month != null && (month < 1 || month > 12)) {
537                                throw Messages.INSTANCE.getIllegalArgumentException(37, "Month", 1, 12);
538                        }
539
540                        components[MONTH] = month;
541                        return this;
542                }
543
544                /**
545                 * Sets the date component.
546                 * @param date the date
547                 * @return this
548                 * @throws IllegalArgumentException if the date is not between 1 and 31
549                 * inclusive
550                 */
551                public Builder date(Integer date) {
552                        if (date != null && (date < 1 || date > 31)) {
553                                throw Messages.INSTANCE.getIllegalArgumentException(37, "Date", 1, 31);
554                        }
555
556                        components[DATE] = date;
557                        return this;
558                }
559
560                /**
561                 * Sets the hour component.
562                 * @param hour the hour
563                 * @return this
564                 * @throws IllegalArgumentException if the hour is not between 0 and 23
565                 * inclusive
566                 */
567                public Builder hour(Integer hour) {
568                        if (hour != null && (hour < 0 || hour > 23)) {
569                                throw Messages.INSTANCE.getIllegalArgumentException(37, "Hour", 0, 23);
570                        }
571
572                        components[HOUR] = hour;
573                        return this;
574                }
575
576                /**
577                 * Sets the minute component.
578                 * @param minute the minute
579                 * @return this
580                 * @throws IllegalArgumentException if the minute is not between 0 and
581                 * 59 inclusive
582                 */
583                public Builder minute(Integer minute) {
584                        if (minute != null && (minute < 0 || minute > 59)) {
585                                throw Messages.INSTANCE.getIllegalArgumentException(37, "Minute", 0, 59);
586                        }
587
588                        components[MINUTE] = minute;
589                        return this;
590                }
591
592                /**
593                 * Sets the second component.
594                 * @param second the second
595                 * @return this
596                 * @throws IllegalArgumentException if the second is not between 0 and
597                 * 59 inclusive
598                 */
599                public Builder second(Integer second) {
600                        if (second != null && (second < 0 || second > 59)) {
601                                throw Messages.INSTANCE.getIllegalArgumentException(37, "Second", 0, 59);
602                        }
603
604                        components[SECOND] = second;
605                        return this;
606                }
607
608                /**
609                 * Sets the timezone offset.
610                 * @param offset the offset
611                 * @return this
612                 */
613                public Builder offset(ZoneOffset offset) {
614                        if (offset == null) {
615                                components[OFFSET_HOUR] = null;
616                                components[OFFSET_MINUTE] = null;
617                        } else {
618                                components[OFFSET_HOUR] = offset.getTotalSeconds() / 3600;
619                                components[OFFSET_MINUTE] = (offset.getTotalSeconds() % 3600) / 60;
620                        }
621                        return this;
622                }
623
624                /**
625                 * Builds the {@link PartialDate} object.
626                 * @return the {@link PartialDate} object
627                 * @throws IllegalArgumentException if the year and date are defined,
628                 * but the month is not, or if the hour and second are defined, but the
629                 * minute is not
630                 */
631                public PartialDate build() {
632                        if (components[YEAR] != null && components[MONTH] == null && components[DATE] != null) {
633                                throw Messages.INSTANCE.getIllegalArgumentException(38);
634                        }
635                        if (components[HOUR] != null && components[MINUTE] == null && components[SECOND] != null) {
636                                throw Messages.INSTANCE.getIllegalArgumentException(39);
637                        }
638
639                        return new PartialDate(components);
640                }
641        }
