040/**
041 * <p>
042 * Simply identifies the vCard as a "vCard".
043 * </p>
044 * 
045 * <p>
046 * <b>Code sample</b>
047 * </p>
048 * 
049 * <pre class="brush:java">
050 * VCard vcard = new VCard();
051 * 
052 * Profile profile = new Profile();
053 * vcard.setProfile(profile);
054 * </pre>
055 * 
056 * <p>
057 * <b>Property name:</b> {@code PROFILE}
058 * </p>
059 * <p>
060 * <b>Supported versions:</b> {@code 3.0}
061 * </p>
062 * @author Michael Angstadt
063 */
064public class Profile extends TextProperty {
065        public Profile() {
066                super("VCARD");
067        }
068
069        @Override
070        public Set<VCardVersion> _supportedVersions() {
071                return EnumSet.of(VCardVersion.V3_0);
072        }
073
074        @Override
075        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
076                if (!"VCARD".equalsIgnoreCase(value)) {
077                        //see RFC 2426 p.5
078                        warnings.add(new Warning(18, value));
079                }
080        }
081}
