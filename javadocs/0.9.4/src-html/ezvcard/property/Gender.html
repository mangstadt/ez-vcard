040/**
041 * <p>
042 * Defines the person's sex.
043 * </p>
044 * 
045 * <p>
046 * <b>Code sample (creating)</b>
047 * </p>
048 * 
049 * <pre class="brush:java">
050 * VCard vcard = new VCard();
051 * 
052 * Gender gender = Gender.male();
053 * vcard.setGender(gender);
054 * </pre>
055 * 
056 * <p>
057 * <b>Code sample (retrieving)</b>
058 * </p>
059 * 
060 * <pre class="brush:java">
061 * VCard vcard = ...
062 * Gender gender = vcard.getGender();
063 * if (gender.isMale()){
064 *   //gender is male
065 * } else if (gender.isFemale()){
066 *   //gender is female
067 * }
068 * //etc
069 * </pre>
070 * 
071 * <p>
072 * <b>Property name:</b> {@code GENDER}
073 * </p>
074 * <p>
075 * <b>Supported versions:</b> {@code 4.0}
076 * </p>
077 * @author Michael Angstadt
078 */
079public class Gender extends VCardProperty {
080        public static final String MALE = "M";
081        public static final String FEMALE = "F";
082        public static final String OTHER = "O";
083        public static final String NONE = "N";
084        public static final String UNKNOWN = "U";
085
086        private String gender;
087        private String text;
088
089        /**
090         * Creates a gender property. Use of this constructor is discouraged. Please
091         * use one of the static factory methods to create a new GENDER property.
092         * @param gender the gender value (e.g. "F")
093         */
094        public Gender(String gender) {
095                this.gender = gender;
096        }
097
098        @Override
099        public Set<VCardVersion> _supportedVersions() {
100                return EnumSet.of(VCardVersion.V4_0);
101        }
102
103        /**
104         * Gets the additional text associated with this property.
105         * @return the additional text or null if there is no text
106         */
107        public String getText() {
108                return text;
109        }
110
111        /**
112         * Sets the additional text associated with this property.
113         * @param text additional text or null to remove
114         */
115        public void setText(String text) {
116                this.text = text;
117        }
118
119        /**
120         * Gets the gender value.
121         * @return the gender value (see static strings for the possible values)
122         */
123        public String getGender() {
124                return gender;
125        }
126
127        /**
128         * Sets the gender value.
129         * @param gender the gender value (see static strings for the possible
130         * values)
131         */
132        public void setGender(String gender) {
133                this.gender = gender;
134        }
135
136        /**
137         * Determines if the gender is "male" or not.
138         * @return true if the gender is "male", false if not
139         */
140        public boolean isMale() {
141                return MALE.equals(gender);
142        }
143
144        /**
145         * Determines if the gender is "female" or not.
146         * @return true if the gender is "female", false if not
147         */
148        public boolean isFemale() {
149                return FEMALE.equals(gender);
150        }
151
152        /**
153         * Determines if the gender is "other" or not.
154         * @return true if the gender is "other", false if not
155         */
156        public boolean isOther() {
157                return OTHER.equals(gender);
158        }
159
160        /**
161         * Determines if the gender is "none" or not. A group, organization, or
162         * location may have this gender property.
163         * @return true if the gender is "none", false if not
164         */
165        public boolean isNone() {
166                return NONE.equals(gender);
167        }
168
169        /**
170         * Determines if the gender is "unknown" or not.
171         * @return true if the gender is "unknown", false if not
172         */
173        public boolean isUnknown() {
174                return UNKNOWN.equals(gender);
175        }
176
177        /**
178         * Creates a gender property whose value is set to "male".
179         * @return a "male" gender property
180         */
181        public static Gender male() {
182                return new Gender(MALE);
183        }
184
185        /**
186         * Creates a gender property whose value is set to "female".
187         * @return a "female" gender property
188         */
189        public static Gender female() {
190                return new Gender(FEMALE);
191        }
192
193        /**
194         * Creates a gender property whose value is set to "other".
195         * @return an "other" gender property
196         */
197        public static Gender other() {
198                return new Gender(OTHER);
199        }
200
201        /**
202         * Creates a gender property whose value is set to "none". Groups,
203         * organizations, and locations should be given this gender property.
204         * @return a "none" gender property
205         */
206        public static Gender none() {
207                return new Gender(NONE);
208        }
209
210        /**
211         * Creates a gender property whose value is set to "unknown".
212         * @return a "unknown" gender property
213         */
214        public static Gender unknown() {
215                return new Gender(UNKNOWN);
216        }
217
218        @Override
219        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
220                if (gender == null) {
221                        warnings.add(new Warning(8));
222                }
223        }
224}
