035/**
036 * Thrown during the marshalling/unmarshalling of a property to signal to the
037 * marshaller that the property's value is a nested (2.1 style) or embedded (3.0
038 * style) vCard.
039 * @author Michael Angstadt
040 */
041@SuppressWarnings("serial")
042public class EmbeddedVCardException extends RuntimeException {
043        /*
044         * Serialization note: Not going to bother adding a serialVersionUID, since
045         * VCard is not serializable.
046         */
047        private final VCard vcard;
048        private final InjectionCallback callback;
049
050        /**
051         * Thrown to unmarshal a nested or embedded vCard.
052         * @param callback injects the unmarshalled vCard into the property object
053         */
054        public EmbeddedVCardException(InjectionCallback callback) {
055                this.callback = callback;
056                this.vcard = null;
057        }
058
059        /**
060         * Thrown to marshal a nested or embedded vCard.
061         * @param vcard the vCard to marshal
062         */
063        public EmbeddedVCardException(VCard vcard) {
064                this.callback = null;
065                this.vcard = vcard;
066        }
067
068        /**
069         * Gets the vCard to marshal.
070         * @return the vCard to marshal
071         */
072        public VCard getVCard() {
073                return vcard;
074        }
075
076        /**
077         * Injects the unmarshalled vCard into the property object that threw this
078         * exception.
079         * @param vcard the vCard to inject
080         */
081        public void injectVCard(VCard vcard) {
082                if (callback == null) {
083                        return;
084                }
085
086                callback.injectVCard(vcard);
087        }
088
089        /**
090         * Gets the property object that threw the exception.
091         * @return the property object
092         */
093        public VCardProperty getProperty() {
094                if (callback == null) {
095                        return null;
096                }
097
098                return callback.getProperty();
099        }
100
101        /**
102         * Injects an unmarshalled vCard into the property object.
103         */
104        public interface InjectionCallback {
105                /**
106                 * Injects an unmarshalled vCard into the property object.
107                 * @param vcard the vCard to inject
108                 */
109                void injectVCard(VCard vcard);
110
111                /**
112                 * Gets the property object that threw the
113                 * {@link EmbeddedVCardException}.
114                 * @return the property object
115                 */
116                VCardProperty getProperty();
117        }
118}
