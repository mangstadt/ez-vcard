031/**
032 * Marshals {@link Language} properties.
033 * @author Michael Angstadt
034 */
035public class LanguageScribe extends StringPropertyScribe<Language> {
036        public LanguageScribe() {
037                super(Language.class, "LANG", VCardDataType.LANGUAGE_TAG);
038        }
039
040        @Override
041        protected Language _parseValue(String value) {
042                return new Language(value);
043        }
044}
