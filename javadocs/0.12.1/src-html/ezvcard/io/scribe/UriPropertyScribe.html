033/**
034 * Marshals properties that have URI values.
035 * @param <T> the property class
036 * @author Michael Angstadt
037 */
038public abstract class UriPropertyScribe<T extends TextProperty> extends StringPropertyScribe<T> {
039        public UriPropertyScribe(Class<T> clazz, String propertyName) {
040                super(clazz, propertyName, VCardDataType.URI);
041        }
042
043        @Override
044        protected T _parseHtml(HCardElement element, ParseContext context) {
045                String href = element.absUrl("href");
046                if (href.isEmpty()) {
047                        return super._parseHtml(element, context);
048                }
049                return _parseValue(href);
050        }
051}
