030/**
031 * Marshals {@link Kind} properties.
032 * @author Michael Angstadt
033 */
034public class KindScribe extends StringPropertyScribe<Kind> {
035        public KindScribe() {
036                super(Kind.class, "KIND");
037        }
038
039        @Override
040        protected Kind _parseValue(String value) {
041                return new Kind(value);
042        }
043}
