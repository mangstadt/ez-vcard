039/**
040 * <p>
041 * Simply identifies the vCard as a "vCard".
042 * </p>
043 * 
044 * <p>
045 * <b>Code sample</b>
046 * </p>
047 * 
048 * <pre class="brush:java">
049 * VCard vcard = new VCard();
050 * 
051 * Profile profile = new Profile();
052 * vcard.setProfile(profile);
053 * </pre>
054 * 
055 * <p>
056 * <b>Property name:</b> {@code PROFILE}
057 * </p>
058 * <p>
059 * <b>Supported versions:</b> {@code 3.0}
060 * </p>
061 * @author Michael Angstadt
062 * @see <a href="http://tools.ietf.org/html/rfc2426#page-5">RFC 2426 p.5</a>
063 */
064@SupportedVersions(VCardVersion.V3_0)
065public class Profile extends TextProperty {
066        public Profile() {
067                super("VCARD");
068        }
069
070        /**
071         * Copy constructor.
072         * @param original the property to make a copy of
073         */
074        public Profile(Profile original) {
075                super(original);
076        }
077
078        @Override
079        protected void _validate(List<ValidationWarning> warnings, VCardVersion version, VCard vcard) {
080                if (!"VCARD".equalsIgnoreCase(value)) {
081                        //see RFC 2426 p.5
082                        warnings.add(new ValidationWarning(18, value));
083                }
084        }
085
086        @Override
087        public Profile copy() {
088                return new Profile(this);
089        }
090}
