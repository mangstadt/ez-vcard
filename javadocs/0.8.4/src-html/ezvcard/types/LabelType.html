041    /**
042     * <p>
043     * Defines the exact text to put on the mailing label when mailing a package or
044     * letter to the person.
045     * </p>
046     * 
047     * <p>
048     * The LABEL type is not supported in 4.0. Instead, labels are included as a
049     * parameter to their corresponding {@link AddressType}. When marshalling a
050     * vCard, ez-vcard will use either the LABEL type or the LABEL parameter,
051     * depending on the requested vCard version.
052     * </p>
053     * 
054     * <p>
055     * To add a label to a vCard, the {@link AddressType#setLabel} method should be
056     * used.
057     * </p>
058     * 
059     * <pre class="brush:java">
060     * VCard vcard = new VCard();
061     * AddressType adr = new AddressType();
062     * adr.setStreetAddress(&quot;123 Main St.&quot;);
063     * adr.setLocality(&quot;Austin&quot;);
064     * adr.setRegion(&quot;TX&quot;);
065     * adr.setPostalCode(&quot;12345&quot;);
066     * adr.setLabel(&quot;123 Main St.\nAustin, TX 12345&quot;); //newlines are allowed
067     * vcard.addAddress(adr);
068     * </pre>
069     * 
070     * <p>
071     * The {@link VCard#addOrphanedLabel} method adds a LABEL type to the vCard.
072     * However, use of this method is discouraged because it creates a LABEL type
073     * that's not associated with an address. Also, orphaned LABEL types are ignored
074     * when creating version 4.0 vCards because the LABEL type is not supported by
075     * vCard 4.0.
076     * </p>
077     * 
078     * <p>
079     * The {@link VCard#getOrphanedLabels} method can be used after parsing a
080     * version 2.1 or 3.0 vCard to retrieve any LABEL types which the parser could
081     * not assign to an address. A LABEL is assigned to an address if the LABEL's
082     * list of TYPE parameters is identical to the address's list of TYPE
083     * parameters.
084     * </p>
085     * 
086     * <pre class="brush:java">
087     * VCard vcard = ...
088     * for (LabelType label : vcard.getOrphanedLabels()) {
089     *      System.out.println(label.getValue());
090     * }
091     * </pre>
092     * 
093     * <p>
094     * <b>Property name:</b> {@code LABEL}
095     * </p>
096     * <p>
097     * <b>Supported versions:</b> {@code 2.1, 3.0}
098     * </p>
099     * @author Michael Angstadt
100     */
101    public class LabelType extends MultiValuedTypeParameterType<AddressTypeParameter> {
102            public static final String NAME = "LABEL";
103    
104            private String value;
105    
106            /**
107             * Creates an empty label property.
108             */
109            public LabelType() {
110                    this(null);
111            }
112    
113            /**
114             * Creates a label property.
115             * @param label the label value
116             */
117            public LabelType(String label) {
118                    super(NAME);
119                    setValue(label);
120            }
121    
122            @Override
123            protected AddressTypeParameter buildTypeObj(String type) {
124                    return AddressTypeParameter.get(type);
125            }
126    
127            /**
128             * Gets the label value.
129             * @return the label value
130             */
131            public String getValue() {
132                    return value;
133            }
134    
135            /**
136             * Sets the label value.
137             * @param value the label value
138             */
139            public void setValue(String value) {
140                    this.value = value;
141            }
142    
143            @Override
144            public String getLanguage() {
145                    return super.getLanguage();
146            }
147    
148            @Override
149            public void setLanguage(String language) {
150                    super.setLanguage(language);
151            }
152    
153            @Override
154            public VCardVersion[] getSupportedVersions() {
155                    return new VCardVersion[] { VCardVersion.V2_1, VCardVersion.V3_0 };
156            }
157    
158            @Override
159            protected void doMarshalText(StringBuilder sb, VCardVersion version, CompatibilityMode compatibilityMode) {
160                    if (value != null) {
161                            sb.append(VCardStringUtils.escape(value));
162                    }
163            }
164    
165            @Override
166            protected void doUnmarshalText(String value, VCardVersion version, List<String> warnings, CompatibilityMode compatibilityMode) {
167                    setValue(VCardStringUtils.unescape(value));
168            }
169    
170            @Override
171            protected void doUnmarshalHtml(HCardElement element, List<String> warnings) {
172                    List<String> types = element.types();
173                    for (String type : types) {
174                            subTypes.addType(type);
175                    }
176    
177                    setValue(element.value());
178            }
179    }
