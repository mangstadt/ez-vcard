055/**
056 * Base class for vCard property scribes (aka "marshallers" or "serializers").
057 * @param <T> the property class
058 * @author Michael Angstadt
059 */
060public abstract class VCardPropertyScribe<T extends VCardProperty> {
061        protected final Class<T> clazz;
062        protected final String propertyName;
063        protected final QName qname;
064
065        /**
066         * Creates a new scribe.
067         * @param clazz the property class
068         * @param propertyName the property name (e.g. "FN")
069         */
070        public VCardPropertyScribe(Class<T> clazz, String propertyName) {
071                this(clazz, propertyName, new QName(VCardVersion.V4_0.getXmlNamespace(), propertyName.toLowerCase()));
072        }

075         * Creates a new scribe.
076         * @param clazz the property class
077         * @param propertyName the property name (e.g. "FN")

079         * documents (by default, the XML element name is set to the lower-cased
080         * property name, and the element namespace is set to the xCard namespace)
081         */

112        /**
113         * Sanitizes a property's parameters (called before the property is
114         * written). Note that a copy of the parameters is returned so that the
115         * property object does not get modified.
116         * @param property the property
117         * @param version the version of the vCard that is being generated
118         * @param vcard the vCard that the property belongs to

121        public final VCardParameters prepareParameters(T property, VCardVersion version, VCard vcard) {

392         * @param parameters the parsed parameters. These parameters will be
393         * assigned to the property object once this method returns. Therefore, do

395         * this method, or else they will be overwritten.
396         * @param warnings allows the programmer to alert the user to any
397         * note-worthy (but non-critical) issues that occurred during the
398         * unmarshalling process
399         * @return the unmarshalled property object

407        /**
408         * <p>
409         * Unmarshals a property from an XML document (xCard).
410         * </p>
411         * <p>
412         * This method should be overridden by child classes that wish to support
413         * xCard. The default implementation of this method will find the first
414         * child element with the xCard namespace. The element's name will be used
415         * as the property's data type and its text content (escaped for inclusion
416         * in a text-based vCard, e.g. escaping comma characters) will be passed
417         * into the {@link #_parseText} method. If no such child element is found,
418         * then the parent element's text content will be passed into
419         * {@link #_parseText} and the data type will be {@code null}.
420         * </p>
421         * @param element the property's XML element
422         * @param parameters the parsed parameters. These parameters will be
423         * assigned to the property object once this method returns. Therefore, do
424         * not assign any parameters to the property object itself whilst inside of
425         * this method, or else they will be overwritten.
426         * @param warnings allows the programmer to alert the user to any
427         * note-worthy (but non-critical) issues that occurred during the
428         * unmarshalling process
429         * @return the unmarshalled property object
430         * @throws CannotParseException if the marshaller could not parse the
431         * property's value
432         * @throws SkipMeException if the property should not be added to the final
433         * {@link VCard} object
434         */
435        protected T _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
436                XCardValue firstValue = element.firstValue();
437                VCardDataType dataType = firstValue.getDataType();
438                String value = VObjectPropertyValues.escape(firstValue.getValue());
439                return _parseText(value, dataType, element.version(), parameters, warnings);
440        }

471        /**
472         * <p>
473         * Unmarshals a property from a JSON data stream (jCard).
474         * </p>
475         * <p>
476         * This method should be overridden by child classes that wish to support
477         * jCard. The default implementation of this method will convert the jCard
478         * property value to a string and pass it into the {@link #_parseText}
479         * method.
480         * </p>
481         * 
482         * <hr>
483         * 
484         * <p>
485         * The following paragraphs describe the way in which this method's default
486         * implementation converts a jCard value to a string:
487         * </p>
488         * <p>
489         * If the jCard value consists of a single, non-array, non-object value,
490         * then the value is converted to a string. Special characters (backslashes,
491         * commas, and semicolons) are escaped in order to simulate what the value
492         * might look like in a plain-text vCard.<br>
493         * <code>["x-foo", {}, "text", "the;value"] --&gt; "the\;value"</code><br>
494         * <code>["x-foo", {}, "text", 2] --&gt; "2"</code>
495         * </p>
496         * <p>
497         * If the jCard value consists of multiple, non-array, non-object values,
498         * then all the values are appended together in a single string, separated
499         * by commas. Special characters (backslashes, commas, and semicolons) are
500         * escaped for each value in order to prevent commas from being treated as
501         * delimiters, and to simulate what the value might look like in a
502         * plain-text vCard.<br>
503         * <code>["x-foo", {}, "text", "one", "two,three"] --&gt;
504         * "one,two\,three"</code>
505         * </p>
506         * <p>
507         * If the jCard value is a single array, then this array is treated as a
508         * "structured value", and converted its plain-text representation. Special
509         * characters (backslashes, commas, and semicolons) are escaped for each
510         * value in order to prevent commas and semicolons from being treated as
511         * delimiters.<br>
512         * <code>["x-foo", {}, "text", ["one", ["two", "three"], "four;five"]]
513         * --&gt; "one;two,three;four\;five"</code>
514         * </p>
515         * <p>
516         * If the jCard value starts with a JSON object, then it is converted to an
517         * empty string (JSON objects are not supported by this method).<br>
518         * <code>["x-foo", , "text", {"one": 1}] --&gt; ""}</code>
519         * </p>
520         * <p>
521         * For all other cases, behavior is undefined.
522         * </p>
523         * @param value the property's JSON value
524         * @param dataType the data type
525         * @param parameters the parsed parameters. These parameters will be
526         * assigned to the property object once this method returns. Therefore, do
527         * not assign any parameters to the property object itself whilst inside of
528         * this method, or else they will be overwritten.
529         * @param warnings allows the programmer to alert the user to any
530         * note-worthy (but non-critical) issues that occurred during the
531         * unmarshalling process
532         * @return the unmarshalled property object
533         * @throws CannotParseException if the marshaller could not parse the
534         * property's value
535         * @throws SkipMeException if the property should not be added to the final
536         * {@link VCard} object
537         */
538        protected T _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
539                String valueStr = jcardValueToString(value);
540                return _parseText(valueStr, dataType, VCardVersion.V4_0, parameters, warnings);
541        }

543        /**
544         * Converts a jCard value to its plain-text format representation.
545         * @param value the jCard value
546         * @return the plain-text format representation (for example, "1,2,3" for a
547         * list of values)
548         */
549        private static String jcardValueToString(JCardValue value) {
550                List<JsonValue> values = value.getValues();
551                if (values.size() > 1) {
552                        List<String> multi = value.asMulti();
553                        if (!multi.isEmpty()) {
554                                return VObjectPropertyValues.writeList(multi);
555                        }
556                }
557
558                if (!values.isEmpty() && values.get(0).getArray() != null) {
559                        List<List<String>> structured = value.asStructured();
560                        if (!structured.isEmpty()) {
561                                return VObjectPropertyValues.writeStructured(structured, true);
562                        }
563                }
564
565                return VObjectPropertyValues.escape(value.asSingle());
566        }

587        /**
588         * A helper class for writing date values.
589         */
590        protected static class DateWriter {
591                private Date date;
592                private boolean hasTime = true;
593                private boolean extended = false;
594                private boolean utc = true;
595
596                /**
597                 * Creates a new date writer object.
598                 * @param date the date to format
599                 */
600                public DateWriter(Date date) {
601                        this.date = date;
602                }
603
604                /**
605                 * Sets whether to output the date's time component.
606                 * @param hasTime true include the time, false if it's strictly a date
607                 * (defaults to "true")
608                 * @return this
609                 */
610                public DateWriter time(boolean hasTime) {
611                        this.hasTime = hasTime;
612                        return this;
613                }
614
615                /**
616                 * Sets whether to use extended format or basic.
617                 * @param extended true to use extended format, false to use basic
618                 * (defaults to "false")
619                 * @return this
620                 */
621                public DateWriter extended(boolean extended) {
622                        this.extended = extended;
623                        return this;
624                }
625
626                /**
627                 * Sets whether to format the date in UTC time, or to include a UTC
628                 * offset.
629                 * @param utc true to format in UTC time, false to include the local
630                 * timezone's UTC offset (defaults to "true")
631                 * @return this
632                 */
633                public DateWriter utc(boolean utc) {
634                        this.utc = utc;
635                        return this;
636                }
637
638                /**
639                 * Creates the date string.
640                 * @return the date string
641                 */
642                public String write() {
643                        VCardDateFormat format;
644                        if (hasTime) {
645                                if (utc) {
646                                        format = extended ? VCardDateFormat.UTC_DATE_TIME_EXTENDED : VCardDateFormat.UTC_DATE_TIME_BASIC;
647                                } else {
648                                        format = extended ? VCardDateFormat.DATE_TIME_EXTENDED : VCardDateFormat.DATE_TIME_BASIC;
649                                }
650                        } else {
651                                format = extended ? VCardDateFormat.DATE_EXTENDED : VCardDateFormat.DATE_BASIC;
652                        }
653
654                        return format.format(date);
655                }
656        }

658        /**
659         * Creates a {@link CannotParseException} to indicate that a scribe could
660         * not find the necessary XML elements required in order to successfully
661         * parse a property (xCards only).
662         * @param dataTypes the expected data types (null for "unknown")
663         * @return the exception object (note that the exception is NOT thrown!)
664         */
665        protected static CannotParseException missingXmlElements(VCardDataType... dataTypes) {
666                String[] elements = new String[dataTypes.length];
667                for (int i = 0; i < dataTypes.length; i++) {
668                        VCardDataType dataType = dataTypes[i];
669                        elements[i] = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
670                }
671                return missingXmlElements(elements);
672        }

674        /**
675         * Creates a {@link CannotParseException} to indicate that a scribe could
676         * not find the necessary XML elements required in order to successfully
677         * parse a property (xCards only).
678         * @param elements the names of the expected XML elements.
679         * @return the exception object (note that the exception is NOT thrown!)
680         */
681        protected static CannotParseException missingXmlElements(String... elements) {
682                return new CannotParseException(0, Arrays.toString(elements));
683        }

685        /**
686         * A utility method for switching between the "PREF" and "TYPE=PREF"
687         * parameters when marshalling a property (version 4.0 vCards use "PREF=1",
688         * while version 3.0 vCards use "TYPE=PREF"). This method is meant to be
689         * called from a scribe's {@link #_prepareParameters} method.
690         * @param property the property that is being marshalled
691         * @param copy the parameters that are being marshalled
692         * @param version the vCard version
693         * @param vcard the vCard that's being marshalled
694         */
695        protected static void handlePrefParam(VCardProperty property, VCardParameters copy, VCardVersion version, VCard vcard) {
696                switch (version) {
697                case V2_1:
698                case V3_0:
699                        copy.setPref(null);
700
701                        //find the property with the lowest PREF value in the vCard
702                        VCardProperty mostPreferred = null;
703                        for (VCardProperty p : vcard.getProperties(property.getClass())) {
704                                Integer pref = p.getParameters().getPref();
705                                if (pref == null) {
706                                        continue;
707                                }
708
709                                if (mostPreferred == null || pref < mostPreferred.getParameters().getPref()) {
710                                        mostPreferred = p;
711                                }
712                        }
713
714                        if (property == mostPreferred) {
715                                copy.put(VCardParameters.TYPE, "pref");
716                        }
717
718                        break;
719                case V4_0:
720                        for (String type : property.getParameters().get(VCardParameters.TYPE)) {
721                                if ("pref".equalsIgnoreCase(type)) {
722                                        copy.remove(VCardParameters.TYPE, type);
723                                        copy.setPref(1);
724                                        break;
725                                }
726                        }
727                        break;
728                }
729        }
