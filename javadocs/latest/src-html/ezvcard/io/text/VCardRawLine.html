037/**
038 * Represents a parsed line from a vCard file.
039 * @author Michael Angstadt
040 */
041public class VCardRawLine {
042        private final String group, name, value;
043        private final VCardParameters parameters;
044
045        /**
046         * @param group the group
047         * @param name the property name
048         * @param parameters the parameters
049         * @param value the property value
050         */
051        public VCardRawLine(String group, String name, VCardParameters parameters, String value) {
052                this.group = group;
053                this.name = name;
054                this.value = value;
055                this.parameters = parameters;
056        }
057
058        /**
059         * Gets the property's group.
060         * @return the group or null if there is no group
061         */
062        public String getGroup() {
063                return group;
064        }
065
066        /**
067         * Gets the property name.
068         * @return the property name
069         */
070        public String getName() {
071                return name;
072        }
073
074        /**
075         * Gets the property value.
076         * @return the property value
077         */
078        public String getValue() {
079                return value;
080        }
081
082        /**
083         * Gets the property's parameters.
084         * @return the parameters
085         */
086        public VCardParameters getParameters() {
087                return parameters;
088        }
089
090        @Override
091        public int hashCode() {
092                final int prime = 31;
093                int result = 1;
094                result = prime * result + ((group == null) ? 0 : group.hashCode());
095                result = prime * result + ((name == null) ? 0 : name.hashCode());
096                result = prime * result + ((parameters == null) ? 0 : parameters.hashCode());
097                result = prime * result + ((value == null) ? 0 : value.hashCode());
098                return result;
099        }
100
101        @Override
102        public boolean equals(Object obj) {
103                if (this == obj)
104                        return true;
105                if (obj == null)
106                        return false;
107                if (getClass() != obj.getClass())
108                        return false;
109                VCardRawLine other = (VCardRawLine) obj;
110                if (group == null) {
111                        if (other.group != null)
112                                return false;
113                } else if (!group.equals(other.group))
114                        return false;
115                if (name == null) {
116                        if (other.name != null)
117                                return false;
118                } else if (!name.equals(other.name))
119                        return false;
120                if (parameters == null) {
121                        if (other.parameters != null)
122                                return false;
123                } else if (!parameters.equals(other.parameters))
124                        return false;
125                if (value == null) {
126                        if (other.value != null)
127                                return false;
128                } else if (!value.equals(other.value))
129                        return false;
130                return true;
131        }
132
133        public static class Builder {
134                private String group, name, value;
135                private VCardParameters parameters = new VCardParameters();
136
137                public Builder group(String group) {
138                        this.group = group;
139                        return this;
140                }
141
142                public Builder name(String name) {
143                        this.name = name;
144                        return this;
145                }
146
147                public Builder value(String value) {
148                        this.value = value;
149                        return this;
150                }
151
152                public Builder param(String name, String... values) {
153                        this.parameters.putAll(name, Arrays.asList(values));
154                        return this;
155                }
156
157                public VCardRawLine build() {
158                        if (name == null) {
159                                throw Messages.INSTANCE.getIllegalArgumentException(4);
160                        }
161                        return new VCardRawLine(group, name, parameters, value);
162                }
163        }
164}
