042/**
043 * Reads lines of text from a reader, transparently unfolding lines that are
044 * folded.
045 * @author Michael Angstadt
046 */
047public class FoldedLineReader extends BufferedReader {
048        /**
049         * Regular expression used to detect the first line of folded,
050         * "quoted-printable" property values.
051         */
052        private final Pattern foldedQuotedPrintableValueRegex = Pattern.compile("[^:]*?QUOTED-PRINTABLE.*?:.*?=", Pattern.CASE_INSENSITIVE);
053
054        private String lastLine;
055        private int lastLineNum = 0, lineCount = 0;
056        private final Charset charset;
057
058        /**
059         * Creates a folded line reader.
060         * @param reader the reader object to wrap
061         */
062        public FoldedLineReader(Reader reader) {
063                super(reader);
064
065                if (reader instanceof InputStreamReader) {
066                        InputStreamReader isr = (InputStreamReader) reader;
067                        String charsetStr = isr.getEncoding();
068                        charset = (charsetStr == null) ? null : Charset.forName(charsetStr);
069                } else {
070                        charset = null;
071                }
072        }
073
074        /**
075         * Creates a folded line reader.
076         * @param text the text to read
077         */
078        public FoldedLineReader(String text) {
079                this(new StringReader(text));
080        }
081
082        /**
083         * Gets the starting line number of the last unfolded line that was read.
084         * @return the line number
085         */
086        public int getLineNum() {
087                return lastLineNum;
088        }
089
090        /**
091         * Gets the character encoding of the reader.
092         * @return the character encoding or null if none is defined
093         */
094        public Charset getEncoding() {
095                return charset;
096        }
097
098        /**
099         * <p>
100         * Reads the next non-empty line.
101         * </p>
102         * <p>
103         * Empty lines must be ignored because some vCards (such as vCards created
104         * by iPhones) contain empty lines. These empty lines appear in between
105         * folded lines, which, if not ignored, will cause the parser to incorrectly
106         * parse the vCard.
107         * </p>
108         * @return the next non-empty line or null of EOF
109         * @throws IOException if there's a problem reading from the reader
110         */
111        private String readNonEmptyLine() throws IOException {
112                while (true) {
113                        String line = super.readLine();
114                        if (line == null) {
115                                return null;
116                        }
117
118                        lineCount++;
119                        if (line.length() > 0) {
120                                return line;
121                        }
122                }
123        }
124
125        /**
126         * Reads the next unfolded line.
127         * @return the next unfolded line or null if the end of the stream has been
128         * reached
129         * @throws IOException if there's a problem reading from the reader
130         */
131        @Override
132        public String readLine() throws IOException {
133                String wholeLine = (lastLine == null) ? readNonEmptyLine() : lastLine;
134                lastLine = null;
135                if (wholeLine == null) {
136                        //end of stream
137                        return null;
138                }
139
140                //@formatter:off
141                /*
142                 * Lines that are QUOTED-PRINTABLE are folded in a strange way. A "=" is
143                 * appended to the end of a line to signal that the next line is folded.
144                 * Also, each folded line is *not* prepend with whitespace (which it should
145                 * be, according to the 2.1 specs).
146                 * 
147                 * For example:
148                 * 
149                 * ------------
150                 * BEGIN:VCARD
151                 * NOTE;QUOTED-PRINTABLE: This is an=0D=0A=
152                 * annoyingly formatted=0D=0A=
153                 * note=
154                 * 
155                 * END:VCARD
156                 * ------------
157                 * 
158                 * In the example above, note how there is an empty line directly above
159                 * END. This is still part of the NOTE property value because the 3rd
160                 * line of NOTE ends with a "=".
161                 * 
162                 * This behavior has only been observed in Outlook vCards. >:(
163                 */
164                //@formatter:on
165
166                boolean foldedQuotedPrintableLine = foldedQuotedPrintableValueRegex.matcher(wholeLine).matches();
167                if (foldedQuotedPrintableLine) {
168                        //chop off the trailing "="
169                        wholeLine = chop(wholeLine);
170                }
171
172                lastLineNum = lineCount;
173                StringBuilder unfoldedLine = new StringBuilder(wholeLine);
174                while (true) {
175                        String line = foldedQuotedPrintableLine ? super.readLine() : readNonEmptyLine();
176                        if (line == null) {
177                                //end of stream
178                                break;
179                        }
180
181                        if (foldedQuotedPrintableLine) {
182                                //remove any folding whitespace
183                                line = ltrim(line);
184
185                                boolean endsInEquals = line.endsWith("=");
186                                if (endsInEquals) {
187                                        //chop off the trailing "="
188                                        line = chop(line);
189                                }
190
191                                unfoldedLine.append(line);
192
193                                if (endsInEquals) {
194                                        //there are more folded lines
195                                        continue;
196                                }
197
198                                //end of the folded line
199                                break;
200                        }
201
202                        if (line.length() > 0 && Character.isWhitespace(line.charAt(0))) {
203                                //the line is folded
204                                line = ltrim(line);
205                                unfoldedLine.append(line);
206                                continue;
207                        }
208
209                        lastLine = line;
210                        break;
211                }
212
213                return unfoldedLine.toString();
214        }
215
216        /**
217         * Removes the last character from a string.
218         * @param string the string
219         * @return the modified string
220         */
221        private static String chop(String string) {
222                return (string.length() > 0) ? string.substring(0, string.length() - 1) : string;
223        }
224}
