059/**
060 * <p>
061 * Parses {@link VCard} objects from a JSON data stream (jCard format).
062 * </p>
063 * <p>
064 * <b>Example:</b>
065 * </p>
066 * 
067 * <pre class="brush:java">
068 * File file = new File("vcards.json");
069 * JCardReader reader = null;
070 * try {
071 *   reader = new JCardReader(file);
072 *   VCard vcard;
073 *   while ((vcard = reader.readNext()) != null) {
074 *     //...
075 *   }
076 * } finally {
077 *   if (reader != null) reader.close();
078 * }
079 * </pre>
080 * @author Michael Angstadt
081 * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
082 */
083public class JCardReader extends StreamReader {
084        private final JCardRawReader reader;
085
086        /**
087         * @param json the JSON string to read from
088         */
089        public JCardReader(String json) {
090                this(new StringReader(json));
091        }
092
093        /**
094         * @param in the input stream to read from
095         */
096        public JCardReader(InputStream in) {
097                this(utf8Reader(in));
098        }
099
100        /**
101         * @param file the file to read from
102         * @throws FileNotFoundException if the file doesn't exist
103         */
104        public JCardReader(File file) throws FileNotFoundException {
105                this(new BufferedReader(utf8Reader(file)));
106        }
107
108        /**
109         * @param reader the reader to read from
110         */
111        public JCardReader(Reader reader) {
112                this.reader = new JCardRawReader(reader);
113        }
114
115        /**
116         * @param parser the parser to read from
117         */
118        public JCardReader(JsonParser parser) {
119                this.reader = new JCardRawReader(parser, true);
120        }
121
122        @Override
123        protected VCard _readNext() throws IOException {
124                if (reader.eof()) {
125                        return null;
126                }
127
128                warnings.clear();
129
130                JCardDataStreamListenerImpl listener = new JCardDataStreamListenerImpl();
131                reader.readNext(listener);
132                VCard vcard = listener.vcard;
133                if (vcard != null && !listener.versionFound) {
134                        warnings.add(reader.getLineNum(), null, 29);
135                }
136                return vcard;
137        }
138
139        public void close() throws IOException {
140                reader.close();
141        }
142
143        private class JCardDataStreamListenerImpl implements JCardDataStreamListener {
144                private VCard vcard = null;
145                private boolean versionFound = false;
146
147                public void beginVCard() {
148                        vcard = new VCard();
149                        vcard.setVersion(VCardVersion.V4_0);
150                }
151
152                public void readProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) {
153                        if ("version".equalsIgnoreCase(propertyName)) {
154                                //don't unmarshal "version" because we don't treat it as a property
155                                versionFound = true;
156
157                                VCardVersion version = VCardVersion.valueOfByStr(value.asSingle());
158                                if (version != VCardVersion.V4_0) {
159                                        warnings.add(reader.getLineNum(), propertyName, 30);
160                                }
161                                return;
162                        }
163
164                        VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(propertyName);
165                        if (scribe == null) {
166                                scribe = new RawPropertyScribe(propertyName);
167                        }
168
169                        Result<? extends VCardProperty> result;
170                        try {
171                                result = scribe.parseJson(value, dataType, parameters);
172                                for (String warning : result.getWarnings()) {
173                                        warnings.add(reader.getLineNum(), warning, propertyName);
174                                }
175                        } catch (SkipMeException e) {
176                                warnings.add(reader.getLineNum(), propertyName, 22, e.getMessage());
177                                return;
178                        } catch (CannotParseException e) {
179                                scribe = new RawPropertyScribe(propertyName);
180                                result = scribe.parseJson(value, dataType, parameters);
181
182                                VCardProperty property = result.getProperty();
183                                String valueStr = ((RawProperty) property).getValue();
184                                warnings.add(reader.getLineNum(), propertyName, 25, valueStr, e.getMessage());
185                        } catch (EmbeddedVCardException e) {
186                                warnings.add(reader.getLineNum(), propertyName, 31);
187                                return;
188                        }
189
190                        VCardProperty property = result.getProperty();
191                        property.setGroup(group);
192                        vcard.addProperty(property);
193                }
194        }
195}
