036/**
037 * Marshals {@link Sound} properties.
038 * @author Michael Angstadt
039 */
040public class SoundScribe extends BinaryPropertyScribe<Sound, SoundType> {
041        public SoundScribe() {
042                super(Sound.class, "SOUND");
043        }
044
045        @Override
046        protected SoundType _mediaTypeFromTypeParameter(String type) {
047                return SoundType.get(type, null, null);
048        }
049
050        @Override
051        protected SoundType _mediaTypeFromMediaTypeParameter(String mediaType) {
052                return SoundType.get(null, mediaType, null);
053        }
054
055        @Override
056        protected SoundType _mediaTypeFromFileExtension(String extension) {
057                return SoundType.find(null, null, extension);
058        }
059
060        @Override
061        protected Sound _newInstance(String uri, SoundType contentType) {
062                return new Sound(uri, contentType);
063        }
064
065        @Override
066        protected Sound _newInstance(byte[] data, SoundType contentType) {
067                return new Sound(data, contentType);
068        }
069
070        @Override
071        protected Sound _parseHtml(HCardElement element, List<String> warnings) {
072                String elementName = element.tagName();
073                if (!"audio".equals(elementName) && !"source".equals(elementName)) {
074                        return super._parseHtml(element, warnings);
075                }
076
077                if ("audio".equals(elementName)) {
078                        //parse its child "<source>" element
079                        org.jsoup.nodes.Element source = element.getElement().getElementsByTag("source").first();
080                        if (source == null) {
081                                throw new CannotParseException(16);
082                        }
083
084                        element = new HCardElement(source);
085                }
086
087                String src = element.absUrl("src");
088                if (src.length() == 0) {
089                        throw new CannotParseException(17);
090                }
091
092                String type = element.attr("type");
093                SoundType mediaType = (type.length() == 0) ? null : _mediaTypeFromMediaTypeParameter(type);
094
095                try {
096                        DataUri uri = DataUri.parse(src);
097                        mediaType = _mediaTypeFromMediaTypeParameter(uri.getContentType());
098                        return new Sound(uri.getData(), mediaType);
099                } catch (IllegalArgumentException e) {
100                        //not a data URI
101                        if (mediaType == null) {
102                                String extension = getFileExtension(src);
103                                mediaType = (extension == null) ? null : _mediaTypeFromFileExtension(extension);
104                        }
105                        return new Sound(src, mediaType);
106                }
107        }
108}
