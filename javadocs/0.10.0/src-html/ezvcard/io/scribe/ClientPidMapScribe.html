043/**
044 * Marshals {@link ClientPidMap} properties.
045 * @author Michael Angstadt
046 */
047public class ClientPidMapScribe extends VCardPropertyScribe<ClientPidMap> {
048        public ClientPidMapScribe() {
049                super(ClientPidMap.class, "CLIENTPIDMAP");
050        }
051
052        @Override
053        protected VCardDataType _defaultDataType(VCardVersion version) {
054                return VCardDataType.TEXT;
055        }
056
057        @Override
058        protected String _writeText(ClientPidMap property, WriteContext context) {
059                SemiStructuredValueBuilder builder = new SemiStructuredValueBuilder();
060                builder.append(property.getPid());
061                builder.append(property.getUri());
062                return builder.build(context.isIncludeTrailingSemicolons());
063        }
064
065        @Override
066        protected ClientPidMap _parseText(String value, VCardDataType dataType, VCardVersion version, VCardParameters parameters, List<String> warnings) {
067                SemiStructuredValueIterator it = new SemiStructuredValueIterator(value, 2);
068                String pid = it.next();
069                String uri = it.next();
070                if (pid == null || uri == null) {
071                        throw new CannotParseException(3);
072                }
073
074                return parse(pid, uri);
075        }
076
077        @Override
078        protected void _writeXml(ClientPidMap property, XCardElement parent) {
079                Integer pid = property.getPid();
080                parent.append("sourceid", (pid == null) ? "" : pid.toString());
081
082                parent.append(VCardDataType.URI, property.getUri());
083        }
084
085        @Override
086        protected ClientPidMap _parseXml(XCardElement element, VCardParameters parameters, List<String> warnings) {
087                String sourceid = element.first("sourceid");
088                String uri = element.first(VCardDataType.URI);
089
090                if (uri == null && sourceid == null) {
091                        throw missingXmlElements(VCardDataType.URI.getName().toLowerCase(), "sourceid");
092                }
093                if (uri == null) {
094                        throw missingXmlElements(VCardDataType.URI);
095                }
096                if (sourceid == null) {
097                        throw missingXmlElements("sourceid");
098                }
099
100                return parse(sourceid, uri);
101        }
102
103        @Override
104        protected JCardValue _writeJson(ClientPidMap property) {
105                return JCardValue.structured(property.getPid(), property.getUri());
106        }
107
108        @Override
109        protected ClientPidMap _parseJson(JCardValue value, VCardDataType dataType, VCardParameters parameters, List<String> warnings) {
110                StructuredValueIterator it = new StructuredValueIterator(value.asStructured());
111                String pid = it.nextValue();
112                String uri = it.nextValue();
113                return parse(pid, uri);
114        }
115
116        private ClientPidMap parse(String pid, String uri) {
117                try {
118                        return new ClientPidMap(Integer.parseInt(pid), uri);
119                } catch (NumberFormatException e) {
120                        throw new CannotParseException(4);
121                }
122        }
123}
