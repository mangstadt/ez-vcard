148            //@Override
149            public void registerExtendedType(Class<? extends VCardType> clazz) {
150                    extendedTypeClasses.put(getTypeNameFromTypeClass(clazz), clazz);
151            }
152    
153            //@Override
154            public void unregisterExtendedType(Class<? extends VCardType> clazz) {
155                    extendedTypeClasses.remove(getTypeNameFromTypeClass(clazz));
156            }

158            //@Override
159            public List<String> getWarnings() {
160                    return new ArrayList<String>(warnings);
161            }

163            //@Override
164            public VCard readNext() throws IOException {
165                    if (reader.eof()) {
166                            return null;
167                    }
168    
169                    warnings.clear();
170    
171                    VCardDataStreamListenerImpl listener = new VCardDataStreamListenerImpl();
172                    reader.start(listener);
173    
174                    return listener.root;
175            }

177            /**
178             * Creates the appropriate {@link VCardType} instance, given the type name.
179             * This method does not unmarshal the type, it just creates the type object.
180             * @param name the type name (e.g. "FN")
181             * @return the Type that was created
182             */
183            private VCardType createTypeObject(String name) {
184                    Class<? extends VCardType> clazz = TypeList.getTypeClass(name);
185                    VCardType t;
186                    if (clazz != null) {
187                            try {
188                                    //create a new instance of the class
189                                    t = clazz.newInstance();
190                            } catch (Exception e) {
191                                    //it is the responsibility of the ez-vcard developer to ensure that this exception is never thrown
192                                    //all type classes defined in the ez-vcard library MUST have public, no-arg constructors
193                                    throw new RuntimeException(e);
194                            }
195                    } else {
196                            Class<? extends VCardType> extendedTypeClass = extendedTypeClasses.get(name);
197                            if (extendedTypeClass != null) {
198                                    try {
199                                            t = extendedTypeClass.newInstance();
200                                    } catch (Exception e) {
201                                            //this should never happen because the type class is checked to see if it has a public, no-arg constructor in the "registerExtendedType" method
202                                            throw new RuntimeException("Extended type class \"" + extendedTypeClass.getName() + "\" must have a public, no-arg constructor.");
203                                    }
204                            } else {
205                                    t = new RawType(name); //use RawType instead of TextType because we don't want to unescape any characters that might be meaningful to this type
206                                    if (!name.startsWith("X-")) {
207                                            addWarning("Non-standard property found.  Treating it as an extended property.", name);
208                                    }
209                            }
210                    }
211                    return t;
212            }

214            /**
215             * Gets the type name from a type class.
216             * @param clazz the type class
217             * @return the type name
218             */
219            private String getTypeNameFromTypeClass(Class<? extends VCardType> clazz) {
220                    try {
221                            VCardType t = clazz.newInstance();
222                            return t.getTypeName().toUpperCase();
223                    } catch (Exception e) {
224                            //there is no public, no-arg constructor
225                            throw new RuntimeException(e);
226                    }
227            }

229            /**
230             * Assigns names to all nameless parameters. v3.0 and v4.0 requires all
231             * parameters to have names, but v2.1 does not.
232             * @param parameters the parameters
233             */
234            private void handleNamelessParameters(VCardSubTypes parameters) {
235                    List<String> namelessParamValues = parameters.get(null);
236                    for (String paramValue : namelessParamValues) {
237                            String paramName;
238                            if (ValueParameter.valueOf(paramValue) != null) {
239                                    paramName = ValueParameter.NAME;
240                            } else if (EncodingParameter.valueOf(paramValue) != null) {
241                                    paramName = EncodingParameter.NAME;
242                            } else {
243                                    //otherwise, assume it's a TYPE
244                                    paramName = TypeParameter.NAME;
245                            }
246                            parameters.put(paramName, paramValue);
247                    }
248                    parameters.removeAll(null);
249            }
250    
251            /**
252             * <p>
253             * Accounts for multi-valued TYPE parameters being enclosed entirely in
254             * double quotes (for example: ADR;TYPE="home,work").
255             * </p>
256             * <p>
257             * Many examples throughout the 4.0 specs show TYPE parameters being encoded
258             * in this way. This conflicts with the ABNF and is noted in the errata.
259             * This method will split the value by comma incase the vendor implemented
260             * it this way.
261             * </p>
262             * @param parameters the parameters
263             */
264            private void handleQuotedMultivaluedTypeParams(VCardSubTypes parameters) {
265                    //account for multi-valued TYPE parameters being enclosed entirely in double quotes
266                    //e.g. ADR;TYPE="home,work"
267                    List<String> typeParams = new ArrayList<String>(parameters.get(TypeParameter.NAME));
268                    for (String typeParam : typeParams) {
269                            if (!typeParam.contains(",")) {
270                                    continue;
271                            }
272    
273                            parameters.remove(TypeParameter.NAME, typeParam);
274                            for (String splitValue : typeParam.split(",")) {
275                                    parameters.put(TypeParameter.NAME, splitValue);
276                            }
277                    }
278            }
279    
280            /**
281             * Decodes the property value if it's encoded in quoted-printable encoding.
282             * Quoted-printable encoding is only supported in v2.1.
283             * @param name the property name
284             * @param parameters the parameters
285             * @param value the property value
286             * @return the decoded property value
287             */
288            private String decodeQuotedPrintable(String name, VCardSubTypes parameters, String value) {
289                    if (parameters.getEncoding() != EncodingParameter.QUOTED_PRINTABLE) {
290                            return value;
291                    }
292    
293                    parameters.setEncoding(null); //remove encoding sub type
294    
295                    QuotedPrintableCodec codec = new QuotedPrintableCodec();
296                    String charset = parameters.getCharset();
297                    try {
298                            if (charset == null) {
299                                    return codec.decode(value);
300                            } else {
301                                    try {
302                                            return codec.decode(value, charset);
303                                    } catch (UnsupportedEncodingException e) {
304                                            addWarning("The specified charset is not supported.  Using default charset instead: " + charset, name);
305                                            return codec.decode(value);
306                                    }
307                            }
308                    } catch (DecoderException e) {
309                            addWarning("Property value was marked as \"quoted-printable\", but it could not be decoded.  Treating the value as plain text.", name);
310                    }
311    
312                    return value;
313            }

319                    reader.close();
320            }

322            private void addWarning(String message) {
323                    addWarning(message, null);
324            }
325    
326            private void addWarning(String message, String propertyName) {
327                    StringBuilder sb = new StringBuilder();
328                    sb.append("Line ").append(reader.getLineNum());
329                    if (propertyName != null) {
330                            sb.append(" (").append(propertyName).append(" property)");
331                    }
332                    sb.append(": ").append(message);
333    
334                    warnings.add(sb.toString());
335            }

337            private class VCardDataStreamListenerImpl implements VCardRawReader.VCardDataStreamListener {
338                    private VCard root;
339                    private final List<LabelType> labels = new ArrayList<LabelType>();
340                    private final List<String> warningsBuf = new ArrayList<String>();
341                    private final LinkedList<VCard> vcardStack = new LinkedList<VCard>();
342                    private EmbeddedVCardException embeddedVCardException;
343    
344                    public void beginComponent(String name) {
345                            if (!"VCARD".equalsIgnoreCase(name)) {
346                                    return;
347                            }
348    
349                            VCard vcard = new VCard();
350    
351                            //initialize version to 2.1, since the VERSION property can exist anywhere in a 2.1 vCard
352                            vcard.setVersion(VCardVersion.V2_1);
353    
354                            vcardStack.add(vcard);
355    
356                            if (root == null) {
357                                    root = vcard;
358                            }
359    
360                            if (embeddedVCardException != null) {
361                                    embeddedVCardException.injectVCard(vcard);
362                                    embeddedVCardException = null;
363                            }
364                    }
365    
366                    public void readVersion(VCardVersion version) {
367                            if (vcardStack.isEmpty()) {
368                                    //not in a "VCARD" component
369                                    return;
370                            }
371    
372                            vcardStack.getLast().setVersion(version);
373                    }
374    
375                    public void readProperty(String group, String name, VCardSubTypes parameters, String value) {
376                            if (vcardStack.isEmpty()) {
377                                    //not in a "VCARD" component
378                                    return;
379                            }
380    
381                            if (embeddedVCardException != null) {
382                                    //the next property was supposed to be the start of a nested vCard, but it wasn't
383                                    embeddedVCardException.injectVCard(null);
384                                    embeddedVCardException = null;
385                            }
386    
387                            handleNamelessParameters(parameters);
388    
389                            handleQuotedMultivaluedTypeParams(parameters);
390    
391                            value = decodeQuotedPrintable(name, parameters, value);
392    
393                            //create the type object
394                            VCardType type = createTypeObject(name);
395                            type.setGroup(group);
396    
397                            //unmarshal the text string into the object
398                            VCard curVCard = vcardStack.getLast();
399                            VCardVersion version = curVCard.getVersion();
400                            warningsBuf.clear();
401                            try {
402                                    type.unmarshalText(parameters, value, version, warningsBuf, compatibilityMode);
403    
404                                    //add to vcard
405                                    if (type instanceof LabelType) {
406                                            //LABELs must be treated specially so they can be matched up with their ADRs
407                                            labels.add((LabelType) type);
408                                    } else {
409                                            curVCard.addType(type);
410                                    }
411                            } catch (SkipMeException e) {
412                                    warningsBuf.add("Property has requested that it be skipped: " + e.getMessage());
413                            } catch (EmbeddedVCardException e) {
414                                    //parse an embedded vCard (i.e. the AGENT type)
415    
416                                    if (value.length() == 0 || version == VCardVersion.V2_1) {
417                                            //a nested vCard is expected to be next (2.1 style)
418                                            embeddedVCardException = e;
419                                    } else {
420                                            //the property value should be an embedded vCard (3.0 style)
421                                            value = VCardStringUtils.unescape(value);
422                                            VCardReader agentReader = new VCardReader(value);
423    
424                                            agentReader.setCompatibilityMode(compatibilityMode);
425                                            try {
426                                                    VCard nestedVCard = agentReader.readNext();
427                                                    if (nestedVCard != null) {
428                                                            e.injectVCard(nestedVCard);
429                                                    }
430                                            } catch (IOException e2) {
431                                                    //reading from a string
432                                            } finally {
433                                                    for (String w : agentReader.getWarnings()) {
434                                                            addWarning("Problem unmarshalling nested vCard value: " + w, type.getTypeName());
435                                                    }
436                                            }
437                                    }
438    
439                                    curVCard.addType(type);
440                            } finally {
441                                    for (String warning : warningsBuf) {
442                                            addWarning(warning, name);
443                                    }
444                            }
445                    }
446    
447                    public void endComponent(String name) {
448                            if (vcardStack.isEmpty()) {
449                                    //not in a "VCARD" component
450                                    return;
451                            }
452    
453                            if (!"VCARD".equalsIgnoreCase(name)) {
454                                    //not a "VCARD" component
455                                    return;
456                            }
457    
458                            VCard curVCard = vcardStack.removeLast();
459    
460                            //assign labels to their addresses
461                            for (LabelType label : labels) {
462                                    boolean orphaned = true;
463                                    for (AddressType adr : curVCard.getAddresses()) {
464                                            if (adr.getLabel() == null && adr.getTypes().equals(label.getTypes())) {
465                                                    adr.setLabel(label.getValue());
466                                                    orphaned = false;
467                                                    break;
468                                            }
469                                    }
470                                    if (orphaned) {
471                                            curVCard.addOrphanedLabel(label);
472                                    }
473                            }
474    
475                            if (vcardStack.isEmpty()) {
476                                    throw new VCardRawReader.StopReadingException();
477                            }
478                    }
479    
480                    public void invalidLine(String line) {
481                            if (vcardStack.isEmpty()) {
482                                    //not in a "VCARD" component
483                                    return;
484                            }
485    
486                            addWarning("Skipping malformed line: \"" + line + "\"");
487                    }
488    
489                    public void invalidVersion(String version) {
490                            if (vcardStack.isEmpty()) {
491                                    //not in a "VCARD" component
492                                    return;
493                            }
494    
495                            addWarning("Ignoring invalid version value: " + version, "VERSION");
496                    }
497            }
