049/**
050 * Writes data to an vCard data stream.
051 * @author Michael Angstadt
052 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
053 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
054 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
055 */
056public class VCardRawWriter implements Closeable, Flushable {
057        /**
058         * If any of these characters are found within a parameter value, then the
059         * entire parameter value must be wrapped in double quotes (applies to
060         * versions 3.0 and 4.0 only).
061         */
062        private final CharacterBitSet specialParameterCharacters = new CharacterBitSet(",:;");
063
064        /**
065         * Regular expression used to detect newline character sequences.
066         */
067        private final Pattern newlineRegex = Pattern.compile("\\r\\n|\\r|\\n");
068        private final CharacterBitSet newlineBitSet = new CharacterBitSet("\r\n");
069
070        /**
071         * List of characters which would break the syntax of the vCard if used
072         * inside a property name. The list of characters permitted by the
073         * specification is much more strict, but the goal here is to be as lenient
074         * as possible.
075         */
076        private final CharacterBitSet invalidPropertyNameCharacters = new CharacterBitSet(".;:\n\r");
077
078        /**
079         * List of characters which would break the syntax of the vCard if used
080         * inside a group name. The list of characters permitted by the
081         * specification is much more strict, but the goal here is to be as lenient
082         * as possible.
083         */
084        private final CharacterBitSet invalidGroupNameCharacters = invalidPropertyNameCharacters;
085
086        /**
087         * List of characters which would break the syntax of the vCard if used
088         * inside a parameter name. The list of characters permitted by the
089         * specification is much more strict, but the goal here is to be as lenient
090         * as possible.
091         */
092        private final CharacterBitSet invalidParameterNameCharacters = new CharacterBitSet(";:=\n\r");
093
094        /**
095         * List of characters which would break the syntax of the vCard if used
096         * inside a parameter value with caret encoding disabled. These characters
097         * cannot be escaped or encoded, so they are impossible to include inside of
098         * a parameter value. The list of characters permitted by the specification
099         * is much more strict, but the goal here is to be as lenient as possible.
100         */
101        private final Map<VCardVersion, CharacterBitSet> invalidParamValueChars;
102        {
103                Map<VCardVersion, CharacterBitSet> map = new EnumMap<VCardVersion, CharacterBitSet>(VCardVersion.class);
104
105                map.put(VCardVersion.V2_1, new CharacterBitSet(",:\n\r")); //note: semicolons can be escaped
106                map.put(VCardVersion.V3_0, new CharacterBitSet("\"\r\n"));
107                map.put(VCardVersion.V4_0, new CharacterBitSet("\""));
108
109                invalidParamValueChars = Collections.unmodifiableMap(map);
110        }
111
112        /**
113         * List of characters which would break the syntax of the vCard if used
114         * inside a parameter value with caret encoding enabled. These characters
115         * cannot be escaped or encoded, so they are impossible to include inside of
116         * a parameter value. The list of characters permitted by the specification
117         * is much more strict, but the goal here is to be as lenient as possible.
118         */
119        private final Map<VCardVersion, CharacterBitSet> invalidParamValueCharsWithCaretEncoding;
120        {
121                Map<VCardVersion, CharacterBitSet> map = new EnumMap<VCardVersion, CharacterBitSet>(VCardVersion.class);
122
123                map.put(VCardVersion.V2_1, invalidParamValueChars.get(VCardVersion.V2_1)); //2.1 does not support caret encoding
124                map.put(VCardVersion.V3_0, new CharacterBitSet(""));
125                map.put(VCardVersion.V4_0, new CharacterBitSet(""));
126
127                invalidParamValueCharsWithCaretEncoding = Collections.unmodifiableMap(map);
128        }
129
130        private final FoldedLineWriter writer;
131        private boolean caretEncodingEnabled = false;
132        private VCardVersion version;

135         * @param writer the writer to wrap

138        public VCardRawWriter(Writer writer, VCardVersion version) {
139                this.writer = new FoldedLineWriter(writer);
140                this.version = version;
141        }

151        /**
152         * <p>
153         * Gets whether the writer will apply circumflex accent encoding on
154         * parameter values (disabled by default, only applies to 3.0 and 4.0
155         * vCards). This escaping mechanism allows for newlines and double quotes to
156         * be included in parameter values.
157         * </p>
158         * 
159         * <p>
160         * Note that this encoding mechanism is defined separately from the vCard
161         * specification and may not be supported by the vCard consumer.
162         * </p>
163         * 
164         * <table class="simpleTable">
165         * <tr>
166         * <th>Raw Character</th>
167         * <th>Encoded Character</th>
168         * </tr>
169         * <tr>
170         * <td>{@code "}</td>
171         * <td>{@code ^'}</td>
172         * </tr>
173         * <tr>
174         * <td><i>newline</i></td>
175         * <td>{@code ^n}</td>
176         * </tr>
177         * <tr>
178         * <td>{@code ^}</td>
179         * <td>{@code ^^}</td>
180         * </tr>
181         * </table>
182         * 
183         * <p>
184         * Example:
185         * </p>
186         * 
187         * <pre>
188         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPittsburgh, PA 15212":40.446816;80.00566
189         * </pre>
190         * 
191         * @return true if circumflex accent encoding is enabled, false if not
192         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
193         */
194        public boolean isCaretEncodingEnabled() {
195                return caretEncodingEnabled;
196        }

199         * <p>
200         * Sets whether the writer will apply circumflex accent encoding on
201         * parameter values (disabled by default, only applies to 3.0 and 4.0
202         * vCards). This escaping mechanism allows for newlines and double quotes to
203         * be included in parameter values.
204         * </p>
205         * 
206         * <p>
207         * Note that this encoding mechanism is defined separately from the vCard
208         * specification and may not be supported by the vCard consumer.
209         * </p>
210         * 
211         * <table class="simpleTable">
212         * <tr>
213         * <th>Raw Character</th>
214         * <th>Encoded Character</th>
215         * </tr>
216         * <tr>
217         * <td>{@code "}</td>
218         * <td>{@code ^'}</td>
219         * </tr>
220         * <tr>
221         * <td><i>newline</i></td>

224         * <tr>

237         *

240         */

242                caretEncodingEnabled = enable;
243        }

288        /**
289         * Writes a property to the vCard data stream.
290         * @param propertyName the property name (e.g. "FN")
291         * @param value the property value
292         * @throws IllegalArgumentException if the property data contains one or
293         * more characters which break the vCard syntax and which cannot be escaped
294         * or encoded
295         * @throws IOException if there's a problem writing to the data stream
296         */
297        public void writeProperty(String propertyName, String value) throws IOException {
298                writeProperty(null, propertyName, new VCardParameters(), value);
299        }
300
301        /**
302         * Writes a property to the vCard data stream.
303         * @param group the group or null if there is no group
304         * @param propertyName the property name (e.g. "FN")
305         * @param parameters the property parameters
306         * @param value the property value (will be converted to "quoted-printable"
307         * encoding if the {@link Encoding#QUOTED_PRINTABLE} parameter is set)
308         * @throws IllegalArgumentException if the property data contains one or
309         * more characters which break the vCard syntax and which cannot be escaped
310         * or encoded
311         * @throws IOException if there's a problem writing to the data stream
312         */
313        public void writeProperty(String group, String propertyName, VCardParameters parameters, String value) throws IOException {
314                //validate the group name
315                if (group != null) {
316                        if (invalidGroupNameCharacters.containsAny(group)) {
317                                throw Messages.INSTANCE.getIllegalArgumentException(7, propertyName, group, printableCharacterList(invalidGroupNameCharacters.characters()));
318                        }
319                        if (beginsWithWhitespace(group)) {
320                                throw Messages.INSTANCE.getIllegalArgumentException(8, propertyName, group);
321                        }
322                }
323
324                //validate the property name
325                if (invalidPropertyNameCharacters.containsAny(propertyName)) {
326                        throw Messages.INSTANCE.getIllegalArgumentException(9, propertyName, printableCharacterList(invalidPropertyNameCharacters.characters()));
327                }
328                if (beginsWithWhitespace(propertyName)) {
329                        throw Messages.INSTANCE.getIllegalArgumentException(10, propertyName);
330                }
331
332                value = sanitizePropertyValue(value, parameters);
333
334                /*
335                 * Determine if the property value must be encoded in quoted printable
336                 * encoding. If so, then determine what charset to use for the encoding.
337                 */
338                boolean useQuotedPrintable = (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE);
339                Charset quotedPrintableCharset = null;
340                if (useQuotedPrintable) {
341                        String charsetParam = parameters.getCharset();
342                        if (charsetParam == null) {
343                                quotedPrintableCharset = Charset.forName("UTF-8");
344                        } else {
345                                try {
346                                        quotedPrintableCharset = Charset.forName(charsetParam);
347                                } catch (Throwable t) {
348                                        quotedPrintableCharset = Charset.forName("UTF-8");
349                                }
350                        }
351                        parameters.setCharset(quotedPrintableCharset.name());
352                }
353
354                //write the group
355                if (group != null) {
356                        writer.append(group).append('.');
357                }
358
359                //write the property name
360                writer.append(propertyName);
361
362                //write the parameters
363                for (Map.Entry<String, List<String>> parameter : parameters) {
364                        String parameterName = parameter.getKey();
365                        List<String> parameterValues = parameter.getValue();
366                        if (parameterValues.isEmpty()) {
367                                continue;
368                        }
369
370                        //check the parameter name for invalid characters
371                        if (invalidParameterNameCharacters.containsAny(parameterName)) {
372                                throw Messages.INSTANCE.getIllegalArgumentException(11, propertyName, parameterName, printableCharacterList(invalidParameterNameCharacters.characters()));
373                        }
374
375                        if (version == VCardVersion.V2_1) {
376                                boolean isTypeParameter = VCardParameters.TYPE.equalsIgnoreCase(parameterName);
377                                for (String parameterValue : parameterValues) {
378                                        parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
379
380                                        if (isTypeParameter) {
381                                                //e.g. ADR;HOME;WORK:
382                                                writer.append(';').append(parameterValue.toUpperCase());
383                                        } else {
384                                                //e.g. ADR;FOO=bar;FOO=car:
385                                                writer.append(';').append(parameterName).append('=').append(parameterValue);
386                                        }
387                                }
388                                continue;
389                        }
390
391                        //e.g. ADR;TYPE=home,work,"another,value":
392                        boolean first = true;
393                        writer.append(';').append(parameterName).append('=');
394                        for (String parameterValue : parameterValues) {
395                                if (!first) {
396                                        writer.append(',');
397                                }
398
399                                parameterValue = sanitizeParameterValue(parameterValue, parameterName, propertyName);
400                                if (specialParameterCharacters.containsAny(parameterValue)) {
401                                        writer.append('"').append(parameterValue).append('"');
402                                } else {
403                                        writer.append(parameterValue);
404                                }
405
406                                first = false;
407                        }
408                }
409
410                writer.append(':');
411                writer.append(value, useQuotedPrintable, quotedPrintableCharset);
412                writer.append(writer.getNewline());
413        }

415        /**
416         * Determines if a given string starts with whitespace.
417         * @param string the string
418         * @return true if it starts with whitespace, false if not
419         */
420        private boolean beginsWithWhitespace(String string) {
421                if (string.length() == 0) {
422                        return false;
423                }
424                char first = string.charAt(0);
425                return (first == ' ' || first == '\t');
426        }

428        /**
429         * Sanitizes a property value for safe inclusion in a vCard.
430         * @param value the value to sanitize
431         * @param parameters the property's parameters
432         * @return the sanitized value
433         */
434        private String sanitizePropertyValue(String value, VCardParameters parameters) {
435                if (value == null) {
436                        return "";
437                }
438
439                /*
440                 * 2.1 does not support the "\n" escape sequence (see "Delimiters"
441                 * sub-section in section 2 of the specs) so encode the value in
442                 * quoted-printable encoding if any newline characters exist.
443                 */
444                if (version == VCardVersion.V2_1 && newlineBitSet.containsAny(value)) {
445                        parameters.setEncoding(Encoding.QUOTED_PRINTABLE);
446                        return value;
447                }
448
449                return escapeNewlines(value);
450        }
451
452        /**
453         * Sanitizes a parameter value for serialization.
454         * @param parameterValue the parameter value
455         * @param parameterName the parameter name
456         * @param propertyName the name of the property to which the parameter
457         * belongs
458         * @return the sanitized parameter value
459         * @throws IllegalArgumentException if the value contains invalid characters
460         */
461        private String sanitizeParameterValue(String parameterValue, String parameterName, String propertyName) {
462                CharacterBitSet invalidChars = (caretEncodingEnabled ? invalidParamValueCharsWithCaretEncoding : invalidParamValueChars).get(version);
463                if (invalidChars.containsAny(parameterValue)) {
464                        throw Messages.INSTANCE.getIllegalArgumentException(12, propertyName, parameterName, printableCharacterList(invalidChars.characters()));
465                }
466
467                String sanitizedValue = parameterValue;
468                switch (version) {
469                case V2_1:
470                        //Note: 2.1 does not support caret encoding.
471
472                        //escape backslashes
473                        sanitizedValue = sanitizedValue.replace("\\", "\\\\");
474
475                        //escape semi-colons (see section 2)
476                        return sanitizedValue.replace(";", "\\;");
477
478                case V3_0:
479                        if (caretEncodingEnabled) {
480                                return applyCaretEncoding(sanitizedValue);
481                        }
482                        return sanitizedValue;
483
484                case V4_0:
485                        if (caretEncodingEnabled) {
486                                return applyCaretEncoding(sanitizedValue);
487                        }
488
489                        /*
490                         * 4.0 allows newlines to be escaped (for the LABEL parameter).
491                         */
492                        return newlineRegex.matcher(sanitizedValue).replaceAll("\\\\\\n");
493                }
494
495                return "";
496        }
497
498        private String printableCharacterList(String list) {
499                return list.replace("\n", "\\n").replace("\r", "\\r");
500        }
501
502        /**
503         * Applies circumflex accent encoding to a string.
504         * @param value the string
505         * @return the encoded string
506         */
507        private String applyCaretEncoding(String value) {
508                value = value.replace("^", "^^");
509                value = newlineRegex.matcher(value).replaceAll("^n");
510                value = value.replace("\"", "^'");
511                return value;
512        }
513
514        /**
515         * Escapes all newlines in a string.
516         * @param string the string to escape
517         * @return the escaped string
518         */
519        private String escapeNewlines(String string) {
520                return newlineRegex.matcher(string).replaceAll("\\\\n");
521        }
