050/**
051 * Parses the components out of each line in a plain-text vCard data stream.
052 * @author Michael Angstadt
053 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
054 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
055 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
056 */
057public class VCardRawReader implements Closeable {
058        private final Reader reader;
059        private final ClearableStringBuilder buffer = new ClearableStringBuilder();
060        private final ClearableStringBuilder unfoldedLine = new ClearableStringBuilder();
061        private final Map<String, VCardVersion> versionAliases = new HashMap<String, VCardVersion>();
062        {
063                for (VCardVersion version : VCardVersion.values()) {
064                        versionAliases.put(version.getVersion(), version);
065                }
066        }
067
068        private boolean eos = false;
069        private boolean caretDecodingEnabled = true;
070        private VCardVersion version = VCardVersion.V2_1; //initialize to 2.1, since the VERSION property can exist anywhere in the file in this version
071        private int prevChar = -1;
072        private int propertyLineNum = 1;
073        private int lineNum = 1;
074
075        /**
076         * @param reader the reader to read from
077         */
078        public VCardRawReader(Reader reader) {
079                this.reader = reader;
080        }
081
082        /**
083         * Gets the line number of the line that was just read. If the line was
084         * folded, this will be the line number of the first line.
085         * @return the line number
086         */
087        public int getLineNumber() {
088                return propertyLineNum;
089        }
090
091        /**
092         * Gets the vCard version that the reader is currently parsing with.
093         * @return the vCard version
094         */
095        public VCardVersion getVersion() {
096                return version;
097        }
098
099        /**
100         * Reads the next line from the input stream. Folded lines are automatically
101         * unfolded.
102         * @return the parsed components of the line
103         * @throws VCardParseException if a line cannot be parsed
104         * @throws IOException if there's a problem reading from the stream
105         */
106        public VCardRawLine readLine() throws IOException {
107                if (eos) {
108                        return null;
109                }
110
111                propertyLineNum = lineNum;
112                buffer.clear();
113                unfoldedLine.clear();
114
115                /*
116                 * The property's group.
117                 */
118                String group = null;
119
120                /*
121                 * The property's name.
122                 */
123                String propertyName = null;
124
125                /*
126                 * The name of the parameter we're currently inside of.
127                 */
128                String curParamName = null;
129
130                /*
131                 * The property's parameters.
132                 */
133                VCardParameters parameters = new VCardParameters();
134
135                /*
136                 * The character that used to escape the current character (for
137                 * parameter values).
138                 */
139                char escapeChar = 0;
140
141                /*
142                 * Are we currently inside a parameter value that is surrounded with
143                 * double-quotes?
144                 */
145                boolean inQuotes = false;
146
147                /*
148                 * Are we currently inside the property value?
149                 */
150                boolean inValue = false;
151
152                /*
153                 * Does the line use quoted-printable encoding, and does it end all of
154                 * its folded lines with a "=" character?
155                 */
156                boolean quotedPrintableLine = false;
157
158                /*
159                 * The current character.
160                 */
161                char ch = 0;
162
163                /*
164                 * The previous character.
165                 */
166                char prevChar;
167
168                while (true) {
169                        prevChar = ch;
170
171                        int read = nextChar();
172                        if (read < 0) {
173                                eos = true;
174                                break;
175                        }
176
177                        ch = (char) read;
178
179                        if (prevChar == '\r' && ch == '\n') {
180                                /*
181                                 * The newline was already processed when the "\r" character was
182                                 * encountered, so ignore the accompanying "\n" character.
183                                 */
184                                continue;
185                        }
186
187                        if (isNewline(ch)) {
188                                quotedPrintableLine = (inValue && prevChar == '=' && isQuotedPrintable(parameters));
189                                if (quotedPrintableLine) {
190                                        /*
191                                         * Remove the "=" character that some vCards put at the end
192                                         * of quoted-printable lines that are followed by a folded
193                                         * line.
194                                         */
195                                        buffer.chop();
196                                        unfoldedLine.chop();
197                                }
198
199                                //keep track of the current line number
200                                lineNum++;
201
202                                continue;
203                        }
204
205                        if (isNewline(prevChar)) {
206                                if (isWhitespace(ch)) {
207                                        /*
208                                         * This line is a continuation of the previous line (the
209                                         * line is folded).
210                                         */
211                                        continue;
212                                }
213
214                                if (quotedPrintableLine) {
215                                        /*
216                                         * The property's parameters indicate that the property
217                                         * value is quoted-printable. And the previous line ended
218                                         * with an equals sign. This means that folding whitespace
219                                         * may not be prepended to folded lines like it should...
220                                         */
221                                } else {
222                                        /*
223                                         * We're reached the end of the property.
224                                         */
225                                        this.prevChar = ch;
226                                        break;
227                                }
228                        }
229
230                        unfoldedLine.append(ch);
231
232                        if (inValue) {
233                                buffer.append(ch);
234                                continue;
235                        }
236
237                        if (escapeChar != 0) {
238                                //this character was escaped
239                                if (escapeChar == '\\') {
240                                        if (ch == '\\') {
241                                                buffer.append(ch);
242                                        } else if (ch == 'n' || ch == 'N') {
243                                                //newlines appear as "\n" or "\N" (see RFC 2426 p.7)
244                                                buffer.append(NEWLINE);
245                                        } else if (ch == '"' && version != VCardVersion.V2_1) {
246                                                //double quotes don't need to be escaped in 2.1 parameter values because they have no special meaning
247                                                buffer.append(ch);
248                                        } else if (ch == ';' && version == VCardVersion.V2_1) {
249                                                //semi-colons can only be escaped in 2.1 parameter values (see section 2 of specs)
250                                                //if a 3.0/4.0 param value has semi-colons, the value should be surrounded in double quotes
251                                                buffer.append(ch);
252                                        } else {
253                                                //treat the escape character as a normal character because it's not a valid escape sequence
254                                                buffer.append(escapeChar).append(ch);
255                                        }
256                                } else if (escapeChar == '^') {
257                                        if (ch == '^') {
258                                                buffer.append(ch);
259                                        } else if (ch == 'n') {
260                                                buffer.append(NEWLINE);
261                                        } else if (ch == '\'') {
262                                                buffer.append('"');
263                                        } else {
264                                                //treat the escape character as a normal character because it's not a valid escape sequence
265                                                buffer.append(escapeChar).append(ch);
266                                        }
267                                }
268                                escapeChar = 0;
269                                continue;
270                        }
271
272                        if (ch == '\\' || (ch == '^' && version != VCardVersion.V2_1 && caretDecodingEnabled)) {
273                                //an escape character was read
274                                escapeChar = ch;
275                                continue;
276                        }
277
278                        if (ch == '.' && group == null && propertyName == null) {
279                                //set the group
280                                group = buffer.getAndClear();
281                                continue;
282                        }
283
284                        if ((ch == ';' || ch == ':') && !inQuotes) {
285                                if (propertyName == null) {
286                                        //property name
287                                        propertyName = buffer.getAndClear();
288                                } else {
289                                        //parameter value
290                                        String paramValue = buffer.getAndClear();
291                                        if (version == VCardVersion.V2_1) {
292                                                //2.1 allows whitespace to surround the "=", so remove it
293                                                paramValue = StringUtils.ltrim(paramValue);
294                                        }
295                                        parameters.put(curParamName, paramValue);
296                                        curParamName = null;
297                                }
298
299                                if (ch == ':') {
300                                        //the rest of the line is the property value
301                                        inValue = true;
302                                }
303                                continue;
304                        }
305
306                        if (ch == ',' && !inQuotes && version != VCardVersion.V2_1) {
307                                //multi-valued parameter
308                                parameters.put(curParamName, buffer.getAndClear());
309                                continue;
310                        }
311
312                        if (ch == '=' && curParamName == null) {
313                                //parameter name
314                                String paramName = buffer.getAndClear();
315                                if (version == VCardVersion.V2_1) {
316                                        //2.1 allows whitespace to surround the "=", so remove it
317                                        paramName = StringUtils.rtrim(paramName);
318                                }
319                                curParamName = paramName;
320                                continue;
321                        }
322
323                        if (ch == '"' && version != VCardVersion.V2_1) {
324                                //2.1 doesn't use the quoting mechanism
325                                inQuotes = !inQuotes;
326                                continue;
327                        }
328
329                        buffer.append(ch);
330                }
331
332                if (unfoldedLine.length() == 0) {
333                        //input stream was empty
334                        return null;
335                }
336
337                if (propertyName == null) {
338                        throw new VCardParseException(unfoldedLine.get(), propertyLineNum, Messages.INSTANCE.getExceptionMessage(5));
339                }
340
341                String value = buffer.getAndClear();
342
343                if ("VERSION".equalsIgnoreCase(propertyName)) {
344                        VCardVersion version = versionAliases.get(value);
345                        if (version == null) {
346                                throw new VCardParseException(unfoldedLine.get(), propertyLineNum, Messages.INSTANCE.getExceptionMessage(6));
347                        }
348                        this.version = version;
349                }
350
351                return new VCardRawLine(group, propertyName, parameters, value);
352        }

354        /**
355         * <p>
356         * Gets whether the reader will decode parameter values that use circumflex
357         * accent encoding (enabled by default). This escaping mechanism allows
358         * newlines and double quotes to be included in parameter values.
359         * </p>
360         * 
361         * <table class="simpleTable">
362         * <tr>
363         * <th>Raw Character</th>
364         * <th>Encoded Character</th>
365         * </tr>
366         * <tr>
367         * <td>{@code "}</td>
368         * <td>{@code ^'}</td>
369         * </tr>
370         * <tr>
371         * <td><i>newline</i></td>
372         * <td>{@code ^n}</td>
373         * </tr>
374         * <tr>
375         * <td>{@code ^}</td>
376         * <td>{@code ^^}</td>
377         * </tr>
378         * </table>
379         * 
380         * <p>
381         * Example:
382         * </p>
383         * 
384         * <pre>
385         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
386         *  sburgh, PA 15212":40.446816;80.00566
387         * </pre>
388         * 
389         * @return true if circumflex accent decoding is enabled, false if not
390         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
391         */
392        public boolean isCaretDecodingEnabled() {
393                return caretDecodingEnabled;
394        }
395
396        /**
397         * <p>
398         * Sets whether the reader will decode parameter values that use circumflex
399         * accent encoding (enabled by default). This escaping mechanism allows
400         * newlines and double quotes to be included in parameter values.
401         * </p>
402         * 
403         * <table class="simpleTable">
404         * <tr>
405         * <th>Raw Character</th>
406         * <th>Encoded Character</th>
407         * </tr>
408         * <tr>
409         * <td>{@code "}</td>
410         * <td>{@code ^'}</td>
411         * </tr>
412         * <tr>
413         * <td><i>newline</i></td>
414         * <td>{@code ^n}</td>
415         * </tr>
416         * <tr>
417         * <td>{@code ^}</td>
418         * <td>{@code ^^}</td>
419         * </tr>
420         * </table>
421         * 
422         * <p>
423         * Example:
424         * </p>
425         * 
426         * <pre>
427         * GEO;X-ADDRESS="Pittsburgh Pirates^n115 Federal St^nPitt
428         *  sburgh, PA 15212":geo:40.446816,-80.00566
429         * </pre>
430         * 
431         * @param enable true to use circumflex accent decoding, false not to
432         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
433         */
434        public void setCaretDecodingEnabled(boolean enable) {
435                caretDecodingEnabled = enable;
436        }

438        /**
439         * Defines how the reader should parse a vCard when it encounters a
440         * non-standard version number. By default, the reader throws a
441         * {@link VCardParseException} when this happens.
442         * @param version the version number
443         * @param parseAccordingTo the parsing rules the reader should use when a
444         * vCard with the given version number is encountered
445         */
446        public void setVersionAlias(String version, VCardVersion parseAccordingTo) {
447                versionAliases.put(version, parseAccordingTo);
448        }
449
450        /**
451         * Gets the character encoding of the reader.
452         * @return the character encoding or null if none is defined
453         */
454        public Charset getEncoding() {
455                if (reader instanceof InputStreamReader) {
456                        InputStreamReader isr = (InputStreamReader) reader;
457                        String charsetStr = isr.getEncoding();
458                        return (charsetStr == null) ? null : Charset.forName(charsetStr);
459                }
460
461                return null;
462        }
463
464        private int nextChar() throws IOException {
465                if (prevChar >= 0) {
466                        /*
467                         * Use the character that was left over from the previous invocation
468                         * of "readLine()".
469                         */
470                        int ch = prevChar;
471                        prevChar = -1;
472                        return ch;
473                }
474
475                return reader.read();
476        }
477
478        private boolean isNewline(char ch) {
479                return ch == '\n' || ch == '\r';
480        }
481
482        private boolean isWhitespace(char ch) {
483                return ch == ' ' || ch == '\t';
484        }
485
486        /**
487         * Determines if the property value is quoted-printable. This must be
488         * checked in order to account for the fact that some vCards fold
489         * quoted-printed lines in a non-standard way.
490         * @param parameters the property's parameters
491         * @return true if the property is quoted-printable, false if not
492         */
493        private boolean isQuotedPrintable(VCardParameters parameters) {
494                if (parameters.getEncoding() == Encoding.QUOTED_PRINTABLE) {
495                        return true;
496                }
497
498                List<String> namelessValues = parameters.get(null);
499                for (String value : namelessValues) {
500                        if ("QUOTED-PRINTABLE".equalsIgnoreCase(value)) {
501                                return true;
502                        }
503                }
504
505                return false;
506        }
507
508        /**
509         * Closes the underlying {@link Reader} object.
510         */
511        public void close() throws IOException {
512                reader.close();
513        }
