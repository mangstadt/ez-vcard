046/**
047 * Writes data to an vCard JSON data stream (jCard).
048 * @author Michael Angstadt
049 * @see <a href="http://tools.ietf.org/html/rfc7095">RFC 7095</a>
050 */
051public class JCardRawWriter implements Closeable, Flushable {
052        private final Writer writer;
053        private final boolean wrapInArray;
054        private JsonGenerator generator;
055        private boolean indent = false;
056        private boolean open = false;
057
058        /**
059         * @param writer the writer to wrap
060         * @param wrapInArray true to wrap everything in an array, false not to
061         * (useful when writing more than one vCard)
062         */
063        public JCardRawWriter(Writer writer, boolean wrapInArray) {
064                this.writer = writer;
065                this.wrapInArray = wrapInArray;
066        }
067
068        /**
069         * Gets whether or not the JSON will be pretty-printed.
070         * @return true if it will be pretty-printed, false if not (defaults to
071         * false)
072         */
073        public boolean isIndent() {
074                return indent;
075        }
076
077        /**
078         * Sets whether or not to pretty-print the JSON.
079         * @param indent true to pretty-print it, false not to (defaults to false)
080         */
081        public void setIndent(boolean indent) {
082                this.indent = indent;
083        }
084
085        /**
086         * Writes the beginning of a new "vcard" component.
087         * @throws IOException if there's an I/O problem
088         */
089        public void writeStartVCard() throws IOException {
090                if (generator == null) {
091                        init();
092                }
093
094                if (open) {
095                        writeEndVCard();
096                }
097
098                generator.writeStartArray();
099                indent(0);
100                generator.writeString("vcard");
101                generator.writeStartArray(); //start properties array
102
103                open = true;
104        }
105
106        /**
107         * Closes the "vcard" component array.
108         * @throws IllegalStateException if the component was never opened (
109         * {@link #writeStartVCard} must be called first)
110         * @throws IOException if there's an I/O problem
111         */
112        public void writeEndVCard() throws IOException {
113                if (!open) {
114                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(1));
115                }
116
117                generator.writeEndArray(); //end the properties array
118                generator.writeEndArray(); //end the "vcard" component array
119
120                open = false;
121        }
122
123        /**
124         * Writes a property to the current component.
125         * @param propertyName the property name (e.g. "version")
126         * @param dataType the data type or null for "unknown"
127         * @param value the property value
128         * @throws IllegalStateException if the "vcard" component was never opened
129         * or was just closed ({@link #writeStartVCard} must be called first)
130         * @throws IOException if there's an I/O problem
131         */
132        public void writeProperty(String propertyName, VCardDataType dataType, JCardValue value) throws IOException {
133                writeProperty(null, propertyName, new VCardParameters(), dataType, value);
134        }
135
136        /**
137         * Writes a property to the current vCard.
138         * @param group the group or null if there is no group
139         * @param propertyName the property name (e.g. "version")
140         * @param parameters the parameters
141         * @param dataType the data type or null for "unknown"
142         * @param value the property value
143         * @throws IllegalStateException if the "vcard" component was never opened
144         * or was just closed ({@link #writeStartVCard} must be called first)
145         * @throws IOException if there's an I/O problem
146         */
147        public void writeProperty(String group, String propertyName, VCardParameters parameters, VCardDataType dataType, JCardValue value) throws IOException {
148                if (!open) {
149                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(1));
150                }
151
152                generator.writeStartArray();
153                indent(2);
154
155                //write the property name
156                generator.writeString(propertyName);
157
158                //write parameters
159                generator.writeStartObject();
160                for (Map.Entry<String, List<String>> entry : parameters) {
161                        String name = entry.getKey().toLowerCase();
162                        List<String> values = entry.getValue();
163                        if (values.isEmpty()) {
164                                continue;
165                        }
166
167                        if (values.size() == 1) {
168                                generator.writeStringField(name, values.get(0));
169                        } else {
170                                generator.writeArrayFieldStart(name);
171                                for (String paramValue : values) {
172                                        generator.writeString(paramValue);
173                                }
174                                generator.writeEndArray();
175                        }
176                }
177
178                //write group
179                if (group != null) {
180                        generator.writeStringField("group", group);
181                }
182
183                generator.writeEndObject(); //end parameters object
184
185                //write data type
186                generator.writeString((dataType == null) ? "unknown" : dataType.getName().toLowerCase());
187
188                //write value
189                if (value.getValues().isEmpty()) {
190                        generator.writeString("");
191                } else {
192                        for (JsonValue jsonValue : value.getValues()) {
193                                writeValue(jsonValue);
194                        }
195                }
196
197                generator.writeEndArray();
198        }
199
200        private void writeValue(JsonValue jsonValue) throws IOException {
201                if (jsonValue.isNull()) {
202                        generator.writeNull();
203                        return;
204                }
205
206                Object val = jsonValue.getValue();
207                if (val != null) {
208                        if (val instanceof Byte) {
209                                generator.writeNumber((Byte) val);
210                        } else if (val instanceof Short) {
211                                generator.writeNumber((Short) val);
212                        } else if (val instanceof Integer) {
213                                generator.writeNumber((Integer) val);
214                        } else if (val instanceof Long) {
215                                generator.writeNumber((Long) val);
216                        } else if (val instanceof Float) {
217                                generator.writeNumber((Float) val);
218                        } else if (val instanceof Double) {
219                                generator.writeNumber((Double) val);
220                        } else if (val instanceof Boolean) {
221                                generator.writeBoolean((Boolean) val);
222                        } else {
223                                generator.writeString(val.toString());
224                        }
225                        return;
226                }
227
228                List<JsonValue> array = jsonValue.getArray();
229                if (array != null) {
230                        generator.writeStartArray();
231                        for (JsonValue element : array) {
232                                writeValue(element);
233                        }
234                        generator.writeEndArray();
235                        return;
236                }
237
238                Map<String, JsonValue> object = jsonValue.getObject();
239                if (object != null) {
240                        generator.writeStartObject();
241                        for (Map.Entry<String, JsonValue> entry : object.entrySet()) {
242                                generator.writeFieldName(entry.getKey());
243                                writeValue(entry.getValue());
244                        }
245                        generator.writeEndObject();
246                        return;
247                }
248        }
249
250        /**
251         * Checks to see if pretty-printing is enabled, and adds indentation
252         * whitespace if it is.
253         * @param spaces the number of spaces to indent with
254         * @throws IOException
255         */
256        private void indent(int spaces) throws IOException {
257                if (!indent) {
258                        return;
259                }
260
261                generator.writeRaw(NEWLINE);
262                generator.writeRaw(StringUtils.repeat(' ', spaces));
263        }
264
265        /**
266         * Flushes the JSON stream.
267         */
268        public void flush() throws IOException {
269                if (generator == null) {
270                        return;
271                }
272
273                generator.flush();
274        }
275
276        /**
277         * Finishes writing the JSON document so that it is syntactically correct.
278         * No more data can be written once this method is called.
279         * @throws IOException if there's a problem closing the stream
280         */
281        public void closeJsonStream() throws IOException {
282                if (generator == null) {
283                        return;
284                }
285
286                while (open) {
287                        writeEndVCard();
288                }
289
290                if (wrapInArray) {
291                        indent(0);
292                        generator.writeEndArray();
293                }
294
295                generator.close();
296        }
297
298        /**
299         * Finishes writing the JSON document and closes the underlying
300         * {@link Writer}.
301         * @throws IOException if there's a problem closing the stream
302         */
303        public void close() throws IOException {
304                if (generator == null) {
305                        return;
306                }
307
308                closeJsonStream();
309                writer.close();
310        }
311
312        private void init() throws IOException {
313                JsonFactory factory = new JsonFactory();
314                factory.configure(Feature.AUTO_CLOSE_TARGET, false);
315                generator = factory.createGenerator(writer);
316
317                if (wrapInArray) {
318                        generator.writeStartArray();
319                        indent(0);
320                }
321        }
322}
