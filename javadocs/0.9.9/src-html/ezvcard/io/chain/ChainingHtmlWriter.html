038/**
039 * Chainer class for writing hCards (HTML-encoded vCards).
040 * @see Ezvcard#writeHtml(Collection)
041 * @see Ezvcard#writeHtml(VCard...)
042 * @author Michael Angstadt
043 */
044public class ChainingHtmlWriter extends ChainingWriter<ChainingHtmlWriter> {
045        /**
046         * @param vcards the vCards to write
047         */
048        public ChainingHtmlWriter(Collection<VCard> vcards) {
049                super(vcards);
050        }
051
052        /**
053         * Writes the hCards to a string.
054         * @return the HTML page
055         */
056        public String go() {
057                return buildPage().write();
058        }
059
060        /**
061         * Writes the hCards to an output stream.
062         * @param out the output stream to write to
063         * @throws IOException if there's a problem writing to the output stream
064         */
065        public void go(OutputStream out) throws IOException {
066                buildPage().write(out);
067        }
068
069        /**
070         * Writes the hCards to a file.
071         * @param file the file to write to
072         * @throws IOException if there's a problem writing to the file
073         */
074        public void go(File file) throws IOException {
075                buildPage().write(file);
076        }
077
078        /**
079         * Writes the hCards to a writer.
080         * @param writer the writer to write to
081         * @throws IOException if there's a problem writing to the writer
082         */
083        public void go(Writer writer) throws IOException {
084                buildPage().write(writer);
085        }
086
087        private HCardPage buildPage() {
088                HCardPage page = new HCardPage();
089                for (VCard vcard : vcards) {
090                        page.add(vcard);
091                }
092                return page;
093        }
094}
