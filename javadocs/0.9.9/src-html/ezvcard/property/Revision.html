034/**
035 * <p>
036 * Defines the date that the vCard was last modified by its owner.
037 * </p>
038 * 
039 * <p>
040 * <b>Code sample</b>
041 * </p>
042 * 
043 * <pre class="brush:java">
044 * VCard vcard = new VCard();
045 * 
046 * Revision rev = new Revision(new Date());
047 * vcard.setRevision(rev);
048 * </pre>
049 * 
050 * <p>
051 * <b>Property name:</b> {@code REV}
052 * </p>
053 * <p>
054 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
055 * </p>
056 * @author Michael Angstadt
057 * @see <a href="http://tools.ietf.org/html/rfc6350#page-45">RFC 6350 p.45</a>
058 * @see <a href="http://tools.ietf.org/html/rfc2426#page-22">RFC 2426 p.22</a>
059 * @see <a href="http://www.imc.org/pdi/vcard-21.doc">vCard 2.1 p.19</a>
060 */
061public class Revision extends SimpleProperty<Date> {
062        /**
063         * Creates a revision property.
064         * @param date the date the vCard was last modified
065         */
066        public Revision(Date date) {
067                super(date);
068        }
069
070        /**
071         * Copy constructor.
072         * @param original the property to make a copy of
073         */
074        public Revision(Revision original) {
075                super(original);
076                value = (original.value == null) ? null : new Date(original.value.getTime());
077        }
078
079        /**
080         * Creates a revision property whose value is the current time.
081         * @return the property
082         */
083        public static Revision now() {
084                return new Revision(new Date());
085        }
086
087        @Override
088        public Revision copy() {
089                return new Revision(this);
090        }
091}
