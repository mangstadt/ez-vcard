047/**
048 * <p>
049 * Defines a mailing address.
050 * </p>
051 * 
052 * <p>
053 * <b>Code sample (creating)</b>
054 * </p>
055 * 
056 * <pre class="brush:java">
057 * VCard vcard = new VCard();
058 * 
059 * Address adr = new Address();
060 * adr.setStreetAddress(&quot;123 Main St.&quot;);
061 * adr.setLocality(&quot;Austin&quot;);
062 * adr.setRegion(&quot;TX&quot;);
063 * adr.setPostalCode(&quot;12345&quot;);
064 * adr.setCountry(&quot;USA&quot;);
065 * adr.addType(AddressType.WORK);
066 * 
067 * //optionally, set the text to print on the mailing label
068 * adr.setLabel(&quot;123 Main St.\nAustin, TX 12345\nUSA&quot;);
069 * 
070 * vcard.addAddress(adr);
071 * </pre>
072 * 
073 * <p>
074 * <b>Code sample (retrieving)</b>
075 * </p>
076 * 
077 * <pre class="brush:java">
078 * VCard vcard = ...
079 * for (Address adr : vcard.getAddresses()){
080 *   String street = adr.getStreetAddress();
081 *   String city = adr.getLocality();
082 *   //etc.
083 * }
084 * </pre>
085 * 
086 * <p>
087 * <b>Only part of the street address is being returned!</b>
088 * </p>
089 * <p>
090 * This usually means that the vCard you parsed contains unescaped comma
091 * characters. To get the full address, use the {@link #getStreetAddressFull}
092 * method.
093 * </p>
094 * 
095 * <p>
096 * <b>Property name:</b> {@code ADR}
097 * </p>
098 * <p>
099 * <b>Supported versions:</b> {@code 2.1, 3.0, 4.0}
100 * </p>
101 * @author Michael Angstadt
102 * @see <a href="http://tools.ietf.org/html/rfc6350#page-32">RFC 6350 p.32</a>
103 * @see <a href="http://tools.ietf.org/html/rfc2426#page-11">RFC 2426 p.11</a>
104 * @see <a href="http://www.imc.org/pdi/vcard-21.doc">vCard 2.1 p.11</a>
105 */

398        /**
399         * Gets all the TYPE parameters.
400         * @return the TYPE parameters or empty set if there are none
401         */
402        public Set<AddressType> getTypes() {
403                Set<String> values = parameters.getTypes();
404                Set<AddressType> types = new HashSet<AddressType>(values.size());
405                for (String value : values) {
406                        types.add(AddressType.get(value));
407                }
408                return types;
409        }

539        @Override
540        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
541                for (AddressType type : getTypes()) {
542                        if (type == AddressType.PREF) {
543                                //ignore because it is converted to a PREF parameter for 4.0 vCards
544                                continue;
545                        }
546
547                        if (!type.isSupportedBy(version)) {
548                                warnings.add(new Warning(9, type.getValue()));
549                        }
550                }
551        }

571        @Override
572        public int hashCode() {
573                final int prime = 31;
574                int result = super.hashCode();
575                result = prime * result + countries.hashCode();
576                result = prime * result + extendedAddresses.hashCode();
577                result = prime * result + localities.hashCode();
578                result = prime * result + poBoxes.hashCode();
579                result = prime * result + postalCodes.hashCode();
580                result = prime * result + regions.hashCode();
581                result = prime * result + streetAddresses.hashCode();
582                return result;
583        }

585        @Override
586        public boolean equals(Object obj) {
587                if (this == obj) return true;
588                if (!super.equals(obj)) return false;
589                Address other = (Address) obj;
590                if (!countries.equals(other.countries)) return false;
591                if (!extendedAddresses.equals(other.extendedAddresses)) return false;
592                if (!localities.equals(other.localities)) return false;
593                if (!poBoxes.equals(other.poBoxes)) return false;
594                if (!postalCodes.equals(other.postalCodes)) return false;
595                if (!regions.equals(other.regions)) return false;
596                if (!streetAddresses.equals(other.streetAddresses)) return false;
597                return true;
598        }

601                return list.isEmpty() ? null : list.get(0);
602        }
