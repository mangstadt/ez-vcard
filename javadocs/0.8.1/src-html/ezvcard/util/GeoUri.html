042    /**
043     * <p>
044     * Represents a "geo" URI.
045     * </p>
046     * <p>
047     * Example geo URI: <code>geo:12.341,56.784</code>
048     * </p>
049     * @author Michael Angstadt
050     * @see <a href="http://tools.ietf.org/html/rfc5870">RFC 5870</a>
051     */
052    public class GeoUri {
053            /**
054             * The coordinate reference system used by GPS (the default).
055             */
056            public static final String CRS_WGS84 = "wgs84";
057    
058            /**
059             * The non-alphanumeric characters which are allowed to exist inside of a
060             * parameter value.
061             */
062            private static final char validParamValueChars[] = "!$&'()*+-.:[]_~".toCharArray();
063            static {
064                    //make sure the array is sorted for binary search
065                    Arrays.sort(validParamValueChars);
066            }
067    
068            /**
069             * Finds hex values in a parameter value.
070             */
071            private static final Pattern hexPattern = Pattern.compile("(?i)%([0-9a-f]{2})");
072    
073            private static final String PARAM_CRS = "crs";
074            private static final String PARAM_UNCERTAINTY = "u";
075    
076            private Double coordA;
077            private Double coordB;
078            private Double coordC;
079            private String crs;
080            private Double uncertainty;
081            private Map<String, String> parameters = new LinkedHashMap<String, String>();
082    
083            public GeoUri() {
084                    //do nothing
085            }
086    
087            /**
088             * @param coordA the first coordinate (latitude, required)
089             * @param coordB the second coordinate (longitude, required)
090             * @param coordC the third coordinate (altitude, optional)
091             * @param crs the coordinate system (optional, defaults to WGS-84)
092             * @param uncertainty the accuracy of the coordinates (in meters, optional)
093             */
094            public GeoUri(Double coordA, Double coordB, Double coordC, String crs, Double uncertainty) {
095                    this.coordA = coordA;
096                    this.coordB = coordB;
097                    this.coordC = coordC;
098                    this.crs = crs;
099                    this.uncertainty = uncertainty;
100            }
101    
102            /**
103             * Parses a geo URI string.
104             * @param uri the URI string
105             * @throws IllegalArgumentException if the string is not a valid geo URI
106             */
107            public GeoUri(String uri) {
108                    Pattern p = Pattern.compile("(?i)^geo:(-?\\d+(\\.\\d+)?),(-?\\d+(\\.\\d+)?)(,(-?\\d+(\\.\\d+)?))?(;(.*))?$");
109                    Matcher m = p.matcher(uri);
110                    if (m.find()) {
111                            coordA = Double.parseDouble(m.group(1));
112                            coordB = Double.parseDouble(m.group(3));
113    
114                            String coordCStr = m.group(6);
115                            if (coordCStr != null) {
116                                    coordC = Double.valueOf(coordCStr);
117                            }
118    
119                            String paramsStr = m.group(9);
120                            if (paramsStr != null) {
121                                    String paramsArray[] = paramsStr.split(";");
122    
123                                    for (String param : paramsArray) {
124                                            String paramSplit[] = param.split("=", 2);
125                                            String paramName = paramSplit[0];
126                                            String paramValue = paramSplit.length > 1 ? paramSplit[1] : "";
127                                            if (PARAM_CRS.equalsIgnoreCase(paramName)) {
128                                                    crs = paramValue;
129                                            } else if (PARAM_UNCERTAINTY.equalsIgnoreCase(paramName)) {
130                                                    uncertainty = Double.valueOf(paramValue);
131                                            } else {
132                                                    paramValue = decodeParamValue(paramValue);
133                                                    parameters.put(paramName, paramValue);
134                                            }
135                                    }
136                            }
137                    } else {
138                            throw new IllegalArgumentException("Invalid geo URI: " + uri);
139                    }
140            }

195            public String getCrs() {
196                    return crs;
197            }
198    
199            /**
200             * Sets the coordinate reference system.
201             * @param crs the coordinate reference system (can only contain letters,
202             * numbers, and hyphens) or null to use the default (WGS-84)
203             * @throws IllegalArgumentException if the CRS name contains invalid
204             * characters
205             */
206            public void setCrs(String crs) {
207                    if (crs != null && !isLabelText(crs)) {
208                            throw new IllegalArgumentException("CRS can only contain letters, numbers, and hypens.");
209                    }
210                    this.crs = crs;
211            }
212    
213            /**
214             * Gets the uncertainty (how accurate the coordinates are).
215             * @return the uncertainty (in meters) or null if not set
216             */
217            public Double getUncertainty() {
218                    return uncertainty;
219            }
220    
221            /**
222             * Sets the uncertainty (how accurate the coordinates are).
223             * @param uncertainty the uncertainty (in meters) or null to remove
224             */
225            public void setUncertainty(Double uncertainty) {
226                    this.uncertainty = uncertainty;
227            }

229            /**
230             * Adds a parameter.
231             * @param name the parameter name (can only contain letters, numbers, and
232             * hyphens)
233             * @param value the parameter value
234             * @throws IllegalArgumentException if the parameter name contains invalid
235             * characters
236             */
237            public void addParameter(String name, String value) {
238                    if (!isLabelText(name)) {
239                            throw new IllegalArgumentException("Parameter names can only contain letters, numbers, and hyphens.");
240                    }
241    
242                    if (value == null) {
243                            value = "";
244                    }
245    
246                    parameters.put(name, value);
247            }
248    
249            /**
250             * Removes a parameter.
251             * @param name the name of the parameter to remove
252             */
253            public void removeParameter(String name) {
254                    parameters.remove(name);
255            }

258             * Gets a parameter value.
259             * @param name the parameter name
260             * @return the parameter value or null if not found
261             */
262            public String getParameter(String name) {
263                    return parameters.get(name);
264            }
265    
266            /**
267             * Gets all the parameters.
268             * @return all the parameters
269             */
270            public Map<String, String> getParameters() {
271                    return new LinkedHashMap<String, String>(parameters);
272            }

288                    return coordA != null && coordB != null;
289            }

291            /**
292             * Converts this geo URI to its string representation.
293             * @return the geo URI's string representation
294             */
295            @Override
296            public String toString() {
297                    return toString(6);
298            }
299    
300            /**
301             * Converts this geo URI to its string representation.
302             * @param decimals the number of decimals to display for floating point
303             * values
304             * @return the geo URI's string representation
305             */
306            public String toString(int decimals) {
307                    NumberFormat nf = buildNumberFormat(decimals);
308                    StringBuilder sb = new StringBuilder("geo:");
309    
310                    if (coordA != null) {
311                            sb.append(nf.format(coordA));
312                    }
313    
314                    sb.append(',');
315                    if (coordB != null) {
316                            sb.append(nf.format(coordB));
317                    }
318    
319                    if (coordC != null) {
320                            sb.append(',');
321                            sb.append(coordC);
322                    }
323    
324                    //if the CRS is WGS-84, then it doesn't have to be displayed
325                    if (crs != null && !crs.equalsIgnoreCase(CRS_WGS84)) {
326                            sb.append(';').append(PARAM_CRS).append('=');
327                            sb.append(crs);
328                    }
329    
330                    if (uncertainty != null) {
331                            sb.append(';').append(PARAM_UNCERTAINTY).append('=');
332                            sb.append(nf.format(uncertainty));
333                    }
334    
335                    for (Map.Entry<String, String> entry : parameters.entrySet()) {
336                            String name = entry.getKey();
337                            String value = entry.getValue();
338                            sb.append(';');
339                            sb.append(name); //note: the param name is validated in "addParameter()"
340                            sb.append('=');
341                            sb.append(encodeParamValue(value));
342                    }
343    
344                    return sb.toString();
345            }
346    
347            private boolean isLabelText(String text) {
348                    return text.matches("(?i)[-a-z0-9]+");
349            }
350    
351            private String encodeParamValue(String value) {
352                    StringBuilder sb = new StringBuilder(value.length());
353                    for (char c : value.toCharArray()) {
354                            if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9') || Arrays.binarySearch(validParamValueChars, c) >= 0) {
355                                    sb.append(c);
356                            } else {
357                                    int i = (int) c;
358                                    sb.append('%');
359                                    sb.append(Integer.toString(i, 16));
360                            }
361                    }
362                    return sb.toString();
363            }
364    
365            private String decodeParamValue(String value) {
366                    Matcher m = hexPattern.matcher(value);
367                    StringBuffer sb = new StringBuffer();
368                    while (m.find()) {
369                            int hex = Integer.parseInt(m.group(1), 16);
370                            m.appendReplacement(sb, "" + (char) hex);
371                    }
372                    m.appendTail(sb);
373                    return sb.toString();
374            }
375    
376            /**
377             * Builds a number formatter object for displaying latitudes/longitudes as
378             * strings.
379             * @param decimals the number of decimals to display
380             * @return the number formatter
381             */
382            public static NumberFormat buildNumberFormat(int decimals) {
383                    StringBuilder sb = new StringBuilder();
384                    sb.append('0');
385                    if (decimals > 0) {
386                            sb.append('.');
387                            for (int i = 0; i < decimals; i++) {
388                                    sb.append('#');
389                            }
390                    }
391    
392                    DecimalFormat df = new DecimalFormat(sb.toString());
393    
394                    //decimal separator differs by locale (e.g. Germany uses ",")
395                    DecimalFormatSymbols symbols = new DecimalFormatSymbols();
396                    symbols.setDecimalSeparator('.');
397                    symbols.setMinusSign('-');
398                    df.setDecimalFormatSymbols(symbols);
399    
400                    return df;
401            }
