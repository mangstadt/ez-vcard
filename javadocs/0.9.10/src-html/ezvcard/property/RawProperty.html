042/**
043 * Holds the property value as-is. No escaping or unescaping is done on the
044 * value.
045 * @author Michael Angstadt
046 */
047public class RawProperty extends TextProperty {
048        private String propertyName;
049        private VCardDataType dataType;
050
051        /**
052         * Creates a raw property.
053         * @param propertyName the property name (e.g. "X-GENDER")
054         * @param value the property value
055         */
056        public RawProperty(String propertyName, String value) {
057                this(propertyName, value, null);
058        }
059
060        /**
061         * Creates a raw property.
062         * @param propertyName the property name (e.g. "X-GENDER")
063         * @param value the property value
064         * @param dataType the value's data type
065         */
066        public RawProperty(String propertyName, String value, VCardDataType dataType) {
067                super(value);
068                this.propertyName = propertyName;
069                this.dataType = dataType;
070        }
071
072        /**
073         * Copy constructor.
074         * @param original the property to make a copy of
075         */
076        public RawProperty(RawProperty original) {
077                super(original);
078                propertyName = original.propertyName;
079                dataType = original.dataType;
080        }
081
082        /**
083         * Gets the name of the property.
084         * @return the property name
085         */
086        public String getPropertyName() {
087                return propertyName;
088        }
089
090        /**
091         * Sets the name of the property.
092         * @param propertyName the property name
093         */
094        public void setPropertyName(String propertyName) {
095                this.propertyName = propertyName;
096        }
097
098        /**
099         * Gets the data type of the property's value.
100         * @return the data type or null if unknown
101         */
102        public VCardDataType getDataType() {
103                return dataType;
104        }
105
106        /**
107         * Sets the data type of the property's value.
108         * @param dataType the data type or null if unknown
109         */
110        public void setDataType(VCardDataType dataType) {
111                this.dataType = dataType;
112        }
113
114        @Override
115        protected void _validate(List<Warning> warnings, VCardVersion version, VCard vcard) {
116                CharacterBitSet validCharacters = new CharacterBitSet("-a-zA-Z0-9");
117                if (!validCharacters.containsOnly(propertyName)) {
118                        warnings.add(new Warning(24, propertyName));
119                }
120        }
121
122        @Override
123        protected Map<String, Object> toStringValues() {
124                Map<String, Object> values = new LinkedHashMap<String, Object>();
125                values.put("propertyName", propertyName);
126                values.put("dataType", dataType);
127                values.put("value", value);
128                return values;
129        }
130
131        @Override
132        public RawProperty copy() {
133                return new RawProperty(this);
134        }
135
136        @Override
137        public int hashCode() {
138                final int prime = 31;
139                int result = super.hashCode();
140                result = prime * result + ((dataType == null) ? 0 : dataType.hashCode());
141                result = prime * result + ((propertyName == null) ? 0 : propertyName.toLowerCase().hashCode());
142                return result;
143        }
144
145        @Override
146        public boolean equals(Object obj) {
147                if (this == obj) return true;
148                if (!super.equals(obj)) return false;
149                RawProperty other = (RawProperty) obj;
150                if (dataType == null) {
151                        if (other.dataType != null) return false;
152                } else if (!dataType.equals(other.dataType)) return false;
153                if (propertyName == null) {
154                        if (other.propertyName != null) return false;
155                } else if (!propertyName.equalsIgnoreCase(other.propertyName)) return false;
156                return true;
157        }
158}
