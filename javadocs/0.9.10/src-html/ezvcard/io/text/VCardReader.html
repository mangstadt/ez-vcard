067/**
068 * <p>
069 * Parses {@link VCard} objects from a plain-text vCard data stream.
070 * </p>
071 * <p>
072 * <b>Example:</b>
073 * </p>
074 * 
075 * <pre class="brush:java">
076 * File file = new File("vcards.vcf");
077 * VCardReader reader = null;
078 * try {
079 *   reader = new VCardReader(file);
080 *   VCard vcard;
081 *   while ((vcard = reader.readNext()) != null){
082 *     ...
083 *   }
084 * } finally {
085 *   if (reader != null) reader.close();
086 * }
087 * </pre>
088 * @author Michael Angstadt
089 * @see <a href="http://www.imc.org/pdi/vcard-21.rtf">vCard 2.1</a>
090 * @see <a href="http://tools.ietf.org/html/rfc2426">RFC 2426 (3.0)</a>
091 * @see <a href="http://tools.ietf.org/html/rfc6350">RFC 6350 (4.0)</a>
092 */
093public class VCardReader extends StreamReader {
094        private final VCardRawReader reader;
095        private Charset defaultQuotedPrintableCharset;
096
097        /**
098         * @param str the string to read from
099         */
100        public VCardReader(String str) {
101                this(new StringReader(str));
102        }

105         * @param in the input stream to read from

107        public VCardReader(InputStream in) {
108                this(new InputStreamReader(in));

119        /**
120         * @param reader the reader to read from
121         */
122        public VCardReader(Reader reader) {
123                this.reader = new VCardRawReader(reader);
124                defaultQuotedPrintableCharset = this.reader.getEncoding();
125                if (defaultQuotedPrintableCharset == null) {
126                        defaultQuotedPrintableCharset = Charset.defaultCharset();
127                }
128        }

198        @Override
199        protected VCard _readNext() throws IOException {
200                VCard root = null;
201                LinkedList<VCard> vcardStack = new LinkedList<VCard>();
202                LinkedList<List<Label>> labelStack = new LinkedList<List<Label>>();
203                EmbeddedVCardException embeddedVCardException = null;
204                while (true) {
205                        //read next line
206                        VCardRawLine line;
207                        try {
208                                line = reader.readLine();
209                        } catch (VCardParseException e) {
210                                if (!vcardStack.isEmpty()) {
211                                        warnings.add(e.getLineNumber(), null, 27, e.getMessage(), e.getLine());
212                                }
213                                continue;
214                        }
215
216                        //EOF
217                        if (line == null) {
218                                break;
219                        }
220
221                        //handle BEGIN:VCARD
222                        if ("BEGIN".equalsIgnoreCase(line.getName()) && "VCARD".equalsIgnoreCase(line.getValue())) {
223                                VCard vcard = new VCard();
224                                vcard.setVersion(reader.getVersion());
225                                vcardStack.add(vcard);
226                                labelStack.add(new ArrayList<Label>());
227
228                                if (root == null) {
229                                        root = vcard;
230                                }
231
232                                if (embeddedVCardException != null) {
233                                        embeddedVCardException.injectVCard(vcard);
234                                        embeddedVCardException = null;
235                                }
236
237                                continue;
238                        }
239
240                        if (vcardStack.isEmpty()) {
241                                //BEGIN component hasn't been encountered yet, so skip this line
242                                continue;
243                        }
244
245                        //handle VERSION property
246                        if ("VERSION".equalsIgnoreCase(line.getName())) {
247                                vcardStack.getLast().setVersion(reader.getVersion());
248                                continue;
249                        }
250
251                        //handle END:VCARD
252                        if ("END".equalsIgnoreCase(line.getName()) && "VCARD".equalsIgnoreCase(line.getValue())) {
253                                VCard curVCard = vcardStack.removeLast();
254                                List<Label> labels = labelStack.removeLast();
255                                assignLabels(curVCard, labels);
256
257                                if (vcardStack.isEmpty()) {
258                                        //done reading the vCard
259                                        break;
260                                }
261                                continue;
262                        }
263
264                        //handle property
265                        {
266                                String group = line.getGroup();
267                                VCardParameters parameters = line.getParameters();
268                                String name = line.getName();
269                                String value = line.getValue();
270
271                                if (embeddedVCardException != null) {
272                                        //the next property was supposed to be the start of a nested vCard, but it wasn't
273                                        embeddedVCardException.injectVCard(null);
274                                        embeddedVCardException = null;
275                                }
276
277                                VCard curVCard = vcardStack.getLast();
278                                VCardVersion version = curVCard.getVersion();
279
280                                //sanitize the parameters
281                                processNamelessParameters(parameters);
282                                processQuotedMultivaluedTypeParams(parameters);
283
284                                //decode property value from quoted-printable
285                                try {
286                                        value = decodeQuotedPrintableValue(name, parameters, value);
287                                } catch (DecoderException e) {
288                                        warnings.add(reader.getLineNumber(), name, 38, e.getMessage());
289                                }
290
291                                //get the scribe
292                                VCardPropertyScribe<? extends VCardProperty> scribe = index.getPropertyScribe(name);
293                                if (scribe == null) {
294                                        scribe = new RawPropertyScribe(name);
295                                }
296
297                                //get the data type (VALUE parameter)
298                                VCardDataType dataType = parameters.getValue();
299                                if (dataType == null) {
300                                        //use the default data type if there is no VALUE parameter
301                                        dataType = scribe.defaultDataType(version);
302                                } else {
303                                        //remove VALUE parameter if it is set
304                                        parameters.setValue(null);
305                                }
306
307                                VCardProperty property;
308                                try {
309                                        Result<? extends VCardProperty> result = scribe.parseText(value, dataType, version, parameters);
310
311                                        for (String warning : result.getWarnings()) {
312                                                warnings.add(reader.getLineNumber(), name, warning);
313                                        }
314
315                                        property = result.getProperty();
316                                        property.setGroup(group);
317
318                                        if (property instanceof Label) {
319                                                /*
320                                                 * LABEL properties must be treated specially so they
321                                                 * can be matched up with the ADR properties that they
322                                                 * belong to.
323                                                 */
324                                                Label label = (Label) property;
325                                                labelStack.getLast().add(label);
326                                        } else {
327                                                curVCard.addProperty(property);
328                                        }
329                                } catch (SkipMeException e) {
330                                        warnings.add(reader.getLineNumber(), name, 22, e.getMessage());
331                                } catch (CannotParseException e) {
332                                        warnings.add(reader.getLineNumber(), name, 25, value, e.getMessage());
333                                        property = new RawProperty(name, value);
334                                        property.setGroup(group);
335                                        curVCard.addProperty(property);
336                                } catch (EmbeddedVCardException e) {
337                                        //parse an embedded vCard (i.e. the AGENT type)
338                                        property = e.getProperty();
339
340                                        if (value.length() == 0 || version == VCardVersion.V2_1) {
341                                                //a nested vCard is expected to be next (2.1 style)
342                                                embeddedVCardException = e;
343                                        } else {
344                                                //the property value should be an embedded vCard (3.0 style)
345                                                value = VCardPropertyScribe.unescape(value);
346
347                                                VCardReader agentReader = new VCardReader(value);
348                                                agentReader.setScribeIndex(index);
349                                                try {
350                                                        VCard nestedVCard = agentReader.readNext();
351                                                        if (nestedVCard != null) {
352                                                                e.injectVCard(nestedVCard);
353                                                        }
354                                                } catch (IOException e2) {
355                                                        //shouldn't be thrown because we're reading from a string
356                                                } finally {
357                                                        for (String w : agentReader.getWarnings()) {
358                                                                warnings.add(reader.getLineNumber(), name, 26, w);
359                                                        }
360                                                        IOUtils.closeQuietly(agentReader);
361                                                }
362                                        }
363
364                                        curVCard.addProperty(property);
365                                }
366                        }
367                }
368
369                return root;
370        }

372        /**
373         * Assigns names to all nameless parameters. v3.0 and v4.0 requires all
374         * parameters to have names, but v2.1 does not.
375         * @param parameters the parameters
376         */
377        private void processNamelessParameters(VCardParameters parameters) {
378                List<String> namelessParamValues = parameters.removeAll(null);
379                for (String paramValue : namelessParamValues) {
380                        String paramName = guessParameterName(paramValue);
381                        parameters.put(paramName, paramValue);
382                }
383        }
384
385        /**
386         * Makes a guess as to what a parameter value's name should be.
387         * @param value the parameter value
388         * @return the guessed name
389         */
390        private String guessParameterName(String value) {
391                if (VCardDataType.find(value) != null) {
392                        return VCardParameters.VALUE;
393                }
394
395                if (Encoding.find(value) != null) {
396                        return VCardParameters.ENCODING;
397                }
398
399                //otherwise, assume it's a TYPE
400                return VCardParameters.TYPE;
401        }

403        /**
404         * <p>
405         * Accounts for multi-valued TYPE parameters being enclosed entirely in
406         * double quotes (for example: ADR;TYPE="home,work").
407         * </p>
408         * <p>
409         * Many examples throughout the 4.0 specs show TYPE parameters being encoded
410         * in this way. This conflicts with the ABNF and is noted in the errata.
411         * This method will parse these incorrectly-formatted TYPE parameters as if
412         * they were multi-valued, even though, technically, they are not.
413         * </p>
414         * @param parameters the parameters
415         */
416        private void processQuotedMultivaluedTypeParams(VCardParameters parameters) {
417                List<String> types = parameters.getTypes();
418                String valueWithComma = null;
419                for (String value : types) {
420                        if (value.indexOf(',') >= 0) {
421                                valueWithComma = value;
422                                break;
423                        }
424                }
425                if (valueWithComma == null) {
426                        return;
427                }
428
429                types.clear();
430                int prev = -1, cur;
431                while ((cur = valueWithComma.indexOf(',', prev + 1)) >= 0) {
432                        types.add(valueWithComma.substring(prev + 1, cur));
433                        prev = cur;
434                }
435                types.add(valueWithComma.substring(prev + 1));
436        }

438        /**
439         * Checks to see if a property's value is encoded in quoted-printable
440         * encoding and decodes it if it is.
441         * @param name the property name
442         * @param parameters the property parameters
443         * @param value the property value (may or may not be encoded in
444         * quoted-printable
445         * @return the decoded property value or the untouched property value if it
446         * is not encoded in quoted-printable encoding
447         * @throws DecoderException if the value couldn't be decoded
448         */
449        private String decodeQuotedPrintableValue(String name, VCardParameters parameters, String value) throws DecoderException {
450                if (parameters.getEncoding() != Encoding.QUOTED_PRINTABLE) {
451                        //the property value is not encoded in quoted-printable encoding
452                        return value;
453                }
454
455                //remove the encoding parameter
456                parameters.setEncoding(null);
457
458                //determine the character set
459                Charset charset = null;
460                String charsetStr = parameters.getCharset();
461                if (charsetStr == null) {
462                        charset = defaultQuotedPrintableCharset;
463                } else {
464                        try {
465                                charset = Charset.forName(charsetStr);
466                        } catch (IllegalCharsetNameException e) {
467                                //bad charset name
468                        } catch (UnsupportedCharsetException e) {
469                                //bad charset name
470                        }
471
472                        if (charset == null) {
473                                charset = defaultQuotedPrintableCharset;
474
475                                //the given charset was invalid, so add a warning
476                                warnings.add(reader.getLineNumber(), name, 23, charsetStr, charset.name());
477                        }
478                }
479
480                QuotedPrintableCodec codec = new QuotedPrintableCodec(charset.name());
481                return codec.decode(value);
482        }

488                reader.close();
