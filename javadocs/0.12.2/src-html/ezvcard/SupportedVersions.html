035/**
036 * <p>
037 * Defines the vCard version(s) that support a property ({@link VCardProperty}
038 * class), parameter value ({@link VCardParameter} instance), or data type (
039 * {@link VCardDataType} instance). If this annotation is not present, then all
040 * versions are presumed to be supported.
041 * </p>
042 * <p>
043 * <b>Examples</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * {@literal @}SupportedVersions({ VCardVersion.V2_1, VCardVersion.V3_0 })
048 * public class Agent extends VCardProperty {
049 *   //...
050 * }
051 * 
052 * public class AddressType extends VCardParameter {
053 *   {@literal @}SupportedVersions({ VCardVersion.V2_1, VCardVersion.V3_0 })
054 *   public static final AddressType DOM = new AddressType("dom");
055 *   //...
056 * }
057 * 
058 * public class VCardDataType {
059 *   {@literal @}SupportedVersions(VCardVersion.V2_1)
060 *   public static final VCardDataType CONTENT_ID = new VCardDataType("content-id");
061 *   //...
062 * }
063 * </pre>
064 * @author Michael Angstadt
065 */
066@Documented
067@Retention(RetentionPolicy.RUNTIME)
068public @interface SupportedVersions {
069        VCardVersion[] value();
070}
